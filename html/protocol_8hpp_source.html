121static inline id_e get_command(byte_t _byte) {
122
123 id_e its_id(id_e::UNKNOWN_ID);
124 if (_byte <= static_cast<byte_t>(id_e::SUSPEND_ID))
125 its_id = static_cast<id_e>(_byte);

127}
