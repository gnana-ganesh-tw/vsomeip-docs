49

78void local_tcp_client_endpoint_impl::stop() {
79 {
80 std::lock_guard<std::recursive_mutex> its_lock(mutex_);
81 sending_blocked_ = true;
82 }
83 {
84 std::lock_guard<std::mutex> its_lock(connect_timer_mutex_);
85 boost::system::error_code ec;
86 connect_timer_.cancel(ec);
87 }
88 connect_timeout_ = VSOMEIP_DEFAULT_CONNECT_TIMEOUT;
89
90 bool is_open(false);
91 {
92 std::lock_guard<std::mutex> its_lock(socket_mutex_);
93 is_open = socket_->is_open();
94 }
95 if (is_open) {
96 bool send_queue_empty(false);
97 std::uint32_t times_slept(0);
98
99 while (times_slept <= 50) {
100 mutex_.lock();
101 send_queue_empty = (queue_.size() == 0);
102 mutex_.unlock();
103 if (send_queue_empty) {
104 break;
105 } else {
106 std::this_thread::sleep_for(std::chrono::milliseconds(10));
107 times_slept++;
108 }
109 }
110 }
111 shutdown_and_close_socket(false);
112}

114void local_tcp_client_endpoint_impl::connect() {
115 boost::system::error_code its_connect_error;
116 std::lock_guard<std::mutex> its_lock(socket_mutex_);
117 boost::system::error_code its_error;
118 socket_->open(remote_.protocol(), its_error);
119 if (!its_error || its_error == boost::asio::error::already_open) {
120 socket_->set_option(boost::asio::socket_base::reuse_address(true), its_error);
121 if (its_error) {
122 VSOMEIP_WARNING << "local_tcp_client_endpoint_impl::" << __func__
123 << ": Cannot enable SO_REUSEADDR"
124 << "(" << its_error.message() << ")";
125 }
126 socket_->bind(local_, its_error);
127 if (its_error) {
128 VSOMEIP_WARNING << "local_tcp_client_endpoint_impl::" << __func__
129 << ": Cannot bind to client port " << local_.port()
130 << "(" << its_error.message() << ")";
131 }
132 state_ = cei_state_e::CONNECTING;
133 start_connecting_timer();
134 socket_->async_connect(
135 remote_,
136 strand_.wrap(
137 std::bind(
138 &local_tcp_client_endpoint_impl::cancel_and_connect_cbk,
139 shared_from_this(),
140 std::placeholders::_1
141 )
142 )
143 );
144 } else {
145 VSOMEIP_WARNING << "local_client_endpoint::connect: Error opening socket: "
146 << its_error.message() << " (" << std::dec << its_error.value() << ")";
147 its_connect_error = its_error;
148 try {
149 strand_.post(
150 std::bind(&client_endpoint_impl::connect_cbk, shared_from_this(),
151 its_connect_error));
152 } catch (const std::exception &e) {
153 VSOMEIP_ERROR << "local_client_endpoint_impl::connect: " << e.what();
154 }
155 }

203void local_tcp_client_endpoint_impl::send_queued(std::pair<message_buffer_ptr_t, uint32_t> &_entry) {
204
205 static const byte_t its_start_tag[] = { 0x67, 0x37, 0x6D, 0x07 };
206 static const byte_t its_end_tag[] = { 0x07, 0x6D, 0x37, 0x67 };
207 std::vector<boost::asio::const_buffer> bufs;
208
209 bufs.push_back(boost::asio::buffer(its_start_tag));
210 bufs.push_back(boost::asio::buffer(*_entry.first));
211 bufs.push_back(boost::asio::buffer(its_end_tag));
212
213 {
214 std::lock_guard<std::mutex> its_lock(socket_mutex_);

247void local_tcp_client_endpoint_impl::receive_cbk(
248 boost::system::error_code const &_error, std::size_t _bytes) {
249
250 if (_error) {
251 VSOMEIP_INFO << "local_tcp_client_endpoint_impl::" << __func__ << " Error: " << _error.message();
252 if (_error == boost::asio::error::operation_aborted) {
253 // endpoint was stopped
254 return;
255 }
256 error_handler_t handler;
257 {
258 std::lock_guard<std::mutex> its_lock(error_handler_mutex_);
259 handler = error_handler_;
260 }
261 if (handler)
262 handler();
263 } else {
264
265#if 0
266 std::stringstream msg;
267 msg << "lce<" << this << ">::recv: ";
268 for (std::size_t i = 0; i < recv_buffer_.size(); i++)
269 msg << std::setw(2) << std::setfill('0') << std::hex
270 << (int)recv_buffer_[i] << " ";
271 VSOMEIP_INFO << msg.str();
272#endif
273
274 // We only handle a single message here. Check whether the message
275 // format matches what we do expect.
276 // TODO: Replace the magic numbers.
277 if (_bytes == VSOMEIP_LOCAL_CLIENT_ENDPOINT_RECV_BUFFER_SIZE
278 && recv_buffer_[0] == 0x67 && recv_buffer_[1] == 0x37
279 && recv_buffer_[2] == 0x6d && recv_buffer_[3] == 0x07
280 && recv_buffer_[4] == byte_t(protocol::id_e::ASSIGN_CLIENT_ACK_ID)
281 && recv_buffer_[15] == 0x07 && recv_buffer_[16] == 0x6d
282 && recv_buffer_[17] == 0x37 && recv_buffer_[18] == 0x67) {
283
284 auto its_routing_host = routing_host_.lock();
285 if (its_routing_host)
286 its_routing_host->on_message(&recv_buffer_[4],
287 static_cast<length_t>(recv_buffer_.size() - 8), this);
288 }
289
290 receive();
291 }

334bool local_tcp_client_endpoint_impl::check_packetizer_space(std::uint32_t _size) {
335 if (train_->buffer_->size() + _size < train_->buffer_->size()) {
336 VSOMEIP_ERROR << "Overflow in packetizer addition ~> abort sending!";
337 return false;
338 }
339 if (train_->buffer_->size() + _size > max_message_size_
340 && !train_->buffer_->empty()) {
341 queue_.push_back(std::make_pair(train_->buffer_, 0));
342 queue_size_ += train_->buffer_->size();
343 train_->buffer_ = std::make_shared<message_buffer_t>();
344 }
345 return true;
346}
