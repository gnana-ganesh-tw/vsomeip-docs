34udp_server_endpoint_impl::udp_server_endpoint_impl(
35 const std::shared_ptr<endpoint_host>& _endpoint_host,
36 const std::shared_ptr<routing_host>& _routing_host,
37 const endpoint_type& _local,
38 boost::asio::io_context &_io,
39 const std::shared_ptr<configuration>& _configuration) :
40#if VSOMEIP_BOOST_VERSION >= 106600
41 server_endpoint_impl<ip::udp>(
42#else
43 server_endpoint_impl<ip::udp_ext>(
44#endif
45 _endpoint_host, _routing_host, _local,
46 _io, VSOMEIP_MAX_UDP_MESSAGE_SIZE,
47 _configuration->get_endpoint_queue_limit(_configuration->get_unicast_address().to_string(), _local.port()),
48 _configuration),
49 unicast_socket_(_io, _local.protocol()),
50 unicast_recv_buffer_(VSOMEIP_MAX_UDP_MESSAGE_SIZE, 0),
51 is_v4_(false),
52 multicast_id_(0),
53 joined_group_(false),
54 netmask_(_configuration->get_netmask()),
55 prefix_(_configuration->get_prefix()),
56 local_port_(_local.port()),
57 tp_reassembler_(std::make_shared<tp::tp_reassembler>(_configuration->get_max_message_size_unreliable(), _io)),
58 tp_cleanup_timer_(_io),
59 is_stopped_(true) {
60 is_supporting_someip_tp_ = true;
61
62 boost::system::error_code ec;
63
64 boost::asio::socket_base::reuse_address optionReuseAddress(true);
65 unicast_socket_.set_option(optionReuseAddress, ec);
66 if (ec)
67 VSOMEIP_ERROR << __func__
68 << ": set reuse address option failed (" << ec.message() << ")";
69
70#if defined(__linux__) || defined(ANDROID) || defined(__QNX__)
71 // If specified, bind to device
72 std::string its_device(configuration_->get_device());
73 if (its_device != "") {
74 if (setsockopt(unicast_socket_.native_handle(),
75 SOL_SOCKET, SO_BINDTODEVICE, its_device.c_str(), (socklen_t)its_device.size()) == -1) {
76 VSOMEIP_WARNING << "UDP Server: Could not bind to device \"" << its_device << "\"";
77 }
78 }
79#endif
80
81 unicast_socket_.bind(_local, ec);
82 if (ec)
83 VSOMEIP_ERROR << __func__
84 << ": bind failed (" << ec.message() << ")";
85
86 if (local_.address().is_v4()) {
87 is_v4_ = true;
88 boost::asio::ip::multicast::outbound_interface option(_local.address().to_v4());
89 unicast_socket_.set_option(option, ec);
90 if (ec)
91 VSOMEIP_ERROR << __func__
92 << ": set IPv4 outbound interface option failed (" << ec.message() << ")";
93 } else {
94 boost::asio::ip::multicast::outbound_interface option(
95 static_cast<unsigned int>(local_.address().to_v6().scope_id()));
96 unicast_socket_.set_option(option, ec);
97 if (ec)
98 VSOMEIP_ERROR << __func__
99 << ": set IPv6 outbound interface option failed (" << ec.message() << ")";
100 }
101
102 boost::asio::socket_base::broadcast option(true);
103 unicast_socket_.set_option(option, ec);
104 if (ec)
105 VSOMEIP_ERROR << __func__
106 << ": set broadcast option failed (" << ec.message() << ")";
107
108 const int its_udp_recv_buffer_size =
109 configuration_->get_udp_receive_buffer_size();
110 unicast_socket_.set_option(boost::asio::socket_base::receive_buffer_size(
111 its_udp_recv_buffer_size), ec);
112
113 if (ec) {
114 VSOMEIP_WARNING << "udp_server_endpoint_impl: couldn't set "
115 << "SO_RCVBUF: " << ec.message() << " to: " << std::dec
116 << its_udp_recv_buffer_size << " local port: " << std::dec
117 << local_port_;
118 }
119
120 boost::asio::socket_base::receive_buffer_size its_option;
121 unicast_socket_.get_option(its_option, ec);
122#ifdef __linux__
123 // If regular setting of the buffer size did not work, try to force
124 // (requires CAP_NET_ADMIN to be successful)
125 if (its_option.value() < 0
126 || its_option.value() < its_udp_recv_buffer_size) {
127 ec.assign(setsockopt(unicast_socket_.native_handle(),
128 SOL_SOCKET, SO_RCVBUFFORCE,
129 &its_udp_recv_buffer_size, sizeof(its_udp_recv_buffer_size)),
130 boost::system::generic_category());
131 if (!ec) {
132 VSOMEIP_INFO << "udp_server_endpoint_impl: "
133 << "SO_RCVBUFFORCE successful.";
134 }
135 unicast_socket_.get_option(its_option, ec);
136 }
137#endif
138 if (ec) {
139 VSOMEIP_WARNING << "udp_server_endpoint_impl: couldn't get "
140 << "SO_RCVBUF: " << ec.message() << " local port:"
141 << std::dec << local_port_;
142 } else {
143 VSOMEIP_INFO << "udp_server_endpoint_impl: SO_RCVBUF is: "
144 << std::dec << its_option.value()
145 << " (" << its_udp_recv_buffer_size << ") local port:"
146 << std::dec << local_port_;
147 }

163 server_endpoint_impl::stop();
164 is_stopped_ = true;
165 {
166 std::lock_guard<std::mutex> its_lock(unicast_mutex_);
167
168 if (unicast_socket_.is_open()) {
169 boost::system::error_code its_error;
170 unicast_socket_.cancel(its_error);
171 }
172 }
173
174 {
175 std::lock_guard<std::recursive_mutex> its_lock(multicast_mutex_);
176
177 if (multicast_socket_ && multicast_socket_->is_open()) {
178 boost::system::error_code its_error;
179 multicast_socket_->cancel(its_error);
180 }
181 }
182
183 tp_reassembler_->stop();
184}

291bool udp_server_endpoint_impl::send_error(
292 const std::shared_ptr<endpoint_definition> _target,
293 const byte_t *_data, uint32_t _size) {
294
295 bool ret(false);
296 std::lock_guard<std::mutex> its_lock(mutex_);
297 const endpoint_type its_target(_target->get_address(), _target->get_port());
298 const auto its_target_iterator(find_or_create_target_unlocked(its_target));
299 auto& its_data = its_target_iterator->second;
300
301 if (check_message_size(nullptr, _size, its_target) == endpoint_impl::cms_ret_e::MSG_OK &&
302 check_queue_limit(_data, _size, its_data.queue_size_)) {
303 its_data.queue_.emplace_back(
304 std::make_pair(std::make_shared<message_buffer_t>(_data, _data + _size), 0));
305 its_data.queue_size_ += _size;
306
307 if (!its_data.is_sending_) { // no writing in progress
308 (void)send_queued(its_target_iterator);
309 }
310 ret = true;
311 }
312 return ret;
313}

315bool udp_server_endpoint_impl::send_queued(
316 const target_data_iterator_type _it) {
317
318 std::lock_guard<std::mutex> its_last_sent_lock(last_sent_mutex_);
319 std::lock_guard<std::mutex> its_unicast_lock(unicast_mutex_);
320
321 const auto its_entry = _it->second.queue_.front();
322#if 0
323 std::stringstream msg;
324 msg << "usei::sq(" << _queue_iterator->first.address().to_string() << ":"
325 << _queue_iterator->first.port() << "): ";
326 for (std::size_t i = 0; i < its_buffer->size(); ++i)
327 msg << std::hex << std::setw(2) << std::setfill('0')
328 << (int)(*its_entry.first)[i] << " ";
329 VSOMEIP_INFO << msg.str();
330#endif
331
332 // Check whether we need to wait (SOME/IP-TP separation time)
333 if (its_entry.second > 0) {
334 if (last_sent_ != std::chrono::steady_clock::time_point()) {
335 const auto its_elapsed
336 = std::chrono::duration_cast<std::chrono::microseconds>(
337 std::chrono::steady_clock::now() - last_sent_).count();
338 if (its_entry.second > its_elapsed)
339 std::this_thread::sleep_for(
340 std::chrono::microseconds(its_entry.second - its_elapsed));
341 }
342 last_sent_ = std::chrono::steady_clock::now();
343 } else {
344 last_sent_ = std::chrono::steady_clock::time_point();
345 }
346
347 _it->second.is_sending_ = true;
348 unicast_socket_.async_send_to(
349 boost::asio::buffer(*its_entry.first),
350 _it->first,
351 std::bind(
352 &udp_server_endpoint_base_impl::send_cbk,
353 shared_from_this(),
354 _it->first,
355 std::placeholders::_1,
356 std::placeholders::_2
357 )
358 );
359
360 return false;
361}

400
401void udp_server_endpoint_impl::join_unlocked(const std::string &_address) {
402
403 bool has_received(false);
404
405 //
406 // join_func must be called with multicast_mutex_ being hold!
407 //
408 auto join_func = [this](const std::string &_address) {
409 try {
410 VSOMEIP_DEBUG << "Joining to multicast group " << _address
411 << " from " << local_.address().to_string();
412
413 auto its_endpoint_host = endpoint_host_.lock();
414 if (its_endpoint_host) {
415 multicast_option_t its_join_option { shared_from_this(), true,
416#if VSOMEIP_BOOST_VERSION < 106600
417 boost::asio::ip::address::from_string(_address) };
418#else
419 boost::asio::ip::make_address(_address) };
420#endif
421 its_endpoint_host->add_multicast_option(its_join_option);
422 }
423
424 joined_[_address] = false;
425 } catch (const std::exception &e) {
426 VSOMEIP_ERROR << "udp_server_endpoint_impl::join" << ":" << e.what()
427 << " address: " << _address;
428 }
429 };
430
431 if (!is_joined(_address, &has_received)) {
432 join_func(_address);
433 } else if (!has_received) {
434 // joined the multicast group but didn't receive a event yet -> rejoin
435 leave_unlocked(_address);
436 join_func(_address);
437 }
438}

440void udp_server_endpoint_impl::leave(const std::string &_address) {
441
442 std::lock_guard<std::recursive_mutex> its_lock(multicast_mutex_);
443 leave_unlocked(_address);
444}
445
446void udp_server_endpoint_impl::leave_unlocked(const std::string &_address) {
447
448 try {
449 if (is_joined(_address)) {
450 VSOMEIP_DEBUG << "Leaving the multicast group " << _address
451 << " from " << local_.address().to_string();
452
453 if (multicast_socket_) {
454 auto its_endpoint_host = endpoint_host_.lock();
455 if (its_endpoint_host) {
456 multicast_option_t its_leave_option { shared_from_this(),
457#if VSOMEIP_BOOST_VERSION < 106600
458 false, boost::asio::ip::address::from_string(_address) };
459#else
460 false, boost::asio::ip::make_address(_address) };
461#endif
462 its_endpoint_host->add_multicast_option(its_leave_option);
463 }
464 }
465
466 joined_.erase(_address);
467 }
468 }
469 catch (const std::exception &e) {
470 VSOMEIP_ERROR << __func__ << ":" << e.what()
471 << " address: " << _address;
472 }
473}

508void udp_server_endpoint_impl::on_unicast_received(
509 boost::system::error_code const &_error,
510 std::size_t _bytes) {
511
512 if (is_stopped_
513 || _error == boost::asio::error::eof
514 || _error == boost::asio::error::connection_reset) {
515 shutdown_and_close();
516 } else if (_error != boost::asio::error::operation_aborted) {
517 {
518 // By locking the multicast mutex here it is ensured that unicast
519 // & multicast messages are not processed in parallel. This aligns
520 // the behavior of endpoints with one and two active sockets.
521 std::lock_guard<std::recursive_mutex> its_lock(multicast_mutex_);
522 on_message_received(_error, _bytes, false,
523 unicast_remote_, unicast_recv_buffer_);
524 }
525 receive_unicast();
526 }
527}
528
529void udp_server_endpoint_impl::on_multicast_received(
530 boost::system::error_code const &_error,
531 std::size_t _bytes,
532 uint8_t _multicast_id,
533 const boost::asio::ip::address &_destination) {
534
535 std::lock_guard<std::recursive_mutex> its_lock(multicast_mutex_);
536 if (is_stopped_
537 || _error == boost::asio::error::eof
538 || _error == boost::asio::error::connection_reset) {
539 shutdown_and_close();
540 } else if (_error != boost::asio::error::operation_aborted) {
541 // Filter messages sent from the same source address
542 if (multicast_remote_.address() != local_.address()
543 && is_same_subnet(multicast_remote_.address())) {
544
545 auto find_joined = joined_.find(_destination.to_string());
546 if (find_joined != joined_.end())
547 find_joined->second = true;
548
549 on_message_received(_error, _bytes, true,
550 multicast_remote_, multicast_recv_buffer_);
551 }
552
553 receive_multicast(_multicast_id);
554 }
555}

557void udp_server_endpoint_impl::on_message_received(
558 boost::system::error_code const &_error, std::size_t _bytes,
559 bool _is_multicast,
560 endpoint_type const &_remote,
561 message_buffer_t const &_buffer) {
562#if 0
563 std::stringstream msg;
564 msg << "usei::rcb(" << _error.message() << "): ";
565 for (std::size_t i = 0; i < _bytes; ++i)
566 msg << std::hex << std::setw(2) << std::setfill('0')
567 << (int) _buffer[i] << " ";
568 VSOMEIP_INFO << msg.str();
569#endif
570 std::shared_ptr<routing_host> its_host = routing_host_.lock();
571
572 if (its_host) {
573 if (!_error && 0 < _bytes) {
574 std::size_t remaining_bytes = _bytes;
575 std::size_t i = 0;
576 const boost::asio::ip::address its_remote_address(_remote.address());
577 const std::uint16_t its_remote_port(_remote.port());
578 do {
579 uint64_t read_message_size
580 = utility::get_message_size(&_buffer[i],
581 remaining_bytes);
582 if (read_message_size > MESSAGE_SIZE_UNLIMITED) {
583 VSOMEIP_ERROR << "Message size exceeds allowed maximum!";
584 return;
585 }
586 uint32_t current_message_size = static_cast<uint32_t>(read_message_size);
587 if (current_message_size > VSOMEIP_SOMEIP_HEADER_SIZE &&
588 current_message_size <= remaining_bytes) {
589 if (remaining_bytes - current_message_size > remaining_bytes) {
590 VSOMEIP_ERROR << "buffer underflow in udp client endpoint ~> abort!";
591 return;
592 } else if (current_message_size > VSOMEIP_RETURN_CODE_POS &&
593 (_buffer[i + VSOMEIP_PROTOCOL_VERSION_POS] != VSOMEIP_PROTOCOL_VERSION ||
594 !utility::is_valid_message_type(tp::tp::tp_flag_unset(_buffer[i + VSOMEIP_MESSAGE_TYPE_POS])) ||
595 !utility::is_valid_return_code(static_cast<return_code_e>(_buffer[i + VSOMEIP_RETURN_CODE_POS])) ||
596 (tp::tp::tp_flag_is_set(_buffer[i + VSOMEIP_MESSAGE_TYPE_POS]) && get_local_port() == configuration_->get_sd_port())
597 )) {
598 if (_buffer[i + VSOMEIP_PROTOCOL_VERSION_POS] != VSOMEIP_PROTOCOL_VERSION) {
599 VSOMEIP_ERROR << "use: Wrong protocol version: 0x"
600 << std::hex << std::setw(2) << std::setfill('0')
601 << std::uint32_t(_buffer[i + VSOMEIP_PROTOCOL_VERSION_POS])
602 << " local: " << get_address_port_local()
603 << " remote: " << its_remote_address << ":" << std::dec << its_remote_port;
604 // ensure to send back a message w/ wrong protocol version
605 its_host->on_message(&_buffer[i],
606 VSOMEIP_SOMEIP_HEADER_SIZE + 8, this,
607 _is_multicast,
608 VSOMEIP_ROUTING_CLIENT,
609 nullptr,
610 its_remote_address, its_remote_port);
611 } else if (!utility::is_valid_message_type(tp::tp::tp_flag_unset(
612 _buffer[i + VSOMEIP_MESSAGE_TYPE_POS]))) {
613 VSOMEIP_ERROR << "use: Invalid message type: 0x"
614 << std::hex << std::setw(2) << std::setfill('0')
615 << std::uint32_t(_buffer[i + VSOMEIP_MESSAGE_TYPE_POS])
616 << " local: " << get_address_port_local()
617 << " remote: " << its_remote_address << ":" << std::dec << its_remote_port;
618 } else if (!utility::is_valid_return_code(static_cast<return_code_e>(
619 _buffer[i + VSOMEIP_RETURN_CODE_POS]))) {
620 VSOMEIP_ERROR << "use: Invalid return code: 0x"
621 << std::hex << std::setw(2) << std::setfill('0')
622 << std::uint32_t(_buffer[i + VSOMEIP_RETURN_CODE_POS])
623 << " local: " << get_address_port_local()
624 << " remote: " << its_remote_address << ":" << std::dec << its_remote_port;
625 } else if (tp::tp::tp_flag_is_set(_buffer[i + VSOMEIP_MESSAGE_TYPE_POS])
626 && get_local_port() == configuration_->get_sd_port()) {
627 VSOMEIP_WARNING << "use: Received a SomeIP/TP message on SD port:"
628 << " local: " << get_address_port_local()
629 << " remote: " << its_remote_address << ":" << std::dec << its_remote_port;
630 }
631 return;
632 }
633 remaining_bytes -= current_message_size;
634 const service_t its_service = VSOMEIP_BYTES_TO_WORD(_buffer[i + VSOMEIP_SERVICE_POS_MIN],
635 _buffer[i + VSOMEIP_SERVICE_POS_MAX]);
636 if (utility::is_request(
637 _buffer[i + VSOMEIP_MESSAGE_TYPE_POS])) {
638 const client_t its_client = VSOMEIP_BYTES_TO_WORD(
639 _buffer[i + VSOMEIP_CLIENT_POS_MIN],
640 _buffer[i + VSOMEIP_CLIENT_POS_MAX]);
641 if (its_client != MAGIC_COOKIE_CLIENT) {
642 const session_t its_session = VSOMEIP_BYTES_TO_WORD(
643 _buffer[i + VSOMEIP_SESSION_POS_MIN],
644 _buffer[i + VSOMEIP_SESSION_POS_MAX]);
645 clients_mutex_.lock();
646 clients_[its_client][its_session] = _remote;
647 clients_mutex_.unlock();
648 }
649 }
650 if (tp::tp::tp_flag_is_set(_buffer[i + VSOMEIP_MESSAGE_TYPE_POS])) {
651 const method_t its_method = VSOMEIP_BYTES_TO_WORD(_buffer[i + VSOMEIP_METHOD_POS_MIN],
652 _buffer[i + VSOMEIP_METHOD_POS_MAX]);
653 if (!tp_segmentation_enabled(its_service, its_method)) {
654 VSOMEIP_WARNING << "use: Received a SomeIP/TP message for service: 0x" << std::hex << its_service
655 << " method: 0x" << its_method << " which is not configured for TP:"
656 << " local: " << get_address_port_local()
657 << " remote: " << its_remote_address << ":" << std::dec << its_remote_port;
658 return;
659 }
660 const auto res = tp_reassembler_->process_tp_message(
661 &_buffer[i], current_message_size,
662 its_remote_address, its_remote_port);
663 if (res.first) {
664 if (utility::is_request(res.second[VSOMEIP_MESSAGE_TYPE_POS])) {
665 const client_t its_client = VSOMEIP_BYTES_TO_WORD(
666 res.second[VSOMEIP_CLIENT_POS_MIN],
667 res.second[VSOMEIP_CLIENT_POS_MAX]);
668 if (its_client != MAGIC_COOKIE_CLIENT) {
669 const session_t its_session = VSOMEIP_BYTES_TO_WORD(
670 res.second[VSOMEIP_SESSION_POS_MIN],
671 res.second[VSOMEIP_SESSION_POS_MAX]);
672 std::lock_guard<std::mutex> its_client_lock(clients_mutex_);
673 clients_[its_client][its_session] = _remote;
674 }
675 }
676 its_host->on_message(&res.second[0],
677 static_cast<std::uint32_t>(res.second.size()),
678 this, _is_multicast, VSOMEIP_ROUTING_CLIENT,
679 nullptr,
680 its_remote_address, its_remote_port);
681 }
682 } else {
683 if (its_service != VSOMEIP_SD_SERVICE ||
684 (current_message_size > VSOMEIP_SOMEIP_HEADER_SIZE &&
685 current_message_size >= remaining_bytes)) {
686 its_host->on_message(&_buffer[i],
687 current_message_size, this, _is_multicast,
688 VSOMEIP_ROUTING_CLIENT,
689 nullptr,
690 its_remote_address, its_remote_port);
691 } else {
692 //ignore messages for service discovery with shorter SomeIP length
693 VSOMEIP_ERROR << "Received an unreliable vSomeIP SD message with too short length field"
694 << " local: " << get_address_port_local()
695 << " remote: " << its_remote_address << ":" << std::dec << its_remote_port;
696 }
697 }
698 i += current_message_size;
699 } else {
700 VSOMEIP_ERROR << "Received an unreliable vSomeIP message with bad length field"
701 << " local: " << get_address_port_local()
702 << " remote: " << its_remote_address << ":" << std::dec << its_remote_port;
703 if (remaining_bytes > VSOMEIP_SERVICE_POS_MAX) {
704 service_t its_service = VSOMEIP_BYTES_TO_WORD(_buffer[VSOMEIP_SERVICE_POS_MIN],
705 _buffer[VSOMEIP_SERVICE_POS_MAX]);
706 if (its_service != VSOMEIP_SD_SERVICE) {
707 if (read_message_size == 0) {
708 VSOMEIP_ERROR << "Ignoring unreliable vSomeIP message with SomeIP message length 0!";
709 } else {
710 auto its_endpoint_host = endpoint_host_.lock();
711 if (its_endpoint_host) {
712 its_endpoint_host->on_error(&_buffer[i],
713 (uint32_t)remaining_bytes, this,
714 its_remote_address, its_remote_port);
715 }
716 }
717 }
718 }
719 remaining_bytes = 0;
720 }
721 } while (remaining_bytes > 0);
722 }
723 }

726bool udp_server_endpoint_impl::is_same_subnet(const boost::asio::ip::address &_address) const {
727
728 bool is_same(true);
729#if VSOMEIP_BOOST_VERSION < 106600
730 // TODO: This needs some (more) testing
731 if (_address.is_v4()) {
732 uint32_t its_local(uint32_t(local_.address().to_v4().to_ulong()));
733 uint32_t its_mask(uint32_t(netmask_.to_v4().to_ulong()));
734 uint32_t its_address(uint32_t(_address.to_v4().to_ulong()));
735
736 return ((its_local & its_mask) == (its_address & its_mask));
737 } else {
738 boost::asio::ip::address_v6::bytes_type its_local(local_.address().to_v6().to_bytes());
739 boost::asio::ip::address_v6::bytes_type its_address(_address.to_v6().to_bytes());
740
741 for (size_t i = 0; i < its_local.size(); ++i) {
742 byte_t its_mask(0x00);
743 if ((i+1) * sizeof(byte_t) <= prefix_)
744 its_mask = 0xff;
745 else if (i <= prefix_)
746 its_mask = byte_t(0xff << (((i+1) * sizeof(byte_t)) - prefix_));
747
748 if ((its_local[i] & its_mask) != (its_address[i] & its_mask))
749 return false;
750 }
751
752 return true;
753 }
754#else
755 if (_address.is_v4()) {
756 boost::asio::ip::network_v4 its_network(local_.address().to_v4(), netmask_.to_v4());
757 boost::asio::ip::address_v4_range its_hosts = its_network.hosts();
758 is_same = (its_hosts.find(_address.to_v4()) != its_hosts.end());
759 } else {
760 boost::asio::ip::network_v6 its_network(local_.address().to_v6(), prefix_);
761 boost::asio::ip::address_v6_range its_hosts = its_network.hosts();
762 is_same = (its_hosts.find(_address.to_v6()) != its_hosts.end());
763 }
764#endif
765 return is_same;
766}

838void
839udp_server_endpoint_impl::set_multicast_option(
840 const boost::asio::ip::address &_address, bool _is_join) {
841
842 boost::system::error_code ec;
843
844 if (_is_join) {
845 if (!multicast_socket_) {
846 std::lock_guard<std::recursive_mutex> its_guard(multicast_mutex_);
847
848 multicast_socket_ = std::make_unique<socket_type>(io_, local_.protocol());
849
850 multicast_socket_->set_option(ip::udp::socket::reuse_address(true), ec);
851 if (ec)
852 VSOMEIP_ERROR << __func__
853 << ": set reuse address option failed (" << ec.message() << ")";
854
855#ifdef _WIN32
856 const char *its_pktinfo_option("0001");
857 ::setsockopt(multicast_socket_->native_handle(),
858 (is_v4_ ? IPPROTO_IP : IPPROTO_IPV6),
859 (is_v4_ ? IP_PKTINFO : IPV6_PKTINFO),
860 its_pktinfo_option, sizeof(its_pktinfo_option));
861#else
862 int its_pktinfo_option(1);
863 ::setsockopt(multicast_socket_->native_handle(),
864 (is_v4_ ? IPPROTO_IP : IPPROTO_IPV6),
865 (is_v4_ ? IP_PKTINFO : IPV6_RECVPKTINFO),
866 &its_pktinfo_option, sizeof(its_pktinfo_option));
867#endif
868
869 if (multicast_recv_buffer_.empty())
870 multicast_recv_buffer_.resize(VSOMEIP_MAX_UDP_MESSAGE_SIZE, 0);
871
872 if (!multicast_local_) {
873 if (is_v4_) {
874 multicast_local_ = std::make_unique<endpoint_type>
875 (boost::asio::ip::address_v4::any(), local_port_);
876 } else { // is_v6
877 multicast_local_ = std::make_unique<endpoint_type>
878 (boost::asio::ip::address_v6::any(), local_port_);
879 }
880 }
881
882 multicast_socket_->bind(*multicast_local_, ec);
883 if (ec)
884 VSOMEIP_ERROR << __func__
885 << ": bind failed (" << ec.message() << ")";
886
887 const int its_udp_recv_buffer_size =
888 configuration_->get_udp_receive_buffer_size();
889
890 multicast_socket_->set_option(boost::asio::socket_base::receive_buffer_size(
891 its_udp_recv_buffer_size), ec);
892
893#ifndef _WIN32
894 // define socket timeout
895 struct timeval timeout;
896 timeout.tv_sec = 0;
897 timeout.tv_usec = VSOMEIP_SETSOCKOPT_TIMEOUT_US;
898
899 if (setsockopt(
900 multicast_socket_->native_handle(),
901 SOL_SOCKET, SO_RCVTIMEO,
902 &timeout, sizeof(timeout)) == -1) {
903 VSOMEIP_WARNING << __func__
904 << ": unable to setsockopt SO_RCVTIMEO";
905 }
906
907 if (setsockopt(
908 multicast_socket_->native_handle(),
909 SOL_SOCKET, SO_SNDTIMEO,
910 &timeout, sizeof(timeout)) == -1) {
911 VSOMEIP_WARNING << __func__
912 << ": unable to setsockopt SO_SNDTIMEO";
913 }
914#endif
915
916 if (ec) {
917 VSOMEIP_WARNING << "udp_server_endpoint_impl<multicast>: couldn't set "
918 << "SO_RCVBUF: " << ec.message() << " to: " << std::dec
919 << its_udp_recv_buffer_size << " local port: " << std::dec
920 << local_port_;
921 }
922
923 boost::asio::socket_base::receive_buffer_size its_option;
924 multicast_socket_->get_option(its_option, ec);
925#ifdef __linux__
926 // If regular setting of the buffer size did not work, try to force
927 // (requires CAP_NET_ADMIN to be successful)
928 if (its_option.value() < 0
929 || its_option.value() < its_udp_recv_buffer_size) {
930 ec.assign(setsockopt(multicast_socket_->native_handle(),
931 SOL_SOCKET, SO_RCVBUFFORCE,
932 &its_udp_recv_buffer_size, sizeof(its_udp_recv_buffer_size)),
933 boost::system::generic_category());
934 if (!ec) {
935 VSOMEIP_INFO << "udp_server_endpoint_impl<multicast>: "
936 << "SO_RCVBUFFORCE: successful.";
937 }
938 multicast_socket_->get_option(its_option, ec);
939 }
940#endif
941 if (ec) {
942 VSOMEIP_WARNING << "udp_server_endpoint_impl<multicast>: couldn't get "
943 << "SO_RCVBUF: " << ec.message() << " local port:"
944 << std::dec << local_port_;
945 } else {
946 VSOMEIP_INFO << "udp_server_endpoint_impl<multicast>: SO_RCVBUF is: "
947 << std::dec << its_option.value()
948 << " (" << its_udp_recv_buffer_size << ") local port:"
949 << std::dec << local_port_;
950 }
951
952 multicast_id_++;
953 receive_multicast(multicast_id_);
954 }
955
956 boost::asio::ip::multicast::join_group its_join_option;
957 {
958 std::lock_guard<std::mutex> its_lock(local_mutex_);
959 if (is_v4_) {
960
961 its_join_option = boost::asio::ip::multicast::join_group(
962 _address.to_v4(),
963 local_.address().to_v4());
964 } else {
965 its_join_option = boost::asio::ip::multicast::join_group(
966 _address.to_v6(),
967 static_cast<unsigned int>(local_.address().to_v6().scope_id()));
968 }
969 }
970 multicast_socket_->set_option(its_join_option, ec);
971
972 if (!ec) {
973 std::lock_guard<std::recursive_mutex> its_guard(multicast_mutex_);
974 joined_[_address.to_string()] = false;
975 joined_group_ = true;
976 }
977 } else {
978 if (multicast_socket_) {
979 boost::asio::ip::multicast::leave_group its_leave_option(_address);
980 multicast_socket_->set_option(its_leave_option, ec);
981
982 if (!ec) {
983 std::lock_guard<std::recursive_mutex> its_guard(multicast_mutex_);
984 joined_.erase(_address.to_string());
985
986 if (0 == joined_.size()) {
987 joined_group_ = false;
988
989 multicast_socket_->cancel(ec);
990
991 multicast_socket_.reset(nullptr);
992 multicast_local_.reset(nullptr);
993 }
994 }
995 }
996 }
