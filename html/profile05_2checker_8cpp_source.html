17void profile_05_checker::check(const e2e_buffer &_buffer, instance_t _instance,
18 e2e::profile_interface::check_status_t &_generic_check_status) {
19
20 (void)_instance;
21
22 std::lock_guard<std::mutex> lock(check_mutex_);
23 _generic_check_status = e2e::profile_interface::generic_check_status::E2E_ERROR;
24
25 if (_instance > VSOMEIP_E2E_PROFILE05_MAX_INSTANCE) {
26 VSOMEIP_ERROR << "E2E Profile 5 can only be used for instances [1-255]";
27 return;
28 }
29
30 if (profile_05::is_buffer_length_valid(config_, _buffer)) {
31 uint8_t its_received_counter;
32 if (read_8(_buffer, its_received_counter, 2)) {
33 uint16_t its_received_crc;
34 if (read_16(_buffer, its_received_crc, 0)) {
35 uint16_t its_crc = profile_05::compute_crc(config_, _buffer);
36 if (its_received_crc != its_crc) {
37 _generic_check_status = e2e::profile_interface::generic_check_status::E2E_WRONG_CRC;
38 VSOMEIP_ERROR << std::hex << "E2E P05 protection: CRC16 does not match: calculated CRC: "
39 << its_crc << " received CRC: " << its_received_crc;
40 } else {
41 if (verify_counter(_instance, its_received_counter)) {
42 _generic_check_status = e2e::profile_interface::generic_check_status::E2E_OK;
43 }
44 }
45 }
46 }
47 }

50bool
51profile_05_checker::verify_counter(instance_t _instance, uint8_t _received_counter) {
52
53 uint8_t its_delta(0);
54
55 auto find_counter = counter_.find(_instance);
56 if (find_counter != counter_.end()) {
57 uint8_t its_counter = find_counter->second;
58 if (its_counter < _received_counter)
59 its_delta = uint8_t(_received_counter - its_counter);
60 else
61 its_delta = uint8_t(uint8_t(0xff) - its_counter + _received_counter);
62 } else {
63 counter_[_instance] = _received_counter;
64 }
65
66 return (its_delta <= config_.max_delta_counter_);
67}
