74void tcp_client_endpoint_impl::restart(bool _force) {
75 auto self = std::dynamic_pointer_cast< tcp_client_endpoint_impl >(shared_from_this());
76 auto restart_func = [self, _force] {
77 if (!_force && self->state_ == cei_state_e::CONNECTING) {
78 std::chrono::steady_clock::time_point its_current
79 = std::chrono::steady_clock::now();
80 std::int64_t its_connect_duration = std::chrono::duration_cast<std::chrono::milliseconds>(
81 its_current - self->connect_timepoint_).count();
82 if (self->aborted_restart_count_ < self->tcp_restart_aborts_max_
83 && its_connect_duration < self->tcp_connect_time_max_) {
84 self->aborted_restart_count_++;
85 return;
86 } else {
87 VSOMEIP_WARNING << "tce::restart: maximum number of aborted restarts ["
88 << self->tcp_restart_aborts_max_ << "] reached! its_connect_duration: "
89 << its_connect_duration;
90 }
91 }
92 self->state_ = cei_state_e::CONNECTING;
93 std::string address_port_local;
94 {
95 std::lock_guard<std::mutex> its_lock(self->socket_mutex_);
96 address_port_local = self->get_address_port_local();
97 self->shutdown_and_close_socket_unlocked(true);
98 self->recv_buffer_ = std::make_shared<message_buffer_t>(self->recv_buffer_size_initial_, 0);
99 }
100 self->was_not_connected_ = true;
101 self->reconnect_counter_ = 0;
102 {
103 std::lock_guard<std::recursive_mutex> its_lock(self->mutex_);
104 for (const auto &q : self->queue_) {
105 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
106 (*q.first)[VSOMEIP_SERVICE_POS_MIN],
107 (*q.first)[VSOMEIP_SERVICE_POS_MAX]);
108 const method_t its_method = VSOMEIP_BYTES_TO_WORD(
109 (*q.first)[VSOMEIP_METHOD_POS_MIN],
110 (*q.first)[VSOMEIP_METHOD_POS_MAX]);
111 const client_t its_client = VSOMEIP_BYTES_TO_WORD(
112 (*q.first)[VSOMEIP_CLIENT_POS_MIN],
113 (*q.first)[VSOMEIP_CLIENT_POS_MAX]);
114 const session_t its_session = VSOMEIP_BYTES_TO_WORD(
115 (*q.first)[VSOMEIP_SESSION_POS_MIN],
116 (*q.first)[VSOMEIP_SESSION_POS_MAX]);
117 VSOMEIP_WARNING << "tce::restart: dropping message: "
118 << "remote:" << self->get_address_port_remote() << " ("
119 << std::hex << std::setfill('0')
120 << std::setw(4) << its_client << "): ["
121 << std::setw(4) << its_service << "."
122 << std::setw(4) << its_method << "."
123 << std::setw(4) << its_session << "]"
124 << " size: " << std::dec << q.first->size();
125 }
126 self->queue_.clear();
127 self->queue_size_ = 0;
128 }
129 VSOMEIP_WARNING << "tce::restart: local: " << address_port_local
130 << " remote: " << self->get_address_port_remote();
131 self->start_connect_timer();
132 };
133 // bind to strand_ to avoid socket closure if
134 // parallel socket operation is currently active
135 strand_.dispatch(restart_func);
136}

138void tcp_client_endpoint_impl::connect() {
139 start_connecting_timer();
140 std::lock_guard<std::mutex> its_lock(socket_mutex_);
141 boost::system::error_code its_error;
142 socket_->open(remote_.protocol(), its_error);
143
144 if (!its_error || its_error == boost::asio::error::already_open) {
145 // Nagle algorithm off
146 socket_->set_option(ip::tcp::no_delay(true), its_error);
147 if (its_error) {
148 VSOMEIP_WARNING << "tcp_client_endpoint::connect: couldn't disable "
149 << "Nagle algorithm: " << its_error.message()
150 << " remote:" << get_address_port_remote();
151 }
152
153 socket_->set_option(boost::asio::socket_base::keep_alive(true), its_error);
154 if (its_error) {
155 VSOMEIP_WARNING << "tcp_client_endpoint::connect: couldn't enable "
156 << "keep_alive: " << its_error.message()
157 << " remote:" << get_address_port_remote();
158 }
159
160 // Enable SO_REUSEADDR to avoid bind problems with services going offline
161 // and coming online again and the user has specified only a small number
162 // of ports in the clients section for one service instance
163 socket_->set_option(boost::asio::socket_base::reuse_address(true), its_error);
164 if (its_error) {
165 VSOMEIP_WARNING << "tcp_client_endpoint::connect: couldn't enable "
166 << "SO_REUSEADDR: " << its_error.message()
167 << " remote:" << get_address_port_remote();
168 }
169 socket_->set_option(boost::asio::socket_base::linger(true, 0), its_error);
170 if (its_error) {
171 VSOMEIP_WARNING << "tcp_client_endpoint::connect: couldn't enable "
172 << "SO_LINGER: " << its_error.message()
173 << " remote:" << get_address_port_remote();
174 }
175
176#if defined(__linux__) || defined(ANDROID) || defined(__QNX__)
177 // If specified, bind to device
178 std::string its_device(configuration_->get_device());
179 if (its_device != "") {
180 if (setsockopt(socket_->native_handle(),
181 SOL_SOCKET, SO_BINDTODEVICE, its_device.c_str(), (socklen_t)its_device.size()) == -1) {
182 VSOMEIP_WARNING << "TCP Client: Could not bind to device \"" << its_device << "\"";
183 }
184 }
185#endif
186
187 // In case a client endpoint port was configured,
188 // bind to it before connecting
189 if (local_.port() != ILLEGAL_PORT) {
190 boost::system::error_code its_bind_error;
191 socket_->bind(local_, its_bind_error);
192 if(its_bind_error) {
193 VSOMEIP_WARNING << "tcp_client_endpoint::connect: "
194 "Error binding socket: " << its_bind_error.message()
195 << " local: " << get_address_port_local()
196 << " remote:" << get_address_port_remote();
197
198 std::shared_ptr<endpoint_host> its_host = endpoint_host_.lock();
199 if (its_host) {
200 // set new client port depending on service / instance / remote port
201 if (!its_host->on_bind_error(shared_from_this(), remote_address_, remote_port_)) {
202 VSOMEIP_WARNING << "tcp_client_endpoint::connect: "
203 "Failed to set new local port for tce: "
204 << " local: " << local_.address().to_string()
205 << ":" << std::dec << local_.port()
206 << " remote:" << get_address_port_remote();
207 } else {
208 VSOMEIP_INFO << "tcp_client_endpoint::connect: "
209 "Using new new local port for tce: "
210 << " local: " << local_.address().to_string()
211 << ":" << std::dec << local_.port()
212 << " remote:" << get_address_port_remote();
213 }
214 }
215 std::size_t operations_cancelled;
216 {
217 std::lock_guard<std::mutex> its_lock(connecting_timer_mutex_);
218 operations_cancelled = connecting_timer_.cancel();
219 }
220 if (operations_cancelled != 0) {
221 try {
222 // don't connect on bind error to avoid using a random port
223 strand_.post(std::bind(&client_endpoint_impl::connect_cbk,
224 shared_from_this(), its_bind_error));
225 } catch (const std::exception &e) {
226 VSOMEIP_ERROR << "tcp_client_endpoint_impl::connect: "
227 << e.what()
228 << " local: " << get_address_port_local()
229 << " remote:" << get_address_port_remote();
230 }
231 }
232 return;
233 }
234 }
235 state_ = cei_state_e::CONNECTING;
236 connect_timepoint_ = std::chrono::steady_clock::now();
237 aborted_restart_count_ = 0;
238 socket_->async_connect(
239 remote_,
240 strand_.wrap(
241 std::bind(
242 &tcp_client_endpoint_base_impl::cancel_and_connect_cbk,
243 shared_from_this(),
244 std::placeholders::_1
245 )
246 )
247 );
248 } else {
249 std::size_t operations_cancelled;
250 {
251 std::lock_guard<std::mutex> its_lock(connecting_timer_mutex_);
252 operations_cancelled = connecting_timer_.cancel();
253 }
254 if (operations_cancelled != 0) {
255 VSOMEIP_WARNING << "tcp_client_endpoint::connect: Error opening socket: "
256 << its_error.message() << " remote:" << get_address_port_remote();
257 strand_.post(std::bind(&tcp_client_endpoint_base_impl::connect_cbk,
258 shared_from_this(), its_error));
259 }
260 }

275void tcp_client_endpoint_impl::receive(message_buffer_ptr_t _recv_buffer,
276 std::size_t _recv_buffer_size,
277 std::size_t _missing_capacity) {
278 std::lock_guard<std::mutex> its_lock(socket_mutex_);
279 if(socket_->is_open()) {
280 const std::size_t its_capacity(_recv_buffer->capacity());
281 size_t buffer_size = its_capacity - _recv_buffer_size;
282 try {
283 if (_missing_capacity) {
284 if (_missing_capacity > MESSAGE_SIZE_UNLIMITED) {
285 VSOMEIP_ERROR << "Missing receive buffer capacity exceeds allowed maximum!";
286 return;
287 }
288 const std::size_t its_required_capacity(_recv_buffer_size + _missing_capacity);
289 if (its_capacity < its_required_capacity) {
290 _recv_buffer->reserve(its_required_capacity);
291 _recv_buffer->resize(its_required_capacity, 0x0);
292 if (_recv_buffer->size() > 1048576) {
293 VSOMEIP_INFO << "tce: recv_buffer size is: " <<
294 _recv_buffer->size()
295 << " local: " << get_address_port_local()
296 << " remote: " << get_address_port_remote();
297 }
298 }
299 buffer_size = _missing_capacity;
300 } else if (buffer_shrink_threshold_
301 && shrink_count_ > buffer_shrink_threshold_
302 && _recv_buffer_size == 0) {
303 _recv_buffer->resize(recv_buffer_size_initial_, 0x0);
304 _recv_buffer->shrink_to_fit();
305 buffer_size = recv_buffer_size_initial_;
306 shrink_count_ = 0;
307 }
308 } catch (const std::exception &e) {
309 handle_recv_buffer_exception(e, _recv_buffer, _recv_buffer_size);
310 // don't start receiving again
311 return;
312 }
313 socket_->async_receive(
314 boost::asio::buffer(&(*_recv_buffer)[_recv_buffer_size], buffer_size),
315 strand_.wrap(
316 std::bind(
317 &tcp_client_endpoint_impl::receive_cbk,
318 std::dynamic_pointer_cast< tcp_client_endpoint_impl >(shared_from_this()),
319 std::placeholders::_1,
320 std::placeholders::_2,
321 _recv_buffer,
322 _recv_buffer_size
323 )
324 )
325 );
326 }

330 const service_t its_service = VSOMEIP_BYTES_TO_WORD(

340 (*_entry.first)[VSOMEIP_SESSION_POS_MIN],

342
343 if (has_enabled_magic_cookies_) {
344 const std::chrono::steady_clock::time_point now =
345 std::chrono::steady_clock::now();
346 if (std::chrono::duration_cast<std::chrono::milliseconds>(

351 }
352
353
354#if 0

364 std::lock_guard<std::mutex> its_lock(socket_mutex_);
365 if (socket_->is_open()) {
366 boost::asio::async_write(
367 *socket_,
368 boost::asio::buffer(*_entry.first),

371 std::static_pointer_cast<tcp_client_endpoint_impl>(shared_from_this()),
372 std::placeholders::_1,
373 std::placeholders::_2,
374 _entry.first->size(),
375 its_service, its_method, its_client, its_session,
376 std::chrono::steady_clock::now()),
377 strand_.wrap(
378 std::bind(
379 &tcp_client_endpoint_base_impl::send_cbk,
380 shared_from_this(),
381 std::placeholders::_1,
382 std::placeholders::_2,
383 _entry.first
384 ))
385 );
386 }

458std::size_t tcp_client_endpoint_impl::write_completion_condition(
459 const boost::system::error_code& _error, std::size_t _bytes_transferred,
460 std::size_t _bytes_to_send, service_t _service, method_t _method,
461 client_t _client, session_t _session,
462 const std::chrono::steady_clock::time_point _start) {
463
464 if (_error) {
465 VSOMEIP_ERROR << "tce::write_completion_condition: "
466 << _error.message() << "(" << std::dec << _error.value()
467 << ") bytes transferred: " << std::dec << _bytes_transferred
468 << " bytes to sent: " << std::dec << _bytes_to_send << " "
469 << "remote:" << get_address_port_remote() << " ("
470 << std::hex << std::setfill('0')
471 << std::setw(4) << _client << "): ["
472 << std::setw(4) << _service << "."
473 << std::setw(4) << _method << "."
474 << std::setw(4) << _session << "]";
475 return 0;
476 }
477
478 const std::chrono::steady_clock::time_point now = std::chrono::steady_clock::now();
479 const std::chrono::milliseconds passed = std::chrono::duration_cast<std::chrono::milliseconds>(now - _start);
480 if (passed > send_timeout_warning_) {
481 if (passed > send_timeout_) {
482 VSOMEIP_ERROR << "tce::write_completion_condition: "
483 << _error.message() << "(" << std::dec << _error.value()
484 << ") took longer than " << std::dec << send_timeout_.count()
485 << "ms bytes transferred: " << std::dec << _bytes_transferred
486 << " bytes to sent: " << std::dec << _bytes_to_send << " "
487 << "remote:" << get_address_port_remote() << " ("
488 << std::hex << std::setfill('0')
489 << std::setw(4) << _client << "): ["
490 << std::setw(4) << _service << "."
491 << std::setw(4) << _method << "."
492 << std::setw(4) << _session << "]";
493 } else {
494 VSOMEIP_WARNING << "tce::write_completion_condition: "
495 << _error.message() << "(" << std::dec << _error.value()
496 << ") took longer than " << std::dec << send_timeout_warning_.count()
497 << "ms bytes transferred: " << std::dec << _bytes_transferred
498 << " bytes to sent: " << std::dec << _bytes_to_send << " "
499 << "remote:" << get_address_port_remote() << " ("
500 << std::hex << std::setfill('0')
501 << std::setw(4) << _client << "): ["
502 << std::setw(4) << _service << "."
503 << std::setw(4) << _method << "."
504 << std::setw(4) << _session << "]";
505 }
506 }
507 return _bytes_to_send - _bytes_transferred;
508}

538void tcp_client_endpoint_impl::receive_cbk(
539 boost::system::error_code const &_error, std::size_t _bytes,
540 const message_buffer_ptr_t& _recv_buffer, std::size_t _recv_buffer_size) {
541 if (_error == boost::asio::error::operation_aborted) {
542 // endpoint was stopped
543 return;
544 }
545#if 0
546 std::stringstream msg;
547 msg << "cei::rcb (" << _error.message() << "): ";
548 for (std::size_t i = 0; i < _bytes + _recv_buffer_size; ++i)
549 msg << std::hex << std::setw(2) << std::setfill('0')
550 << (int) (_recv_buffer)[i] << " ";
551 VSOMEIP_INFO << msg.str();
552#endif
553 std::unique_lock<std::mutex> its_lock(socket_mutex_);
554 std::shared_ptr<routing_host> its_host = routing_host_.lock();
555 if (its_host) {
556 std::uint32_t its_missing_capacity(0);
557 if (!_error && 0 < _bytes) {
558 if (_recv_buffer_size + _bytes > _recv_buffer->size()) {
559 VSOMEIP_ERROR << "receive buffer overflow in tcp client endpoint ~> abort!";
560 return;
561 }
562 _recv_buffer_size += _bytes;
563
564 size_t its_iteration_gap = 0;
565 bool has_full_message(false);
566 do {
567 uint64_t read_message_size
568 = utility::get_message_size(&(*_recv_buffer)[its_iteration_gap],
569 _recv_buffer_size);
570 if (read_message_size > MESSAGE_SIZE_UNLIMITED) {
571 VSOMEIP_ERROR << "Message size exceeds allowed maximum!";
572 return;
573 }
574 uint32_t current_message_size = static_cast<uint32_t>(read_message_size);
575 has_full_message = (current_message_size > VSOMEIP_RETURN_CODE_POS
576 && current_message_size <= _recv_buffer_size);
577 if (has_full_message) {
578 bool needs_forwarding(true);
579 if (is_magic_cookie(_recv_buffer, its_iteration_gap)) {
580 has_enabled_magic_cookies_ = true;
581 } else {
582 if (has_enabled_magic_cookies_) {
583 uint32_t its_offset = find_magic_cookie(&(*_recv_buffer)[its_iteration_gap],
584 (uint32_t) _recv_buffer_size);
585 if (its_offset < current_message_size) {
586 VSOMEIP_ERROR << "Message includes Magic Cookie. Ignoring it.";
587 current_message_size = its_offset;
588 needs_forwarding = false;
589 }
590 }
591 }
592 if (needs_forwarding) {
593 if (!has_enabled_magic_cookies_) {
594 its_lock.unlock();
595 its_host->on_message(&(*_recv_buffer)[its_iteration_gap],
596 current_message_size, this,
597 false,
598 VSOMEIP_ROUTING_CLIENT,
599 nullptr,
600 remote_address_,
601 remote_port_);
602 its_lock.lock();
603 } else {
604 // Only call on_message without a magic cookie in front of the buffer!
605 if (!is_magic_cookie(_recv_buffer, its_iteration_gap)) {
606 its_lock.unlock();
607 its_host->on_message(&(*_recv_buffer)[its_iteration_gap],
608 current_message_size, this,
609 false,
610 VSOMEIP_ROUTING_CLIENT,
611 nullptr,
612 remote_address_,
613 remote_port_);
614 its_lock.lock();
615 }
616 }
617 }
618 calculate_shrink_count(_recv_buffer, _recv_buffer_size);
619 _recv_buffer_size -= current_message_size;
620 its_iteration_gap += current_message_size;
621 its_missing_capacity = 0;
622 } else if (has_enabled_magic_cookies_ && _recv_buffer_size > 0) {
623 const uint32_t its_offset = find_magic_cookie(
624 &(*_recv_buffer)[its_iteration_gap], _recv_buffer_size);
625 if (its_offset < _recv_buffer_size) {
626 _recv_buffer_size -= its_offset;
627 its_iteration_gap += its_offset;
628 has_full_message = true; // trigger next loop
629 VSOMEIP_ERROR << "Detected Magic Cookie within message data."
630 << " Resyncing. local: " << get_address_port_local()
631 << " remote: " << get_address_port_remote();
632 }
633 }
634
635 if (!has_full_message) {
636 if (_recv_buffer_size > VSOMEIP_RETURN_CODE_POS) {
637 bool invalid_parameter_detected { false };
638 if (recv_buffer_->size() <= (its_iteration_gap + VSOMEIP_RETURN_CODE_POS)) {
639 VSOMEIP_ERROR << "TCP Client receive_cbk is trying to access invalid vector position."
640 << " Actual: " << recv_buffer_->size()
641 << " Received: " << _recv_buffer->size()
642 << " Current: " << current_message_size
643 << " Indicated: " << _recv_buffer_size
644 << " Bytes: " << _bytes
645 << " Iteration_gap: " << its_iteration_gap
646 << " Is_full_message: " << has_full_message;
647 return;
648 } else if ((*recv_buffer_)[its_iteration_gap + VSOMEIP_PROTOCOL_VERSION_POS] != VSOMEIP_PROTOCOL_VERSION) {
649 invalid_parameter_detected = true;
650 VSOMEIP_ERROR << "tce: Wrong protocol version: 0x"
651 << std::hex << std::setw(2) << std::setfill('0')
652 << std::uint32_t((*recv_buffer_)[its_iteration_gap + VSOMEIP_PROTOCOL_VERSION_POS])
653 << " local: " << get_address_port_local()
654 << " remote: " << get_address_port_remote();
655 // ensure to send back a message w/ wrong protocol version
656 its_lock.unlock();
657 its_host->on_message(&(*_recv_buffer)[its_iteration_gap],
658 VSOMEIP_SOMEIP_HEADER_SIZE + 8, this,
659 false,
660 VSOMEIP_ROUTING_CLIENT,
661 nullptr,
662 remote_address_,
663 remote_port_);
664 its_lock.lock();
665 } else if (!utility::is_valid_message_type(static_cast<message_type_e>(
666 (*recv_buffer_)[its_iteration_gap + VSOMEIP_MESSAGE_TYPE_POS]))) {
667 invalid_parameter_detected = true;
668 VSOMEIP_ERROR << "tce: Invalid message type: 0x"
669 << std::hex << std::setw(2) << std::setfill('0')
670 << std::uint32_t((*recv_buffer_)[its_iteration_gap + VSOMEIP_MESSAGE_TYPE_POS])
671 << " local: " << get_address_port_local()
672 << " remote: " << get_address_port_remote();
673 } else if (!utility::is_valid_return_code(static_cast<return_code_e>(
674 (*recv_buffer_)[its_iteration_gap + VSOMEIP_RETURN_CODE_POS]))) {
675 invalid_parameter_detected = true;
676 VSOMEIP_ERROR << "tce: Invalid return code: 0x"
677 << std::hex << std::setw(2) << std::setfill('0')
678 << std::uint32_t((*recv_buffer_)[its_iteration_gap + VSOMEIP_RETURN_CODE_POS])
679 << " local: " << get_address_port_local()
680 << " remote: " << get_address_port_remote();
681 }
682
683 if (invalid_parameter_detected) {
684 state_ = cei_state_e::CONNECTING;
685 shutdown_and_close_socket_unlocked(false);
686 its_lock.unlock();
687
688 // wait_until_sent interprets "no error" as timeout.
689 // Therefore call it with an error.
690 wait_until_sent(boost::asio::error::operation_aborted);
691 return;
692 }
693 }
694 if (max_message_size_ != MESSAGE_SIZE_UNLIMITED &&
695 current_message_size > max_message_size_) {
696 _recv_buffer_size = 0;
697 _recv_buffer->resize(recv_buffer_size_initial_, 0x0);
698 _recv_buffer->shrink_to_fit();
699 if (has_enabled_magic_cookies_) {
700 VSOMEIP_ERROR << "Received a TCP message which exceeds "
701 << "maximum message size ("
702 << std::dec << current_message_size
703 << "). Magic Cookies are enabled: "
704 << "Resetting receiver. local: "
705 << get_address_port_local() << " remote: "
706 << get_address_port_remote();
707 } else {
708 VSOMEIP_ERROR << "Received a TCP message which exceeds "
709 << "maximum message size ("
710 << std::dec << current_message_size
711 << ") Magic cookies are disabled, "
712 << "Restarting connection. "
713 << "local: " << get_address_port_local()
714 << " remote: " << get_address_port_remote();
715 state_ = cei_state_e::CONNECTING;
716 shutdown_and_close_socket_unlocked(false);
717 its_lock.unlock();
718
719 // wait_until_sent interprets "no error" as timeout.
720 // Therefore call it with an error.
721 wait_until_sent(boost::asio::error::operation_aborted);
722 return;
723 }
724 } else if (current_message_size > _recv_buffer_size) {
725 its_missing_capacity = current_message_size
726 - static_cast<std::uint32_t>(_recv_buffer_size);
727 } else if (VSOMEIP_SOMEIP_HEADER_SIZE > _recv_buffer_size) {
728 its_missing_capacity = VSOMEIP_SOMEIP_HEADER_SIZE
729 - static_cast<std::uint32_t>(_recv_buffer_size);
730 } else if (has_enabled_magic_cookies_ && _recv_buffer_size > 0) {
731 // no need to check for magic cookie here again: has_full_message
732 // would have been set to true if there was one present in the data
733 _recv_buffer_size = 0;
734 _recv_buffer->resize(recv_buffer_size_initial_, 0x0);
735 _recv_buffer->shrink_to_fit();
736 its_missing_capacity = 0;
737 VSOMEIP_ERROR << "tce::c<" << this
738 << ">rcb: recv_buffer_capacity: "
739 << _recv_buffer->capacity()
740 << " local: " << get_address_port_local()
741 << " remote: " << get_address_port_remote()
742 << ". Didn't find magic cookie in broken data, trying to resync.";
743 } else {
744 VSOMEIP_ERROR << "tce::c<" << this
745 << ">rcb: recv_buffer_size is: " << std::dec
746 << _recv_buffer_size << " but couldn't read "
747 "out message_size. recv_buffer_capacity: "
748 << _recv_buffer->capacity()
749 << " its_iteration_gap: " << its_iteration_gap
750 << " local: " << get_address_port_local()
751 << " remote: " << get_address_port_remote()
752 << ". Restarting connection due to missing/broken data TCP stream.";
753 state_ = cei_state_e::CONNECTING;
754 shutdown_and_close_socket_unlocked(false);
755 its_lock.unlock();
756
757 // wait_until_sent interprets "no error" as timeout.
758 // Therefore call it with an error.
759 wait_until_sent(boost::asio::error::operation_aborted);
760 return;
761 }
762 }
763 } while (has_full_message && _recv_buffer_size);
764 if (its_iteration_gap) {
765 // Copy incomplete message to front for next receive_cbk iteration
766 for (size_t i = 0; i < _recv_buffer_size; ++i) {
767 (*_recv_buffer)[i] = (*_recv_buffer)[i + its_iteration_gap];
768 }
769 // Still more capacity needed after shifting everything to front?
770 if (its_missing_capacity &&
771 its_missing_capacity <= _recv_buffer->capacity() - _recv_buffer_size) {
772 its_missing_capacity = 0;
773 }
774 }
775 its_lock.unlock();
776 auto self = std::dynamic_pointer_cast< tcp_client_endpoint_impl >(shared_from_this());
777 strand_.dispatch([self, &_recv_buffer, _recv_buffer_size, its_missing_capacity](){
778 self->receive(_recv_buffer, _recv_buffer_size, its_missing_capacity);
779 });
780 } else {
781 VSOMEIP_WARNING << "tcp_client_endpoint receive_cbk: "
782 << _error.message() << "(" << std::dec << _error.value()
783 << ") local: " << get_address_port_local()
784 << " remote: " << get_address_port_remote();
785 if (_error == boost::asio::error::eof ||
786 _error == boost::asio::error::timed_out ||
787 _error == boost::asio::error::bad_descriptor ||
788 _error == boost::asio::error::connection_reset) {
789 if (state_ == cei_state_e::CONNECTING) {
790 VSOMEIP_WARNING << "tcp_client_endpoint receive_cbk already"
791 " restarting" << get_remote_information();
792 } else {
793 VSOMEIP_WARNING << "tcp_client_endpoint receive_cbk restarting.";
794 state_ = cei_state_e::CONNECTING;
795 shutdown_and_close_socket_unlocked(false);
796 its_lock.unlock();
797
798 // wait_until_sent interprets "no error" as timeout.
799 // Therefore call it with an error.
800 wait_until_sent(boost::asio::error::operation_aborted);
801 }
802 } else {
803 its_lock.unlock();
804 auto self = std::dynamic_pointer_cast< tcp_client_endpoint_impl >(shared_from_this());
805 strand_.dispatch([self, &_recv_buffer, _recv_buffer_size, its_missing_capacity](){
806 self->receive(_recv_buffer, _recv_buffer_size, its_missing_capacity);
807 });
808 }
809 }
810 }

921void tcp_client_endpoint_impl::send_cbk(boost::system::error_code const &_error,
922 std::size_t _bytes,
923 const message_buffer_ptr_t& _sent_msg) {

925

927 boost::system::error_code ec;
928 sent_timer_.cancel(ec);
929
930 if (!_error) {
931 if (queue_.size() > 0) {
932 queue_size_ -= queue_.front().first->size();
933 queue_.pop_front();
934
935 update_last_departure();
936
937 if (queue_.empty())
938 is_sending_ = false;
939 else {
940 auto its_entry = get_front();
941 if (its_entry.first) {
942 auto self = std::dynamic_pointer_cast< tcp_client_endpoint_impl >(shared_from_this());
943 strand_.dispatch(
944 [self, &its_entry]() { self->send_queued(its_entry);}
945 );
946 }
947 }
948 }

950 } else {

952

957 was_not_connected_ = true;
958 } else if (_error == boost::asio::error::operation_aborted) {
959 // endpoint was stopped
960 shutdown_and_close_socket(false);
961 } else {
962 if (state_ == cei_state_e::CONNECTING) {

969 if (its_host) {
970 its_host->on_disconnect(shared_from_this());

972 restart(true);
973 }

975 method_t its_method(0);
976 client_t its_client(0);
977 session_t its_session(0);
978 if (_sent_msg && _sent_msg->size() > VSOMEIP_SESSION_POS_MAX) {

983 (*_sent_msg)[VSOMEIP_METHOD_POS_MIN],
984 (*_sent_msg)[VSOMEIP_METHOD_POS_MAX]);

986 (*_sent_msg)[VSOMEIP_CLIENT_POS_MIN],

989 (*_sent_msg)[VSOMEIP_SESSION_POS_MIN],
990 (*_sent_msg)[VSOMEIP_SESSION_POS_MAX]);
991 }
992 VSOMEIP_WARNING << "tce::send_cbk received error: "
993 << _error.message() << " (" << std::dec
994 << _error.value() << ") " << get_remote_information()
995 << " " << std::dec << queue_.size()
996 << " " << std::dec << queue_size_ << " ("
997 << std::hex << std::setw(4) << std::setfill('0') << its_client <<"): ["
