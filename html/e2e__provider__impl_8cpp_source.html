35template<typename value_t>
36value_t read_value_from_config(const std::shared_ptr<vsomeip_v3::cfg::e2e> &_config,
37 const std::string &_name,
38 value_t _default_value = value_t()) {
39
40 if (_config && _config->custom_parameters.count(_name)) {
41
42 std::stringstream its_converter;
43 std::string its_value(_config->custom_parameters[_name]);
44
45 if (its_value.size() > 1 && its_value[0] == '0' && its_value[1] == 'x') {
46 its_converter << std::hex << its_value;
47 } else {
48 its_converter << std::dec << its_value;
49 }
50
51 value_t its_converted_value;
52 its_converter >> its_converted_value;
53 return its_converted_value;
54 }
55
56 return _default_value;
57}

76bool e2e_provider_impl::add_configuration(std::shared_ptr<cfg::e2e> config)
77{
78 if (config->profile == "CRC8" || config->profile == "P01") {
79 process_e2e_profile<profile01::profile_config, profile01::profile_01_checker, profile01::protector>(config);
80 return true;
81 }
82
83 if (config->profile == "CRC32" || config->profile == "CSTM") {
84 process_e2e_profile<profile_custom::profile_config, profile_custom::profile_custom_checker, profile_custom::protector>(config);
85 return true;
86 }
87
88 if (config->profile == "P04") {
89 process_e2e_profile<profile04::profile_config, profile04::profile_04_checker, profile04::protector>(config);
90 return true;
91 }
92
93 if (config->profile == "P05") {
94 process_e2e_profile<profile05::profile_config, profile05::profile_05_checker, profile05::protector>(config);
95 return true;
96 }
97
98 if (config->profile == "P07") {
99 process_e2e_profile<profile07::profile_config, profile07::profile_07_checker, profile07::protector>(config);
100 return true;
101 }
102
103 return false;
104}

144template<>
145vsomeip_v3::e2e::profile01::profile_config
146e2e_provider_impl::make_e2e_profile_config(const std::shared_ptr<cfg::e2e> &_config) {
147 uint16_t data_id = read_value_from_config<uint16_t>(_config, "data_id");
148 uint16_t crc_offset = read_value_from_config<uint16_t>(_config, "crc_offset");
149 uint16_t data_length = read_value_from_config<uint16_t>(_config, "data_length");
150
151 // counter field behind CRC8
152 uint16_t counter_offset = read_value_from_config<uint16_t>(_config, "counter_offset", 8);
153
154 // data id nibble behind 4 bit counter value
155 uint16_t data_id_nibble_offset = read_value_from_config<uint16_t>(_config, "data_id_nibble_offset", 12);
156
157 e2e::profile01::p01_data_id_mode data_id_mode =
158 static_cast<e2e::profile01::p01_data_id_mode>(
159 read_value_from_config<uint16_t>(_config, "data_id_mode"));
160
161 return e2e::profile01::profile_config(crc_offset, data_id, data_id_mode,
162 data_length, counter_offset, data_id_nibble_offset);
163}

165template<>
166vsomeip_v3::e2e::profile04::profile_config
167e2e_provider_impl::make_e2e_profile_config(const std::shared_ptr<cfg::e2e> &_config) {
168
169 uint32_t data_id = read_value_from_config<uint32_t>(_config, "data_id");
170
171 size_t offset = read_value_from_config<size_t>(_config, "crc_offset");
172 if (offset % 8)
173 VSOMEIP_ERROR << "Offset in E2E P04 configuration must be multiple of 8"
174 " (" << offset << ")";
175 offset /= 8;
176
177 size_t min_data_length = read_value_from_config<size_t>(_config,
178 "min_data_length", 0);
179
180 size_t max_data_length = read_value_from_config<size_t>(_config,
181 "max_data_length", size_t(0xffff));
182
183 uint16_t max_delta_counter = read_value_from_config<uint16_t>(_config,
184 "max_delta_counter", uint16_t(0xffff));
185
186 return e2e::profile04::profile_config(data_id, offset,
187 min_data_length, max_data_length, max_delta_counter);
188}
