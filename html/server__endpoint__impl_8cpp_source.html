50template<typename Protocol>
51void server_endpoint_impl<Protocol>::prepare_stop(
52 const endpoint::prepare_stop_handler_t &_handler, service_t _service) {
53
54 std::lock_guard<std::mutex> its_lock(mutex_);
55 bool queued_train(false);
56 std::vector<target_data_iterator_type> its_erased;
57 boost::system::error_code ec;
58
59 if (_service == ANY_SERVICE) { // endpoint is shutting down completely
60 endpoint_impl<Protocol>::sending_blocked_ = true;
61 for (auto t = targets_.begin(); t != targets_.end(); t++) {
62 auto its_train (t->second.train_);
63 // cancel dispatch timer
64 t->second.dispatch_timer_->cancel(ec);
65 if (its_train->buffer_->size() > 0) {
66 if (queue_train(t, its_train))
67 its_erased.push_back(t);
68 queued_train = true;
69 }
70 }
71 } else {
72 for (auto t = targets_.begin(); t != targets_.end(); t++) {
73 auto its_train(t->second.train_);
74 for (auto const& passenger_iter : its_train->passengers_) {
75 if (passenger_iter.first == _service) {
76 // cancel dispatch timer
77 t->second.dispatch_timer_->cancel(ec);
78 // TODO: Queue all(!) trains here...
79 if (queue_train(t, its_train))
80 its_erased.push_back(t);
81 queued_train = true;
82 break;
83 }
84 }
85 }
86 }
87
88 for (const auto t : its_erased)
89 targets_.erase(t);
90
91 if (!queued_train) {
92 if (_service == ANY_SERVICE) {
93 if (std::all_of(targets_.begin(), targets_.end(),
94 [&](const typename target_data_type::value_type &_t)
95 { return _t.second.queue_.empty(); })) {
96 // nothing was queued and all queues are empty -> ensure cbk is called
97 auto ptr = this->shared_from_this();
98 endpoint_impl<Protocol>::io_.post([ptr, _handler, _service](){
99 _handler(ptr, _service);
100 });
101 } else {
102 prepare_stop_handlers_[_service] = _handler;
103 }
104 } else {
105 // check if any of the queues contains a message of to be stopped service
106 bool found_service_msg(false);
107 for (const auto &t : targets_) {
108 for (const auto &q : t.second.queue_) {
109 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
110 (*q.first)[VSOMEIP_SERVICE_POS_MIN],
111 (*q.first)[VSOMEIP_SERVICE_POS_MAX]);
112 if (its_service == _service) {
113 found_service_msg = true;
114 break;
115 }
116 }
117 if (found_service_msg) {
118 break;
119 }
120 }
121 if (found_service_msg) {
122 prepare_stop_handlers_[_service] = _handler;
123 } else { // no messages of the to be stopped service are or have been queued
124 auto ptr = this->shared_from_this();
125 endpoint_impl<Protocol>::io_.post([ptr, _handler, _service](){
126 _handler(ptr, _service);
127 });
128 }
129 }
130 } else {
131 prepare_stop_handlers_[_service] = _handler;
132 }
133}

241template<typename Protocol>
242bool server_endpoint_impl<Protocol>::send_intern(
243 endpoint_type _target, const byte_t *_data, uint32_t _size) {
244
245 switch (check_message_size(_data, _size, _target)) {
246 case endpoint_impl<Protocol>::cms_ret_e::MSG_WAS_SPLIT:
247 return true;
248 break;
249 case endpoint_impl<Protocol>::cms_ret_e::MSG_TOO_BIG:
250 return false;
251 break;
252 case endpoint_impl<Protocol>::cms_ret_e::MSG_OK:
253 default:
254 break;
255 }
256 if (!prepare_stop_handlers_.empty()) {
257 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
258 _data[VSOMEIP_SERVICE_POS_MIN], _data[VSOMEIP_SERVICE_POS_MAX]);
259 if (prepare_stop_handlers_.find(its_service) != prepare_stop_handlers_.end()) {
260 const method_t its_method = VSOMEIP_BYTES_TO_WORD(
261 _data[VSOMEIP_METHOD_POS_MIN], _data[VSOMEIP_METHOD_POS_MAX]);
262 const client_t its_client = VSOMEIP_BYTES_TO_WORD(
263 _data[VSOMEIP_CLIENT_POS_MIN], _data[VSOMEIP_CLIENT_POS_MAX]);
264 const session_t its_session = VSOMEIP_BYTES_TO_WORD(
265 _data[VSOMEIP_SESSION_POS_MIN], _data[VSOMEIP_SESSION_POS_MAX]);
266 VSOMEIP_WARNING << "server_endpoint::send: Service is stopping, ignoring message: ["
267 << std::hex << std::setfill('0')
268 << std::setw(4) << its_service << "."
269 << std::setw(4) << its_method << "."
270 << std::setw(4) << its_client << "."
271 << std::setw(4) << its_session << "]";
272 return false;
273 }
274 }
275
276 const auto its_target_iterator = find_or_create_target_unlocked(_target);
277 auto &its_data(its_target_iterator->second);
278
279 bool must_depart(false);
280 auto its_now(std::chrono::steady_clock::now());
281
282#if 0
283 std::stringstream msg;
284 msg << "sei::send_intern: ";
285 for (uint32_t i = 0; i < _size; i++)
286 msg << std::hex << std::setw(2) << std::setfill('0') << (int)_data[i] << " ";
287 VSOMEIP_DEBUG << msg.str();
288#endif
289 // STEP 1: Check queue limit
290 if (!check_queue_limit(_data, _size, its_data.queue_size_)) {
291 return false;
292 }
293 // STEP 2: Cancel the dispatch timer
294 cancel_dispatch_timer(its_target_iterator);
295
296 // STEP 3: Get configured timings
297 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
298 _data[VSOMEIP_SERVICE_POS_MIN], _data[VSOMEIP_SERVICE_POS_MAX]);
299 const method_t its_method = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_METHOD_POS_MIN],
300 _data[VSOMEIP_METHOD_POS_MAX]);
301
302 std::chrono::nanoseconds its_debouncing(0), its_retention(0);
303 if (its_service != VSOMEIP_SD_SERVICE && its_method != VSOMEIP_SD_METHOD) {
304 get_configured_times_from_endpoint(its_service, its_method,
305 &its_debouncing, &its_retention);
306 }
307
308 // STEP 4: Check if the passenger enters an empty train
309 const std::pair<service_t, method_t> its_identifier
310 = std::make_pair(its_service, its_method);
311 if (its_data.train_->passengers_.empty()) {
312 its_data.train_->departure_ = its_now + its_retention;
313 } else {
314 if (its_data.train_->passengers_.end()
315 != its_data.train_->passengers_.find(its_identifier)) {
316 must_depart = true;
317 } else {
318 // STEP 5: Check whether the current message fits into the current train
319 if (its_data.train_->buffer_->size() + _size > endpoint_impl<Protocol>::max_message_size_) {
320 must_depart = true;
321 } else {
322 // STEP 6: Check debouncing time
323 if (its_debouncing > its_data.train_->minimal_max_retention_time_) {
324 // train's latest departure would already undershot new
325 // passenger's debounce time
326 must_depart = true;
327 } else {
328 if (its_now + its_debouncing > its_data.train_->departure_) {
329 // train departs earlier as the new passenger's debounce
330 // time allows
331 must_depart = true;
332 } else {
333 // STEP 7: Check maximum retention time
334 if (its_retention < its_data.train_->minimal_debounce_time_) {
335 // train's earliest departure would already exceed
336 // the new passenger's retention time.
337 must_depart = true;
338 } else {
339 if (its_now + its_retention < its_data.train_->departure_) {
340 its_data.train_->departure_ = its_now + its_retention;
341 }
342 }
343 }
344 }
345 }
346 }
347 }
348
349 // STEP 8: if necessary, send current buffer and create a new one
350 if (must_depart) {
351 // STEP 8.1: check if debounce time would be undershot here if the train
352 // departs. Block sending until train is allowed to depart.
353 schedule_train(its_data);
354
355 its_data.train_ = std::make_shared<train>();
356 its_data.train_->departure_ = its_now + its_retention;
357 }
358
359 // STEP 9: insert current message buffer
360 its_data.train_->buffer_->insert(its_data.train_->buffer_->end(), _data, _data + _size);
361 its_data.train_->passengers_.insert(its_identifier);
362 // STEP 9.1: update the trains minimal debounce time if necessary
363 if (its_debouncing < its_data.train_->minimal_debounce_time_) {
364 its_data.train_->minimal_debounce_time_ = its_debouncing;
365 }
366 // STEP 9.2: update the trains minimal maximum retention time if necessary
367 if (its_retention < its_data.train_->minimal_max_retention_time_) {
368 its_data.train_->minimal_max_retention_time_ = its_retention;
369 }
370
371 // STEP 10: restart timer with current departure time
372 start_dispatch_timer(its_target_iterator, its_now);
373
374 return true;
375}

377template<typename Protocol>
378void server_endpoint_impl<Protocol>::send_segments(
379 const tp::tp_split_messages_t &_segments, std::uint32_t _separation_time,
380 const endpoint_type &_target) {
381
382 if (_segments.size() == 0)
383 return;
384
385 const auto its_target_iterator = find_or_create_target_unlocked(_target);
386 auto &its_data = its_target_iterator->second;
387
388 auto its_now(std::chrono::steady_clock::now());
389
390 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
391 (*(_segments[0]))[VSOMEIP_SERVICE_POS_MIN], (*(_segments[0]))[VSOMEIP_SERVICE_POS_MAX]);
392 const method_t its_method = VSOMEIP_BYTES_TO_WORD(
393 (*(_segments[0]))[VSOMEIP_METHOD_POS_MIN], (*(_segments[0]))[VSOMEIP_METHOD_POS_MAX]);
394
395 std::chrono::nanoseconds its_debouncing(0), its_retention(0);
396 if (its_service != VSOMEIP_SD_SERVICE && its_method != VSOMEIP_SD_METHOD) {
397 get_configured_times_from_endpoint(its_service, its_method,
398 &its_debouncing, &its_retention);
399 }
400 // update the trains minimal debounce time if necessary
401 if (its_debouncing < its_data.train_->minimal_debounce_time_) {
402 its_data.train_->minimal_debounce_time_ = its_debouncing;
403 }
404 // update the trains minimal maximum retention time if necessary
405 if (its_retention < its_data.train_->minimal_max_retention_time_) {
406 its_data.train_->minimal_max_retention_time_ = its_retention;
407 }
408 // We only need to respect the debouncing. There is no need to wait for further
409 // messages as we will send several now anyway.
410 if (!its_data.train_->passengers_.empty()) {
411 schedule_train(its_data);
412 its_data.train_->departure_ = its_now + its_retention;
413 }
414
415 for (const auto &s : _segments) {
416 its_data.queue_.emplace_back(std::make_pair(s, _separation_time));
417 its_data.queue_size_ += s->size();
418 }
419
420 if (!its_data.is_sending_ && !its_data.queue_.empty()) { // no writing in progress
421 // respect minimal debounce time
422 schedule_train(its_data);
423 // ignore retention time and send immediately as the train is full anyway
424 (void)send_queued(its_target_iterator);
425 }
426}

428template<typename Protocol>
429typename server_endpoint_impl<Protocol>::target_data_iterator_type
430server_endpoint_impl<Protocol>::find_or_create_target_unlocked(endpoint_type _target) {
431
432 auto its_iterator = targets_.find(_target);
433 if (its_iterator == targets_.end()) {
434 auto its_result = targets_.emplace(
435 std::make_pair(_target, endpoint_data_type(this->io_)));
436 its_iterator = its_result.first;
437 }
438
439 return its_iterator;

442template<typename Protocol>
443void server_endpoint_impl<Protocol>::schedule_train(endpoint_data_type &_data) {
444
445 if (_data.has_last_departure_) {
446 if (_data.last_departure_ + _data.train_->minimal_debounce_time_
447 > _data.train_->departure_) {
448 _data.train_->departure_ = _data.last_departure_
449 + _data.train_->minimal_debounce_time_;
450 }
451 }
452
453 _data.dispatched_trains_[_data.train_->departure_]
454 .push_back(_data.train_);
455}

457template<typename Protocol>
458typename endpoint_impl<Protocol>::cms_ret_e server_endpoint_impl<Protocol>::check_message_size(
459 const std::uint8_t * const _data, std::uint32_t _size,
460 const endpoint_type& _target) {
461 typename endpoint_impl<Protocol>::cms_ret_e ret(endpoint_impl<Protocol>::cms_ret_e::MSG_OK);
462 if (endpoint_impl<Protocol>::max_message_size_ != MESSAGE_SIZE_UNLIMITED
463 && _size > endpoint_impl<Protocol>::max_message_size_) {
464 if (endpoint_impl<Protocol>::is_supporting_someip_tp_ && _data != nullptr) {
465 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
466 _data[VSOMEIP_SERVICE_POS_MIN],
467 _data[VSOMEIP_SERVICE_POS_MAX]);
468 const method_t its_method = VSOMEIP_BYTES_TO_WORD(
469 _data[VSOMEIP_METHOD_POS_MIN],
470 _data[VSOMEIP_METHOD_POS_MAX]);
471 if (tp_segmentation_enabled(its_service, its_method)) {
472 instance_t its_instance = this->get_instance(its_service);
473 if (its_instance != 0xFFFF) {
474 std::uint16_t its_max_segment_length;
475 std::uint32_t its_separation_time;
476
477 this->configuration_->get_tp_configuration(
478 its_service, its_instance, its_method, false,
479 its_max_segment_length, its_separation_time);
480 send_segments(tp::tp::tp_split_message(_data, _size,
481 its_max_segment_length), its_separation_time, _target);
482 return endpoint_impl<Protocol>::cms_ret_e::MSG_WAS_SPLIT;
483 }
484 }
485 }
486 VSOMEIP_ERROR << "sei::send_intern: Dropping to big message (" << _size
487 << " Bytes). Maximum allowed message size is: "
488 << endpoint_impl<Protocol>::max_message_size_ << " Bytes.";
489 ret = endpoint_impl<Protocol>::cms_ret_e::MSG_TOO_BIG;
490 }
491 return ret;
492}

494template<typename Protocol>
495bool server_endpoint_impl<Protocol>::check_queue_limit(const uint8_t *_data, std::uint32_t _size,
496 std::size_t _current_queue_size) const {
497 if (endpoint_impl<Protocol>::queue_limit_ != QUEUE_SIZE_UNLIMITED
498 && _current_queue_size + _size
499 > endpoint_impl<Protocol>::queue_limit_) {
500 service_t its_service(0);
501 method_t its_method(0);
502 client_t its_client(0);
503 session_t its_session(0);
504 if (_size >= VSOMEIP_SESSION_POS_MAX) {
505 // this will yield wrong IDs for local communication as the commands
506 // are prepended to the actual payload
507 // it will print:
508 // (lowbyte service ID + highbyte methoid)
509 // [(Command + lowerbyte sender's client ID).
510 // highbyte sender's client ID + lowbyte command size.
511 // lowbyte methodid + highbyte vsomeip length]
512 its_service = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_SERVICE_POS_MIN],
513 _data[VSOMEIP_SERVICE_POS_MAX]);
514 its_method = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_METHOD_POS_MIN],
515 _data[VSOMEIP_METHOD_POS_MAX]);
516 its_client = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_CLIENT_POS_MIN],
517 _data[VSOMEIP_CLIENT_POS_MAX]);
518 its_session = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_SESSION_POS_MIN],
519 _data[VSOMEIP_SESSION_POS_MAX]);
520 }
521 VSOMEIP_ERROR << "sei::send_intern: queue size limit (" << std::dec
522 << endpoint_impl<Protocol>::queue_limit_
523 << ") reached. Dropping message ("
524 << std::hex << std::setfill('0')
525 << std::setw(4) << its_client << "): ["
526 << std::setw(4) << its_service << "."
527 << std::setw(4) << its_method << "."
528 << std::setw(4) << its_session << "]"
529 << " queue_size: " << std::dec << _current_queue_size
530 << " data size: " << _size;
531 return false;
532 }
533 return true;
534}

536template<typename Protocol>
537bool server_endpoint_impl<Protocol>::queue_train(
538 target_data_iterator_type _it, const std::shared_ptr<train> &_train) {
539
540 bool must_erase(false);
541
542 auto &its_data = _it->second;
543 its_data.queue_size_ += _train->buffer_->size();
544 its_data.queue_.emplace_back(_train->buffer_, 0);
545
546 if (!its_data.is_sending_) { // no writing in progress
547 must_erase = send_queued(_it);
548 }
549
550 return must_erase;
551}

553template<typename Protocol>
554bool server_endpoint_impl<Protocol>::flush(endpoint_type _key) {
555
556 bool has_queued(true);
557 bool is_current_train(true);
558
559 std::lock_guard<std::mutex> its_lock(mutex_);
560
561 auto it = targets_.find(_key);
562 if (it == targets_.end())
563 return false;
564
565 auto &its_data = it->second;
566 auto its_train(its_data.train_);
567 if (!its_data.dispatched_trains_.empty()) {
568
569 auto its_dispatched = its_data.dispatched_trains_.begin();
570 if (its_dispatched->first <= its_train->departure_) {
571
572 is_current_train = false;
573 if (!its_dispatched->second.empty()) {
574 its_train = its_dispatched->second.front();
575 its_dispatched->second.pop_front();
576 if (its_dispatched->second.empty()) {
577
578 its_data.dispatched_trains_.erase(its_dispatched);
579 }
580 }
581 }
582 }
583
584 if (!its_train->buffer_->empty()) {
585
586 queue_train(it, its_train);
587
588 // Reset current train if necessary
589 if (is_current_train) {
590 its_train->reset();
591 }
592 } else {
593 has_queued = false;
594 }
595
596 if (!is_current_train || !its_data.dispatched_trains_.empty()) {
597
598 auto its_now(std::chrono::steady_clock::now());
599 start_dispatch_timer(it, its_now);
600 }
601
602 return has_queued;
603}

611template<typename Protocol>
612void server_endpoint_impl<Protocol>::send_cbk(
613 const endpoint_type _key,
614 boost::system::error_code const &_error, std::size_t _bytes) {
615 (void)_bytes;
616
617 // Helper
618 auto check_if_all_msgs_for_stopped_service_are_sent = [&]() {
619 bool found_service_msg(false);
620 service_t its_stopped_service(ANY_SERVICE);
621 for (auto stp_hndlr_iter = prepare_stop_handlers_.begin();
622 stp_hndlr_iter != prepare_stop_handlers_.end();) {
623 its_stopped_service = stp_hndlr_iter->first;
624 if (its_stopped_service == ANY_SERVICE) {
625 ++stp_hndlr_iter;
626 continue;
627 }
628 for (const auto& t : targets_) {
629 for (const auto& e : t.second.queue_ ) {
630 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
631 (*e.first)[VSOMEIP_SERVICE_POS_MIN],
632 (*e.first)[VSOMEIP_SERVICE_POS_MAX]);
633 if (its_service == its_stopped_service) {
634 found_service_msg = true;
635 break;
636 }
637 }
638 if (found_service_msg) {
639 break;
640 }
641 }
642 if (found_service_msg) {
643 ++stp_hndlr_iter;
644 } else { // all messages of the to be stopped service have been sent
645 auto handler = stp_hndlr_iter->second;
646 auto ptr = this->shared_from_this();
647 endpoint_impl<Protocol>::io_.post([ptr, handler, its_stopped_service](){
648 handler(ptr, its_stopped_service);
649 });
650 stp_hndlr_iter = prepare_stop_handlers_.erase(stp_hndlr_iter);
651 }
652 }
653 };
654
655 auto check_if_all_queues_are_empty = [&](){
656 if (prepare_stop_handlers_.size() > 1) {
657 // before the endpoint was stopped completely other
658 // prepare_stop_handlers have been queued ensure to call them as well
659 check_if_all_msgs_for_stopped_service_are_sent();
660 }
661 if (std::all_of(targets_.begin(), targets_.end(),
662 [&](const typename target_data_type::value_type &_t)
663 { return _t.second.queue_.empty(); })) {
664 // all outstanding response have been sent.
665 auto found_cbk = prepare_stop_handlers_.find(ANY_SERVICE);
666 if (found_cbk != prepare_stop_handlers_.end()) {
667 auto handler = found_cbk->second;
668 auto ptr = this->shared_from_this();
669 endpoint_impl<Protocol>::io_.post([ptr, handler](){
670 handler(ptr, ANY_SERVICE);
671 });
672 prepare_stop_handlers_.erase(found_cbk);
673 }
674 }
675 };
676
677
678 std::lock_guard<std::mutex> its_lock(mutex_);
679
680 auto it = targets_.find(_key);
681 if (it == targets_.end())
682 return;
683
684 auto& its_data = it->second;
685
686 boost::system::error_code ec;
687 its_data.sent_timer_.cancel(ec);
688
689 // Extracts some information for logging puposes.
690 //
691 // TODO(brunoldsilva): Code like this is used in a lot of places. It might be worth moving this
692 // into a proper function.
693 auto parse_message_ids = [] (
694 const message_buffer_ptr_t& buffer,
695 service_t& its_service,
696 method_t& its_method,
697 client_t& its_client,
698 session_t& its_session
699 ) {
700 if (buffer && buffer->size() > VSOMEIP_SESSION_POS_MAX) {
701 its_service = VSOMEIP_BYTES_TO_WORD(
702 (*buffer)[VSOMEIP_SERVICE_POS_MIN],
703 (*buffer)[VSOMEIP_SERVICE_POS_MAX]);
704 its_method = VSOMEIP_BYTES_TO_WORD(
705 (*buffer)[VSOMEIP_METHOD_POS_MIN],
706 (*buffer)[VSOMEIP_METHOD_POS_MAX]);
707 its_client = VSOMEIP_BYTES_TO_WORD(
708 (*buffer)[VSOMEIP_CLIENT_POS_MIN],
709 (*buffer)[VSOMEIP_CLIENT_POS_MAX]);
710 its_session = VSOMEIP_BYTES_TO_WORD(
711 (*buffer)[VSOMEIP_SESSION_POS_MIN],
712 (*buffer)[VSOMEIP_SESSION_POS_MAX]);
713 }
714 };
715
716 message_buffer_ptr_t its_buffer;
717 if (its_data.queue_.size()) {
718 its_buffer = its_data.queue_.front().first;
719 }
720 service_t its_service(0);
721 method_t its_method(0);
722 client_t its_client(0);
723 session_t its_session(0);
724
725 if (!_error) {
726 const std::size_t payload_size = its_data.queue_.front().first->size();
727 if (payload_size <= its_data.queue_size_) {
728 its_data.queue_size_ -= payload_size;
729 } else {
730 parse_message_ids(its_buffer, its_service, its_method, its_client, its_session);
731 VSOMEIP_WARNING << __func__ << ": prevented queue_size underflow. queue_size: "
732 << its_data.queue_size_ << " payload_size: " << payload_size << " payload: ("
733 << std::hex << std::setw(4) << std::setfill('0') << its_client <<"): ["
734 << std::hex << std::setw(4) << std::setfill('0') << its_service << "."
735 << std::hex << std::setw(4) << std::setfill('0') << its_method << "."
736 << std::hex << std::setw(4) << std::setfill('0') << its_session << "]";
737 its_data.queue_size_ = 0;
738 }
739 its_data.queue_.pop_front();
740
741 update_last_departure(its_data);
742
743 if (!prepare_stop_handlers_.empty() && !endpoint_impl<Protocol>::sending_blocked_) {
744 // only one service instance is stopped
745 check_if_all_msgs_for_stopped_service_are_sent();
746 }
747
748 if (!its_data.queue_.empty()) {
749 (void)send_queued(it);
750 } else {
751 if (!prepare_stop_handlers_.empty() && endpoint_impl<Protocol>::sending_blocked_) {
752 // endpoint is shutting down completely
753 cancel_dispatch_timer(it);
754 targets_.erase(it);
755 check_if_all_queues_are_empty();
756 } else
757 its_data.is_sending_ = false;
758 }
759 } else {
760 // error: sending of outstanding responses isn't started again
761 // delete remaining outstanding responses
762 parse_message_ids(its_buffer, its_service, its_method, its_client, its_session);
763 VSOMEIP_WARNING << "sei::send_cbk received error: " << _error.message()
764 << " (" << std::dec << _error.value() << ") "
765 << get_remote_information(it) << " "
766 << std::dec << its_data.queue_.size() << " "
767 << its_data.queue_size_ << " ("
768 << std::hex << std::setfill('0')
769 << std::setw(4) << its_client << "): ["
770 << std::setw(4) << its_service << "."
771 << std::setw(4) << its_method << "."
772 << std::setw(4) << its_session << "]";
773 cancel_dispatch_timer(it);
774 targets_.erase(it);
775 if (!prepare_stop_handlers_.empty()) {
776 if (endpoint_impl<Protocol>::sending_blocked_) {
777 // endpoint is shutting down completely, ensure to call
778 // prepare_stop_handlers even in error cases
779 check_if_all_queues_are_empty();
780 } else {
781 // only one service instance is stopped
782 check_if_all_msgs_for_stopped_service_are_sent();
783 }
784 }
785 }

824template<typename Protocol>
825void server_endpoint_impl<Protocol>::start_dispatch_timer(
826 target_data_iterator_type _it,
827 const std::chrono::steady_clock::time_point &_now) {
828
829 auto &its_data = _it->second;
830 std::shared_ptr<train> its_train(its_data.train_);
831
832 if (!its_data.dispatched_trains_.empty()) {
833
834 auto its_dispatched = its_data.dispatched_trains_.begin();
835 if (its_dispatched->first < its_train->departure_) {
836
837 its_train = its_dispatched->second.front();
838 }
839 }
840
841 std::chrono::nanoseconds its_offset;
842 if (its_train->departure_ > _now) {
843
844 its_offset = std::chrono::duration_cast<std::chrono::nanoseconds>(
845 its_train->departure_ - _now);
846 } else { // already departure time
847
848 its_offset = std::chrono::nanoseconds::zero();
849 }
850
851#if defined(__linux__) || defined(ANDROID) || defined(__QNX__)
852 its_data.dispatch_timer_->expires_from_now(its_offset);
853#else
854 its_data.dispatch_timer_->expires_from_now(
855 std::chrono::duration_cast<
856 std::chrono::steady_clock::duration>(its_offset));
857#endif
858 its_data.dispatch_timer_->async_wait(
859 std::bind(&server_endpoint_impl<Protocol>::flush_cbk,
860 this->shared_from_this(), _it->first, std::placeholders::_1));
861}
