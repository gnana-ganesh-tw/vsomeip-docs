83std::shared_ptr<endpoint> endpoint_manager_base::create_local_server(
84 const std::shared_ptr<routing_host> &_routing_host) {
85 std::shared_ptr<endpoint> its_server_endpoint;
86 std::stringstream its_path;
87 its_path << utility::get_base_path(configuration_->get_network())
88 << std::hex << rm_->get_client();
89 const client_t its_client = rm_->get_client();
90
91#if defined(__linux__) || defined(ANDROID) || defined(__QNX__)
92 if (is_local_routing_) {
93 if (-1 == ::unlink(its_path.str().c_str()) && errno != ENOENT) {
94 VSOMEIP_ERROR << "endpoint_manager_base::init_receiver unlink failed ("
95 << its_path.str() << "): "<< std::strerror(errno);
96 }
97 try {
98 its_server_endpoint = std::make_shared<local_uds_server_endpoint_impl>(
99 shared_from_this(), _routing_host,
100# if VSOMEIP_BOOST_VERSION < 106600
101 boost::asio::local::stream_protocol_ext::endpoint(its_path.str()),
102# else
103 boost::asio::local::stream_protocol::endpoint(its_path.str()),
104# endif
105 io_,
106 configuration_, false);
107
108 VSOMEIP_INFO << __func__ << ": Listening @ " << its_path.str();
109
110 } catch (const std::exception &e) {
111 VSOMEIP_ERROR << "Local UDS server endpoint creation failed. Client "
112 << std::hex << std::setw(4) << std::setfill('0') << its_client
113 << " Path: " << its_path.str()
114 << " Reason: " << e.what();
115 }
116 } else {
117#else
118 {
119#endif
120 std::lock_guard<std::mutex> its_lock(create_local_server_endpoint_mutex_);
121 ::unlink(its_path.str().c_str());
122 port_t its_port;
123 std::set<port_t> its_used_ports;
124 auto its_address = configuration_->get_routing_guest_address();
125 uint32_t its_current_wait_time { 0 };
126 while (get_local_server_port(its_port, its_used_ports) && !its_server_endpoint) {
127 try {
128 its_server_endpoint = std::make_shared<local_tcp_server_endpoint_impl>(
129 shared_from_this(), _routing_host,
130 boost::asio::ip::tcp::endpoint(its_address, its_port),
131 io_,
132 configuration_, false);
133
134 VSOMEIP_INFO << __func__ << ": Listening @ "
135 << its_address.to_string() << ":" << std::dec << its_port;
136
137 if (rm_->is_routing_manager())
138 local_port_ = port_t(configuration_->get_routing_host_port() + 1);
139 else
140 local_port_ = port_t(its_port + 1);
141 VSOMEIP_INFO << __func__ << ": Connecting to other clients from "
142 << its_address.to_string() << ":" << std::dec << local_port_;
143
144 rm_->set_sec_client_port(local_port_);
145
146 } catch (const boost::system::system_error &e) {
147 if (e.code() == boost::asio::error::address_in_use) {
148 its_used_ports.insert(its_port);
149 } else {
150 its_current_wait_time += LOCAL_TCP_PORT_WAIT_TIME;
151 if (its_current_wait_time > LOCAL_TCP_PORT_MAX_WAIT_TIME)
152 break;
153
154 std::this_thread::sleep_for(
155 std::chrono::milliseconds(LOCAL_TCP_PORT_WAIT_TIME));
156 }
157 }
158 }
159
160 if (!its_server_endpoint) {
161 VSOMEIP_ERROR << "Local TCP server endpoint creation failed. Client "
162 << std::hex << std::setw(4) << std::setfill('0') << its_client
163 << " Reason: No local port available!";
164 } else {
165 rm_->add_guest(its_client, its_address, its_port);
166 }
167 }
168
169 return its_server_endpoint;
170}

223void
224endpoint_manager_base::log_client_states() const {
225 std::vector<std::pair<client_t, size_t> > its_client_queue_sizes;
226 std::stringstream its_log;
227
228 {
229 std::lock_guard<std::mutex> its_lock(local_endpoint_mutex_);
230 for (const auto &e : local_endpoints_) {
231 size_t its_queue_size = e.second->get_queue_size();
232 if (its_queue_size > VSOMEIP_DEFAULT_QUEUE_WARN_SIZE) {
233 its_client_queue_sizes.push_back(
234 std::make_pair(e.first, its_queue_size));
235 }
236 }
237 }
238
239 std::sort(its_client_queue_sizes.begin(), its_client_queue_sizes.end(),
240 [](const std::pair<client_t, size_t> &_a,
241 const std::pair<client_t, size_t> &_b) {
242 return (_a.second > _b.second);
243 });
244
245 size_t its_max(std::min(size_t(10), its_client_queue_sizes.size()));
246 its_log << std::setfill('0');
247 for (size_t i = 0; i < its_max; i++) {
248 its_log << std::hex << std::setw(4) << its_client_queue_sizes[i].first << ":"
249 << std::dec << its_client_queue_sizes[i].second;
250 if (i < its_max-1)
251 its_log << ", ";
252 }
253
254 if (its_log.str().length() > 0)
255 VSOMEIP_WARNING << "ICQ: [" << its_log.str() << "]";
256}

258std::shared_ptr<endpoint>
259endpoint_manager_base::create_local_unlocked(client_t _client) {
260
261 std::stringstream its_path;
262 its_path << utility::get_base_path(configuration_->get_network())
263 << std::hex << _client;
264 std::shared_ptr<endpoint> its_endpoint;
265
266#if defined(__linux__) || defined(ANDROID) || defined(__QNX__)
267 if (is_local_routing_) {
268 its_endpoint = std::make_shared<local_uds_client_endpoint_impl>(
269 shared_from_this(), rm_->shared_from_this(),
270 boost::asio::local::stream_protocol::endpoint(its_path.str()),
271 io_, configuration_);
272 VSOMEIP_INFO << "Client [" << std::hex << rm_->get_client() << "] is connecting to ["
273 << std::hex << _client << "] at " << its_path.str();
274 } else {
275#else
276 {
277#endif
278 boost::asio::ip::address its_local_address, its_remote_address;
279 port_t its_remote_port;
280
281 bool is_guest = rm_->get_guest(_client, its_remote_address, its_remote_port);
282 if (is_guest) {
283 try {
284 its_local_address = configuration_->get_routing_guest_address();
285 its_endpoint = std::make_shared<local_tcp_client_endpoint_impl>(
286 shared_from_this(), rm_->shared_from_this(),
287 boost::asio::ip::tcp::endpoint(its_local_address, local_port_),
288 boost::asio::ip::tcp::endpoint(its_remote_address, its_remote_port),
289 io_, configuration_);
290
291 VSOMEIP_INFO << "Client ["
292 << std::hex << std::setw(4) << std::setfill('0') << rm_->get_client()
293 << "] @ "
294 << its_local_address.to_string() << ":" << std::dec << local_port_
295 << " is connecting to ["
296 << std::hex << std::setw(4) << std::setfill('0') << _client << "] @ "
297 << its_remote_address.to_string() << ":" << std::dec << its_remote_port;
298
299 } catch (...) {
300 }
301 } else {
302 VSOMEIP_ERROR << __func__
303 << ": Cannot get guest address of client ["
304 << std::hex << std::setw(4) << std::setfill('0')
305 << _client << "]";
306 }
307 }
308
309 if (its_endpoint) {
310 // Messages sent to the VSOMEIP_ROUTING_CLIENT are meant to be routed to
311 // external devices. Therefore, its local endpoint must not be found by
312 // a call to find_local. Thus it must not be inserted to the list of local
313 // clients.
314 if (_client != VSOMEIP_ROUTING_CLIENT) {
315 local_endpoints_[_client] = its_endpoint;
316 }
317 rm_->register_client_error_handler(_client, its_endpoint);
318 }
319
320 return its_endpoint;
321}

341bool
342endpoint_manager_base::get_local_server_port(port_t &_port,
343 const std::set<port_t> &_used_ports) const {
344
345#define SERVER_PORT_OFFSET 2
346
347#ifdef _WIN32
348 uid_t its_uid { ANY_UID };
349 gid_t its_gid { ANY_GID };
350#else
351 uid_t its_uid { getuid() };
352 gid_t its_gid { getgid() };
353#endif
354
355 auto its_port_ranges = configuration_->get_routing_guest_ports(
356 its_uid, its_gid);
357
358 if (its_port_ranges.empty()) {
359 VSOMEIP_WARNING << __func__ << ": No configured port ranges for uid/gid="
360 << std::dec << its_uid << '/' << its_gid;
361 }
362
363 for (const auto &its_range : its_port_ranges) {
364 for (int r = its_range.first; r < its_range.second;
365 r += SERVER_PORT_OFFSET) {
366
367 if (_used_ports.find(port_t(r)) == _used_ports.end()
368 && r != configuration_->get_routing_host_port()) {
369
370 _port = port_t(r);
371 return true;
372 }
373 }
374 }
375
376 return false;
377}
