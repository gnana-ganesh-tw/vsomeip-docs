20void
21distribute_security_policies_command::serialize(std::vector<byte_t> &_buffer,
22 error_e &_error) const {
23
24 size_t its_size(COMMAND_HEADER_SIZE +
25 std::min(payload_.size(), size_t(std::numeric_limits<uint32_t>::max())));
26
27 if (its_size > std::numeric_limits<command_size_t>::max()) {
28
29 _error = error_e::ERROR_MAX_COMMAND_SIZE_EXCEEDED;
30 return;
31 }
32
33 // resize buffer
34 _buffer.resize(its_size);
35
36 // set size
37 size_ = static_cast<command_size_t>(its_size - COMMAND_HEADER_SIZE);
38
39 // serialize header
40 command::serialize(_buffer, _error);
41 if (_error != error_e::ERROR_OK)
42 return;
43
44 // serialize (add) payload
45 size_t its_offset(COMMAND_HEADER_SIZE);
46 if (payload_.empty()) { // No policy data (--> set_payloads was not called)
47 std::memset(&_buffer[its_offset], 0, sizeof(uint32_t));
48 } else {

53void
54distribute_security_policies_command::deserialize(const std::vector<byte_t> &_buffer,
55 error_e &_error) {
56
57 if (COMMAND_HEADER_SIZE + sizeof(uint32_t) > _buffer.size()) {
58
59 _error = error_e::ERROR_NOT_ENOUGH_BYTES;
60 return;
61 }
62
63 // deserialize header
64 command::deserialize(_buffer, _error);
65 if (_error != error_e::ERROR_OK)
66 return;
67
68 // deserialize payload
69 size_t its_offset(COMMAND_HEADER_SIZE);
70 uint32_t its_policies_count;
71 std::memcpy(&its_policies_count, &_buffer[its_offset],
72 sizeof(its_policies_count));
73 its_offset += sizeof(its_policies_count);
74
75 for (uint32_t i = 0; i < its_policies_count; i++) {
76
77 uint32_t its_policy_size;
78
79 // Check that the buffer contains the full policy size
80 if (its_offset + sizeof(its_policy_size) > _buffer.size()) {
81
82 policies_.clear();
83 _error = error_e::ERROR_NOT_ENOUGH_BYTES;
84 return;
85 }
86
87 std::memcpy(&its_policy_size, &_buffer[its_offset],
88 sizeof(its_policy_size));
89 its_offset += sizeof(its_policy_size);
90
91 // Check that the buffer contains the full policy
92 if (its_offset + its_policy_size > _buffer.size()) {
93
94 policies_.clear();
95 _error = error_e::ERROR_NOT_ENOUGH_BYTES;
96 return;
97 }
98
99 const byte_t *its_policy_data = &_buffer[its_offset];
100
101 // set offset to the next policy
102 its_offset += its_policy_size;
103
104 auto its_policy = std::make_shared<policy>();
105 if (its_policy_size == 0
106 || !its_policy->deserialize(its_policy_data, its_policy_size)) {
107
108 _error = error_e::ERROR_UNKNOWN;
109 policies_.clear();
110 return;
111 }
112
113 policies_.insert(its_policy);
114 }

123void
124distribute_security_policies_command::set_payloads(
125 const std::map<uint32_t, std::shared_ptr<payload> > &_payloads) {
126
127 uint32_t its_count(uint32_t(_payloads.size()));
128 for (uint32_t i = 0; i < sizeof(its_count); ++i) {
129 payload_.push_back(
130 reinterpret_cast<const byte_t*>(&its_count)[i]);
131 }
132
133 for (const auto &its_uid_gid : _payloads) {
134 // policy payload length including gid and uid
135 std::uint32_t its_length(uint32_t(its_uid_gid.second->get_length()));
136 for (uint32_t i = 0; i < sizeof(its_length); ++i) {
137 payload_.push_back(
138 reinterpret_cast<const byte_t*>(&its_length)[i]);
139 }
140 // payload
141 payload_.insert(payload_.end(), its_uid_gid.second->get_data(),
142 its_uid_gid.second->get_data() + its_uid_gid.second->get_length());
143 }
144}
