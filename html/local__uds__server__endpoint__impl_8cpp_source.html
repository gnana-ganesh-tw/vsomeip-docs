258void local_uds_server_endpoint_impl::accept_cbk(
259 const connection::ptr& _connection, boost::system::error_code const &_error) {
260 if (_error != boost::asio::error::bad_descriptor
261 && _error != boost::asio::error::operation_aborted
262 && _error != boost::asio::error::no_descriptors) {
263 start();
264 } else if (_error == boost::asio::error::no_descriptors) {
265 VSOMEIP_ERROR << "local_usd_server_endpoint_impl::accept_cbk: "
266 << _error.message() << " (" << std::dec << _error.value()
267 << ") Will try to accept again in 1000ms";
268 std::shared_ptr<boost::asio::steady_timer> its_timer =
269 std::make_shared<boost::asio::steady_timer>(io_,
270 std::chrono::milliseconds(1000));
271 auto its_ep = std::dynamic_pointer_cast<local_uds_server_endpoint_impl>(
272 shared_from_this());
273 its_timer->async_wait([its_timer, its_ep]
274 (const boost::system::error_code& _error) {
275 if (!_error) {
276 its_ep->start();
277 }
278 });
279 }
280
281 if (!_error) {
282#ifndef __QNX__
283 auto its_host = endpoint_host_.lock();
284 client_t its_client = 0;
285 std::string its_client_host;
286 vsomeip_sec_client_t its_sec_client;
287
288 its_sec_client.port = VSOMEIP_SEC_PORT_UNUSED;
289 its_sec_client.user = ANY_UID;
290 its_sec_client.group = ANY_GID;
291
292 socket_type &its_socket = _connection->get_socket();
293 if (auto creds = credentials::receive_credentials(its_socket.native_handle())) {
294
295 its_client = std::get<0>(*creds);
296 its_client_host = std::get<3>(*creds);
297
298 its_sec_client.user = std::get<1>(*creds);
299 its_sec_client.group = std::get<2>(*creds);
300 } else {
301 VSOMEIP_WARNING << "vSomeIP Security: Client 0x" << std::hex << its_host->get_client()
302 << " is rejecting new connection because client credentials couldn't be received!";
303 boost::system::error_code er;
304 its_socket.shutdown(its_socket.shutdown_both, er);
305 its_socket.close(er);
306 return;
307 }
308
309 if (its_host && configuration_->is_security_enabled()) {
310 if (!configuration_->check_routing_credentials(its_client, &its_sec_client)) {
311 VSOMEIP_WARNING << "vSomeIP Security: Rejecting new connection with routing manager client ID 0x"
312 << std::hex << its_client
313 << " uid/gid= " << std::dec
314 << its_sec_client.user << "/"
315 << its_sec_client.group
316 << " because passed credentials do not match with routing manager credentials!";
317 boost::system::error_code er;
318 its_socket.shutdown(its_socket.shutdown_both, er);
319 its_socket.close(er);
320 return;
321 }
322
323 if (is_routing_endpoint_) {
324 // rm_impl receives VSOMEIP_CLIENT_UNSET initially -> check later
325 _connection->set_bound_sec_client(its_sec_client);
326 _connection->set_bound_client_host(its_client_host);
327 } else {
328 {
329 std::lock_guard<std::mutex> its_connection_lock(connections_mutex_);
330 // rm_impl receives VSOMEIP_CLIENT_UNSET initially -> check later
331 const auto found_client = connections_.find(its_client);
332 if (found_client != connections_.end()) {
333 VSOMEIP_WARNING << "vSomeIP Security: Client 0x" << std::hex
334 << its_host->get_client() << " is rejecting new connection with client ID 0x"
335 << its_client << " uid/gid= " << std::dec
336 << its_sec_client.user << "/"
337 << its_sec_client.group
338 << " because of already existing connection using same client ID";
339 boost::system::error_code er;
340 its_socket.shutdown(its_socket.shutdown_both, er);
341 its_socket.close(er);
342 return;
343 }
344 }
345
346 // Add to known clients (loads new config if needed)
347 std::shared_ptr<routing_host> its_routing_host = routing_host_.lock();
348 its_routing_host->add_known_client(its_client, its_client_host);
349
350 if (!policy_manager_impl::get()->check_credentials(its_client, &its_sec_client)) {
351 VSOMEIP_WARNING << "vSomeIP Security: Client 0x" << std::hex
352 << its_host->get_client() << " received client credentials from client 0x"
353 << its_client << " which violates the security policy : uid/gid="
354 << std::dec << its_sec_client.user << "/"
355 << its_sec_client.group;
356 boost::system::error_code er;
357 its_socket.shutdown(its_socket.shutdown_both, er);
358 its_socket.close(er);
359 return;
360 }
361 // rm_impl receives VSOMEIP_CLIENT_UNSET initially -> set later
362 _connection->set_bound_client(its_client);
363 _connection->set_bound_client_host(its_client_host);
364 add_connection(its_client, _connection);
365 }
366 } else {
367 policy_manager_impl::get()->store_client_to_sec_client_mapping(its_client, &its_sec_client);
368 policy_manager_impl::get()->store_sec_client_to_client_mapping(&its_sec_client, its_client);
369
370 if (!is_routing_endpoint_) {
371 std::shared_ptr<routing_host> its_routing_host = routing_host_.lock();
372 its_routing_host->add_known_client(its_client, its_client_host);
373 _connection->set_bound_client(its_client);
374 }
375 _connection->set_bound_client_host(its_client_host);
376 }
377#endif
378 _connection->start();
379 }

437void local_uds_server_endpoint_impl::connection::start() {
438 std::lock_guard<std::mutex> its_lock(socket_mutex_);
439 if (socket_.is_open()) {
440 const std::size_t its_capacity(recv_buffer_.capacity());
441 if (recv_buffer_size_ > its_capacity) {
442 VSOMEIP_ERROR << __func__ << "Received buffer size is greater than the buffer capacity!"
443 << " recv_buffer_size_: " << recv_buffer_size_
444 << " its_capacity: " << its_capacity;
445 return;
446 }
447 size_t left_buffer_size = its_capacity - recv_buffer_size_;
448 try {
449 if (missing_capacity_) {
450 if (missing_capacity_ > MESSAGE_SIZE_UNLIMITED) {
451 VSOMEIP_ERROR << "Missing receive buffer capacity exceeds allowed maximum!";
452 return;
453 }
454 const std::size_t its_required_capacity(recv_buffer_size_ + missing_capacity_);
455 if (its_capacity < its_required_capacity) {
456 // Make the resize to its_required_capacity
457 recv_buffer_.reserve(its_required_capacity);
458 recv_buffer_.resize(its_required_capacity, 0x0);
459 }
460 left_buffer_size = missing_capacity_;
461 missing_capacity_ = 0;
462 } else if (buffer_shrink_threshold_
463 && shrink_count_ > buffer_shrink_threshold_
464 && recv_buffer_size_ == 0) {
465 // In this case, make the resize to recv_buffer_size_initial_
466 recv_buffer_.resize(recv_buffer_size_initial_, 0x0);
467 recv_buffer_.shrink_to_fit();
468 // And set buffer_size to recv_buffer_size_initial_, the same of our resize
469 left_buffer_size = recv_buffer_size_initial_;
470 shrink_count_ = 0;
471 }
472 } catch (const std::exception &e) {
473 handle_recv_buffer_exception(e);
474 // don't start receiving again
475 return;
476 }
477
478 is_stopped_ = false;
479#if VSOMEIP_BOOST_VERSION >= 106600
480 auto its_storage = std::make_shared<local_endpoint_receive_op::storage>(
481 socket_,
482 std::bind(
483 &local_uds_server_endpoint_impl::connection::receive_cbk,
484 shared_from_this(),
485 std::placeholders::_1,
486 std::placeholders::_2,
487 std::placeholders::_3,
488 std::placeholders::_4
489 ),
490 &recv_buffer_[recv_buffer_size_],
491 left_buffer_size,
492 std::numeric_limits<std::uint32_t>::max(),
493 std::numeric_limits<std::uint32_t>::max(),
494 std::numeric_limits<std::size_t>::min()
495 );
496
497 socket_.async_wait(socket_type::wait_read, local_endpoint_receive_op::receive_cb(its_storage));
498#else
499 socket_.async_receive(
500 boost::asio::buffer(&recv_buffer_[recv_buffer_size_], left_buffer_size),
501 std::bind(
502 &local_uds_server_endpoint_impl::connection::receive_cbk,
503 shared_from_this(),
504 std::placeholders::_1,
505 std::placeholders::_2,
506 std::placeholders::_3,
507 std::placeholders::_4
508 )
509 );
510#endif
511 }

609void local_uds_server_endpoint_impl::connection::receive_cbk(
610 boost::system::error_code const &_error, std::size_t _bytes,
611 std::uint32_t const &_uid, std::uint32_t const &_gid)
612{
613 std::shared_ptr<local_uds_server_endpoint_impl> its_server(server_.lock());
614 if (!its_server) {
615 VSOMEIP_TRACE << "local_uds_server_endpoint_impl::connection::receive_cbk "
616 " couldn't lock server_";
617 return;
618 }
619 std::shared_ptr<routing_host> its_host = its_server->routing_host_.lock();
620 if (!its_host)
621 return;
622
623 std::shared_ptr<vsomeip_v3::configuration> its_config = its_server->configuration_;
624 if (_error == boost::asio::error::operation_aborted) {
625 if (its_server->is_routing_endpoint_ &&
626 bound_client_ != VSOMEIP_CLIENT_UNSET && its_config) {
627 utility::release_client_id(its_config->get_network(),
628 bound_client_);
629 set_bound_client(VSOMEIP_CLIENT_UNSET);
630 }
631
632 // connection was stopped
633 return;
634 }
635
636 bool is_error(false);
637 std::size_t its_start = 0;
638 std::size_t its_end = 0;
639 std::size_t its_iteration_gap = 0;
640 std::uint32_t its_command_size = 0;
641
642 if (!_error && 0 < _bytes) {
643#if 0
644 std::stringstream msg;
645 msg << "lse::c<" << this << ">rcb: ";
646 for (std::size_t i = 0; i < _bytes + recv_buffer_size_; i++)
647 msg << std::setw(2) << std::setfill('0') << std::hex
648 << (int) (recv_buffer_[i]) << " ";
649 VSOMEIP_INFO << msg.str();
650#endif
651
652 if (recv_buffer_size_ + _bytes < recv_buffer_size_) {
653 VSOMEIP_ERROR << "receive buffer overflow in local server endpoint ~> abort!";
654 return;
655 }
656 recv_buffer_size_ += _bytes;
657
658 bool message_is_empty(false);
659 bool found_message(false);
660
661 do {
662 found_message = false;
663 message_is_empty = false;
664
665 its_start = 0 + its_iteration_gap;
666 if (its_start + 3 < its_start) {
667 VSOMEIP_ERROR << "buffer overflow in local server endpoint ~> abort!";
668 return;
669 }
670 while (its_start + 3 < recv_buffer_size_ + its_iteration_gap &&
671 (recv_buffer_[its_start] != 0x67 ||
672 recv_buffer_[its_start+1] != 0x37 ||
673 recv_buffer_[its_start+2] != 0x6d ||
674 recv_buffer_[its_start+3] != 0x07)) {
675 its_start++;
676 }
677
678 if (its_start + 3 == recv_buffer_size_ + its_iteration_gap) {
679 message_is_empty = true;
680 } else {
681 its_start += 4;
682 }
683
684 if (!message_is_empty) {
685 if (its_start + protocol::COMMAND_POSITION_SIZE + 3 < recv_buffer_size_ + its_iteration_gap) {
686 its_command_size = VSOMEIP_BYTES_TO_LONG(
687 recv_buffer_[its_start + protocol::COMMAND_POSITION_SIZE+3],
688 recv_buffer_[its_start + protocol::COMMAND_POSITION_SIZE+2],
689 recv_buffer_[its_start + protocol::COMMAND_POSITION_SIZE+1],
690 recv_buffer_[its_start + protocol::COMMAND_POSITION_SIZE]);
691
692 its_end = its_start + protocol::COMMAND_POSITION_SIZE + 3 + its_command_size;
693 } else {
694 its_end = its_start;
695 }
696 if (its_command_size && max_message_size_ != MESSAGE_SIZE_UNLIMITED
697 && its_command_size > max_message_size_) {
698 std::lock_guard<std::mutex> its_lock(socket_mutex_);
699 VSOMEIP_ERROR << "Received a local message which exceeds "
700 << "maximum message size (" << std::dec << its_command_size
701 << ") aborting! local: " << get_path_local() << " remote: "
702 << get_path_remote();
703 recv_buffer_.resize(recv_buffer_size_initial_, 0x0);
704 recv_buffer_.shrink_to_fit();
705 return;
706 }
707 if (its_end + 3 < its_end) {
708 VSOMEIP_ERROR << "buffer overflow in local server endpoint ~> abort!";
709 return;
710 }
711 while (its_end + 3 < recv_buffer_size_ + its_iteration_gap &&
712 (recv_buffer_[its_end] != 0x07 ||
713 recv_buffer_[its_end+1] != 0x6d ||
714 recv_buffer_[its_end+2] != 0x37 ||
715 recv_buffer_[its_end+3] != 0x67)) {
716 its_end ++;
717 }
718 if (its_end + 4 < its_end) {
719 VSOMEIP_ERROR << "buffer overflow in local server endpoint ~> abort!";
720 return;
721 }
722 // check if we received a full message
723 if (recv_buffer_size_ + its_iteration_gap < its_end + 4
724 || recv_buffer_[its_end] != 0x07
725 || recv_buffer_[its_end+1] != 0x6d
726 || recv_buffer_[its_end+2] != 0x37
727 || recv_buffer_[its_end+3] != 0x67) {
728 // command (1 Byte) + version (2 Byte) + client id (2 Byte)
729 // + command size (4 Byte) + data itself + stop tag (4 byte)
730 // = 13 Bytes not covered in command size.
731 // If need to change the recv_buffer_, change the value of missing_capacity_
732 // in this if/else, otherwise it is 0
733 if (its_start - its_iteration_gap + its_command_size
734 + protocol::COMMAND_HEADER_SIZE + protocol::TAG_SIZE > recv_buffer_size_) {
735 missing_capacity_ =
736 std::uint32_t(its_start) - std::uint32_t(its_iteration_gap)
737 + its_command_size + std::uint32_t(protocol::COMMAND_HEADER_SIZE + protocol::TAG_SIZE)
738 - std::uint32_t(recv_buffer_size_);
739 } else if (recv_buffer_size_ < protocol::COMMAND_HEADER_SIZE + protocol::TAG_SIZE) {
740 // to little data to read out the command size
741 // minimal amount of data needed to read out command size = header + tag size
742 missing_capacity_ = static_cast<std::uint32_t>(
743 protocol::COMMAND_HEADER_SIZE + protocol::TAG_SIZE - recv_buffer_size_);
744 } else {
745 std::stringstream local_msg;
746 local_msg << std::setfill('0') << std::hex;
747 for (std::size_t i = its_iteration_gap;
748 i < recv_buffer_size_ + its_iteration_gap &&
749 i - its_iteration_gap < 32; i++) {
750 local_msg << std::setw(2) << (int) recv_buffer_[i] << " ";
751 }
752 VSOMEIP_ERROR << "lse::c<" << this
753 << ">rcb: recv_buffer_size is: " << std::dec
754 << recv_buffer_size_ << " but couldn't read "
755 "out command size. recv_buffer_capacity: "
756 << std::dec << recv_buffer_.capacity()
757 << " its_iteration_gap: " << std::dec
758 << its_iteration_gap << " bound client: 0x"
759 << std::hex << bound_client_ << " buffer: "
760 << local_msg.str();
761 recv_buffer_size_ = 0;
762 missing_capacity_ = 0;
763 its_iteration_gap = 0;
764 message_is_empty = true;
765 }
766 }
767 }
768
769 if (!message_is_empty &&
770 its_end + 3 < recv_buffer_size_ + its_iteration_gap) {
771
772 if (its_server->is_routing_endpoint_
773 && recv_buffer_[its_start] == byte_t(protocol::id_e::ASSIGN_CLIENT_ID)) {
774 client_t its_client = its_server->assign_client(
775 &recv_buffer_[its_start], uint32_t(its_end - its_start));
776
777 if (its_config && its_config->is_security_enabled()) {
778 // Add to known clients (loads new config if needed)
779 its_host->add_known_client(its_client, get_bound_client_host());
780
781 if (!its_server->add_connection(its_client, shared_from_this())) {
782 VSOMEIP_WARNING << std::hex << "Client 0x" << its_host->get_client()
783 << " is rejecting new connection with client ID 0x" << its_client
784 << " uid/gid= " << std::dec
785 << sec_client_.user << "/"
786 << sec_client_.group
787 << " because of already existing connection using same client ID";
788 stop();
789 return;
790 } else if (!policy_manager_impl::get()->check_credentials(
791 its_client, &sec_client_)) {
792 VSOMEIP_WARNING << std::hex << "Client 0x" << its_host->get_client()
793 << " received client credentials from client 0x" << its_client
794 << " which violates the security policy : uid/gid="
795 << std::dec << sec_client_.user << "/"
796 << sec_client_.group;
797 its_server->remove_connection(its_client);
798 utility::release_client_id(its_config->get_network(),
799 its_client);
800 stop();
801 return;
802 }
803 else {
804 set_bound_client(its_client);
805 }
806 } else {
807 set_bound_client(its_client);
808 its_host->add_known_client(its_client, get_bound_client_host());
809 its_server->add_connection(its_client, shared_from_this());
810 }
811 its_server->send_client_identifier(its_client);
812 assigned_client_ = true;
813 } else if (!its_server->is_routing_endpoint_ || assigned_client_) {
814
815 vsomeip_sec_client_t its_sec_client{};
816
817 its_sec_client.port = VSOMEIP_SEC_PORT_UNUSED;
818 its_sec_client.user = _uid;
819 its_sec_client.group = _gid;
820
821 its_host->on_message(&recv_buffer_[its_start],
822 uint32_t(its_end - its_start), its_server.get(),
823 false, bound_client_, &its_sec_client);
824 } else {
825 VSOMEIP_WARNING << std::hex << "Client 0x" << its_host->get_client()
826 << " didn't receive VSOMEIP_ASSIGN_CLIENT as first message";
827 }
828 #if 0
829 std::stringstream local_msg;
830 local_msg << "lse::c<" << this << ">rcb::thunk: ";
831 for (std::size_t i = its_start; i < its_end; i++)
832 local_msg << std::setw(2) << std::setfill('0') << std::hex
833 << (int) recv_buffer_[i] << " ";
834 VSOMEIP_INFO << local_msg.str();
835 #endif
836 calculate_shrink_count();
837 recv_buffer_size_ -= (its_end + 4 - its_iteration_gap);
838 missing_capacity_ = 0;
839 its_command_size = 0;
840 found_message = true;
841 its_iteration_gap = its_end + 4;
842 } else {
843 if (its_iteration_gap) {
844 // Message not complete and not in front of the buffer!
845 // Copy last part to front for consume in future receive_cbk call!
846 for (size_t i = 0; i < recv_buffer_size_; ++i) {
847 recv_buffer_[i] = recv_buffer_[i + its_iteration_gap];
848 }
849 // Still more capacity needed after shifting everything to front?
850 if (missing_capacity_ &&
851 missing_capacity_ <= recv_buffer_.capacity() - recv_buffer_size_) {
852 missing_capacity_ = 0;
853 }
854 } else if (message_is_empty) {
855 VSOMEIP_ERROR << "Received garbage data.";
856 is_error = true;
857 }
858 }
859 } while (recv_buffer_size_ > 0 && found_message);
860 }
861
862 if (is_stopped_
863 || _error == boost::asio::error::eof
864 || _error == boost::asio::error::connection_reset
865 || is_error) {
866 shutdown_and_close();
867 its_server->remove_connection(bound_client_);
868 policy_manager_impl::get()->remove_client_to_sec_client_mapping(bound_client_);
869 } else if (_error != boost::asio::error::bad_descriptor) {
870 start();
871 }
