175 inline bool is_connected(client_t _source, client_t _sink) const {
176
177 auto find_source = connection_matrix_.find(_source);
178 if (find_source != connection_matrix_.end())
179 return (find_source->second.find(_sink)
180 != find_source->second.end());
181
182 return false;
183 }
184 inline void add_connection(client_t _source, client_t _sink) {
185
186 connection_matrix_[_source].insert(_sink);
187 }
188 inline void remove_connection(client_t _source, client_t _sink) {
189
190 auto find_source = connection_matrix_.find(_source);
191 if (find_source != connection_matrix_.end())
192 find_source->second.erase(_sink);
193 }
194 inline void remove_source(client_t _source) {
195
196 connection_matrix_.erase(_source);
