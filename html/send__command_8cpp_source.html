77void
78send_command::serialize(std::vector<byte_t> &_buffer,
79 error_e &_error) const {
80
81 size_t its_size(COMMAND_HEADER_SIZE + sizeof(instance_)
82 + sizeof(is_reliable_) + sizeof(status_)
83 + sizeof(target_) + message_.size());
84
85 if (its_size > std::numeric_limits<command_size_t>::max()) {
86
87 _error = error_e::ERROR_MAX_COMMAND_SIZE_EXCEEDED;
88 return;
89 }
90
91 // resize buffer
92 _buffer.resize(its_size);
93
94 // set size
95 size_ = static_cast<command_size_t>(its_size - COMMAND_HEADER_SIZE);
96
97 // serialize header
98 command::serialize(_buffer, _error);
99 if (_error != error_e::ERROR_OK)
100 return;
101
102 // serialize payload
103 size_t its_offset(COMMAND_POSITION_PAYLOAD);
104 std::memcpy(&_buffer[its_offset], &instance_, sizeof(instance_));
105 its_offset += sizeof(instance_);
106 _buffer[its_offset] = static_cast<byte_t>(is_reliable_);
107 its_offset += sizeof(is_reliable_);
108 _buffer[its_offset] = static_cast<byte_t>(status_);
109 its_offset += sizeof(status_);
110 std::memcpy(&_buffer[its_offset], &target_, sizeof(target_));
111 its_offset += sizeof(target_);
112 std::memcpy(&_buffer[its_offset], &message_[0], message_.size());
113}

115void
116send_command::deserialize(const std::vector<byte_t> &_buffer,
117 error_e &_error) {
118
119 size_t its_size(COMMAND_HEADER_SIZE + sizeof(instance_)
120 + sizeof(is_reliable_) + sizeof(status_)
121 + sizeof(target_));
122
123 if (its_size > _buffer.size()) {
124
125 _error = error_e::ERROR_NOT_ENOUGH_BYTES;
126 return;
127 }
128
129 // deserialize header
130 command::deserialize(_buffer, _error);
131 if (_error != error_e::ERROR_OK)
132 return;
133
134 // deserialize payload
135 size_t its_offset(COMMAND_POSITION_PAYLOAD);
136 std::memcpy(&instance_, &_buffer[its_offset], sizeof(instance_));
137 its_offset += sizeof(instance_);
138 is_reliable_ = static_cast<bool>(_buffer[its_offset]);
139 its_offset += sizeof(is_reliable_);
140 status_ = static_cast<uint8_t>(_buffer[its_offset]);
141 its_offset += sizeof(status_);
142 std::memcpy(&target_, &_buffer[its_offset], sizeof(target_));
143 its_offset += sizeof(target_);
144 message_.resize(_buffer.size() - its_offset);
145 std::memcpy(&message_[0], &_buffer[its_offset], message_.size());
146}
