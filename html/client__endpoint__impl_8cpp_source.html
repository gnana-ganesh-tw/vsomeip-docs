118template<typename Protocol>
119void client_endpoint_impl<Protocol>::stop() {
120 {
121 std::lock_guard<std::recursive_mutex> its_lock(mutex_);
122 endpoint_impl<Protocol>::sending_blocked_ = true;
123 // delete unsent messages
124 queue_.clear();
125 queue_size_ = 0;
126 }
127 {
128 std::lock_guard<std::mutex> its_lock(connect_timer_mutex_);
129 boost::system::error_code ec;
130 connect_timer_.cancel(ec);
131 }
132 connect_timeout_ = VSOMEIP_DEFAULT_CONNECT_TIMEOUT;
133
134 // bind to strand as stop() might be called from different thread
135 strand_.dispatch(std::bind(&client_endpoint_impl::shutdown_and_close_socket,
136 this->shared_from_this(),
137 false)
138 );
139}

180template<typename Protocol>
181bool client_endpoint_impl<Protocol>::send(const uint8_t *_data, uint32_t _size) {
182
183 std::lock_guard<std::recursive_mutex> its_lock(mutex_);
184 bool must_depart(false);
185 auto its_now(std::chrono::steady_clock::now());
186
187#if 0
188 std::stringstream msg;
189 msg << "cei::send: ";
190 for (uint32_t i = 0; i < _size; i++)
191 msg << std::hex << std::setw(2) << std::setfill('0') << (int)_data[i] << " ";
192 VSOMEIP_DEBUG << msg.str();
193#endif
194
195 if (endpoint_impl<Protocol>::sending_blocked_ ||
196 !check_queue_limit(_data, _size)) {
197 return false;
198 }
199 switch (check_message_size(_data, _size)) {
200 case endpoint_impl<Protocol>::cms_ret_e::MSG_WAS_SPLIT:
201 return true;
202 break;
203 case endpoint_impl<Protocol>::cms_ret_e::MSG_TOO_BIG:
204 return false;
205 break;
206 case endpoint_impl<Protocol>::cms_ret_e::MSG_OK:
207 default:
208 break;
209 }
210
211 // STEP 1: Cancel dispatch timer
212 cancel_dispatch_timer();
213
214 // STEP 3: Get configured timings
215 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
216 _data[VSOMEIP_SERVICE_POS_MIN], _data[VSOMEIP_SERVICE_POS_MAX]);
217 const method_t its_method = VSOMEIP_BYTES_TO_WORD(
218 _data[VSOMEIP_METHOD_POS_MIN], _data[VSOMEIP_METHOD_POS_MAX]);
219 std::chrono::nanoseconds its_debouncing(0), its_retention(0);
220 get_configured_times_from_endpoint(its_service, its_method,
221 &its_debouncing, &its_retention);
222
223 // STEP 4: Check if the passenger enters an empty train
224 const std::pair<service_t, method_t> its_identifier = std::make_pair(
225 its_service, its_method);
226 if (train_->passengers_.empty()) {
227 train_->departure_ = its_now + its_retention; // latest possible
228 } else {
229 // STEP 4.1: Check whether the current train already contains the message
230 if (train_->passengers_.end() != train_->passengers_.find(its_identifier)) {
231 must_depart = true;
232 } else {
233 // STEP 5: Check whether the current message fits into the current train
234 if (train_->buffer_->size() + _size > endpoint_impl<Protocol>::max_message_size_) {
235 must_depart = true;
236 } else {
237 // STEP 6: Check debouncing time
238 if (its_debouncing > train_->minimal_max_retention_time_) {
239 // train's latest departure would already undershot new
240 // passenger's debounce time
241 must_depart = true;
242 } else {
243 if (its_now + its_debouncing > train_->departure_) {
244 // train departs earlier as the new passenger's debounce
245 // time allows
246 must_depart = true;
247 } else {
248 // STEP 7: Check maximum retention time
249 if (its_retention < train_->minimal_debounce_time_) {
250 // train's earliest departure would already exceed
251 // the new passenger's retention time.
252 must_depart = true;
253 } else {
254 if (its_now + its_retention < train_->departure_) {
255 train_->departure_ = its_now + its_retention;
256 }
257 }
258 }
259 }
260 }
261 }
262 }
263
264 // STEP 8: if necessary, send current buffer and create a new one
265 if (must_depart) {
266 // STEP 8.1: check if debounce time would be undershot here if the train
267 // departs. Schedule departure of current train and create a new one.
268 schedule_train();
269
270 train_ = std::make_shared<train>();
271 train_->departure_ = its_now + its_retention;
272 }
273
274 // STEP 9: insert current message buffer
275 train_->buffer_->insert(train_->buffer_->end(), _data, _data + _size);
276 train_->passengers_.insert(its_identifier);
277 // STEP 9.1: update the trains minimal debounce time if necessary
278 if (its_debouncing < train_->minimal_debounce_time_) {
279 train_->minimal_debounce_time_ = its_debouncing;
280 }
281 // STEP 9.2: update the trains minimal maximum retention time if necessary
282 if (its_retention < train_->minimal_max_retention_time_) {
283 train_->minimal_max_retention_time_ = its_retention;
284 }
285
286 // STEP 10: restart dispatch timer with next departure time
287 start_dispatch_timer(its_now);
288
289 return true;
290}

292template<typename Protocol>
293void client_endpoint_impl<Protocol>::send_segments(
294 const tp::tp_split_messages_t &_segments, std::uint32_t _separation_time) {
295
296 auto its_now(std::chrono::steady_clock::now());
297
298 if (_segments.size() == 0) {
299 return;
300 }
301
302 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
303 (*(_segments[0]))[VSOMEIP_SERVICE_POS_MIN],
304 (*(_segments[0]))[VSOMEIP_SERVICE_POS_MAX]);
305 const method_t its_method = VSOMEIP_BYTES_TO_WORD(
306 (*(_segments[0]))[VSOMEIP_METHOD_POS_MIN],
307 (*(_segments[0]))[VSOMEIP_METHOD_POS_MAX]);
308 std::chrono::nanoseconds its_debouncing(0), its_retention(0);
309 get_configured_times_from_endpoint(its_service, its_method,
310 &its_debouncing, &its_retention);
311 // update the trains minimal debounce time if necessary
312 if (its_debouncing < train_->minimal_debounce_time_) {
313 train_->minimal_debounce_time_ = its_debouncing;
314 }
315 // update the trains minimal maximum retention time if necessary
316 if (its_retention < train_->minimal_max_retention_time_) {
317 train_->minimal_max_retention_time_ = its_retention;
318 }
319
320 // We only need to respect the debouncing. There is no need to wait for further
321 // messages as we will send several now anyway.
322 if (!train_->passengers_.empty()) {
323 schedule_train();
324 train_->departure_ = its_now + its_retention;
325 }
326
327 for (const auto& s : _segments) {
328 queue_.emplace_back(std::make_pair(s, _separation_time));
329 queue_size_ += s->size();
330 }
331
332 if (!is_sending_ && !queue_.empty()) { // no writing in progress
333 // respect minimal debounce time
334 schedule_train();
335 // ignore retention time and send immediately as the train is full anyway
336 auto its_entry = get_front();
337 if (its_entry.first) {
338 is_sending_ = true;
339 strand_.dispatch(std::bind(&client_endpoint_impl::send_queued,
340 this->shared_from_this(), its_entry));
341 }
342 }
343}

345template<typename Protocol>
346void client_endpoint_impl<Protocol>::schedule_train() {
347
348 if (has_last_departure_) {
349 if (last_departure_ + train_->minimal_debounce_time_ > train_->departure_) {
350 train_->departure_ = last_departure_ + train_->minimal_debounce_time_;
351 }
352 }
353
354 dispatched_trains_[train_->departure_].push_back(train_);
355}

366template<typename Protocol>
367bool client_endpoint_impl<Protocol>::flush() {
368
369 bool has_queued(true);
370 bool is_current_train(true);
371
372 std::lock_guard<std::recursive_mutex> its_lock(mutex_);
373
374 std::shared_ptr<train> its_train(train_);
375 if (!dispatched_trains_.empty()) {
376
377 auto its_dispatched = dispatched_trains_.begin();
378 if (its_dispatched->first <= its_train->departure_) {
379
380 is_current_train = false;
381 its_train = its_dispatched->second.front();
382 its_dispatched->second.pop_front();
383 if (its_dispatched->second.empty()) {
384
385 dispatched_trains_.erase(its_dispatched);
386 }
387 }
388 }
389
390 if (!its_train->buffer_->empty()) {
391
392 queue_train(its_train);
393
394 // Reset current train if necessary
395 if (is_current_train) {
396 its_train->reset();
397 }
398 } else {
399 has_queued = false;
400 }
401
402 if (!is_current_train || !dispatched_trains_.empty()) {
403
404 auto its_now(std::chrono::steady_clock::now());
405 start_dispatch_timer(its_now);
406 }
407
408 return has_queued;
409}

411template<typename Protocol>
412void client_endpoint_impl<Protocol>::connect_cbk(
413 boost::system::error_code const &_error) {
414
415 if (_error == boost::asio::error::operation_aborted
416 || endpoint_impl<Protocol>::sending_blocked_) {
417 // endpoint was stopped
418 shutdown_and_close_socket(false);
419 return;
420 }
421 std::shared_ptr<endpoint_host> its_host = this->endpoint_host_.lock();
422 if (its_host) {
423 if (_error && _error != boost::asio::error::already_connected) {
424 shutdown_and_close_socket(true);
425
426 if (state_ != cei_state_e::ESTABLISHED) {
427 state_ = cei_state_e::CLOSED;
428 its_host->on_disconnect(this->shared_from_this());
429 }
430 if (get_max_allowed_reconnects() == MAX_RECONNECTS_UNLIMITED ||
431 get_max_allowed_reconnects() >= ++reconnect_counter_) {
432 start_connect_timer();
433 } else {
434 max_allowed_reconnects_reached();
435 }
436 // Double the timeout as long as the maximum allowed is larger
437 if (connect_timeout_ < VSOMEIP_MAX_CONNECT_TIMEOUT)
438 connect_timeout_ = (connect_timeout_ << 1);
439 } else {
440 {
441 std::lock_guard<std::mutex> its_lock(connect_timer_mutex_);
442 connect_timer_.cancel();
443 }
444 connect_timeout_ = VSOMEIP_DEFAULT_CONNECT_TIMEOUT; // TODO: use config variable
445 reconnect_counter_ = 0;
446 if (was_not_connected_) {
447 was_not_connected_ = false;
448 std::lock_guard<std::recursive_mutex> its_lock(mutex_);
449 auto its_entry = get_front();
450 if (its_entry.first) {
451 is_sending_ = true;
452 strand_.dispatch(std::bind(&client_endpoint_impl::send_queued,
453 this->shared_from_this(), its_entry));
454 VSOMEIP_WARNING << __func__ << ": resume sending to: "
455 << get_remote_information();
456 }
457 }
458 if (state_ != cei_state_e::ESTABLISHED) {
459 its_host->on_connect(this->shared_from_this());
460 }
461 receive();
462 }
463 }

469 std::size_t operations_cancelled;
470 {
471 /* Need this for TCP endpoints for now because we have no
472 direct control about the point in time the connect has finished */
473 std::lock_guard<std::mutex> its_lock(connecting_timer_mutex_);

475 }
476 if (operations_cancelled != 0) {
477 connect_cbk(_error);
478 }

501template<typename Protocol>
502void client_endpoint_impl<Protocol>::send_cbk(
503 boost::system::error_code const &_error, std::size_t _bytes,
504 const message_buffer_ptr_t& _sent_msg) {
505
506 (void)_bytes;
507
508 if (!_error) {
509 std::lock_guard<std::recursive_mutex> its_lock(mutex_);
510 if (queue_.size() > 0) {
511 queue_size_ -= queue_.front().first->size();
512 queue_.pop_front();
513
514 update_last_departure();
515
516 if (queue_.empty())
517 is_sending_ = false;
518 else {
519 auto its_entry = get_front();
520 if (its_entry.first) {
521 send_queued(its_entry);
522 }
523 }
524 }
525 return;
526 } else if (_error == boost::asio::error::broken_pipe) {
527 state_ = cei_state_e::CLOSED;
528 bool stopping(false);
529 {
530 std::lock_guard<std::recursive_mutex> its_lock(mutex_);
531 stopping = endpoint_impl<Protocol>::sending_blocked_;
532 if (stopping) {
533 queue_.clear();
534 queue_size_ = 0;
535 } else {
536 service_t its_service(0);
537 method_t its_method(0);
538 client_t its_client(0);
539 session_t its_session(0);
540 if (_sent_msg && _sent_msg->size() > VSOMEIP_SESSION_POS_MAX) {
541 its_service = VSOMEIP_BYTES_TO_WORD(
542 (*_sent_msg)[VSOMEIP_SERVICE_POS_MIN],
543 (*_sent_msg)[VSOMEIP_SERVICE_POS_MAX]);
544 its_method = VSOMEIP_BYTES_TO_WORD(
545 (*_sent_msg)[VSOMEIP_METHOD_POS_MIN],
546 (*_sent_msg)[VSOMEIP_METHOD_POS_MAX]);
547 its_client = VSOMEIP_BYTES_TO_WORD(
548 (*_sent_msg)[VSOMEIP_CLIENT_POS_MIN],
549 (*_sent_msg)[VSOMEIP_CLIENT_POS_MAX]);
550 its_session = VSOMEIP_BYTES_TO_WORD(
551 (*_sent_msg)[VSOMEIP_SESSION_POS_MIN],
552 (*_sent_msg)[VSOMEIP_SESSION_POS_MAX]);
553 }
554 VSOMEIP_WARNING << "cei::send_cbk received error: "
555 << _error.message() << " (" << std::dec
556 << _error.value() << ") " << get_remote_information()
557 << " " << std::dec << queue_.size()
558 << " " << queue_size_ << " ("
559 << std::hex << std::setfill('0')
560 << std::setw(4) << its_client << "): ["
561 << std::setw(4) << its_service << "."
562 << std::setw(4) << its_method << "."
563 << std::setw(4) << its_session << "]";
564 }
565 }
566 if (!stopping) {
567 print_status();
568 }
569 was_not_connected_ = true;
570 shutdown_and_close_socket(true);
571 strand_.dispatch(std::bind(&client_endpoint_impl::connect,
572 this->shared_from_this()));
573 } else if (_error == boost::asio::error::not_connected
574 || _error == boost::asio::error::bad_descriptor
575 || _error == boost::asio::error::no_permission) {
576 state_ = cei_state_e::CLOSED;
577 if (_error == boost::asio::error::no_permission) {
578 VSOMEIP_WARNING << "cei::send_cbk received error: " << _error.message()
579 << " (" << std::dec << _error.value() << ") "
580 << get_remote_information();
581 std::lock_guard<std::recursive_mutex> its_lock(mutex_);
582 queue_.clear();
583 queue_size_ = 0;
584 }
585 was_not_connected_ = true;
586 shutdown_and_close_socket(true);
587 strand_.dispatch(std::bind(&client_endpoint_impl::connect,
588 this->shared_from_this()));
589 } else if (_error == boost::asio::error::operation_aborted) {
590 VSOMEIP_WARNING << "cei::send_cbk received error: " << _error.message();
591 // endpoint was stopped
592 endpoint_impl<Protocol>::sending_blocked_ = true;
593 shutdown_and_close_socket(false);
594 } else if (_error == boost::system::errc::destination_address_required) {
595 VSOMEIP_WARNING << "cei::send_cbk received error: " << _error.message()
596 << " (" << std::dec << _error.value() << ") "
597 << get_remote_information();
598 was_not_connected_ = true;
599 } else {
600 service_t its_service(0);
601 method_t its_method(0);
602 client_t its_client(0);
603 session_t its_session(0);
604 if (_sent_msg && _sent_msg->size() > VSOMEIP_SESSION_POS_MAX) {
605 its_service = VSOMEIP_BYTES_TO_WORD(
606 (*_sent_msg)[VSOMEIP_SERVICE_POS_MIN],
607 (*_sent_msg)[VSOMEIP_SERVICE_POS_MAX]);
608 its_method = VSOMEIP_BYTES_TO_WORD(
609 (*_sent_msg)[VSOMEIP_METHOD_POS_MIN],
610 (*_sent_msg)[VSOMEIP_METHOD_POS_MAX]);
611 its_client = VSOMEIP_BYTES_TO_WORD(
612 (*_sent_msg)[VSOMEIP_CLIENT_POS_MIN],
613 (*_sent_msg)[VSOMEIP_CLIENT_POS_MAX]);
614 its_session = VSOMEIP_BYTES_TO_WORD(
615 (*_sent_msg)[VSOMEIP_SESSION_POS_MIN],
616 (*_sent_msg)[VSOMEIP_SESSION_POS_MAX]);
617 }
618 VSOMEIP_WARNING << "cei::send_cbk received error: " << _error.message()
619 << " (" << std::dec << _error.value() << ") "
620 << get_remote_information() << " "
621 << " " << queue_.size()
622 << " " << queue_size_ << " ("
623 << std::hex << std::setfill('0')
624 << std::setw(4) << its_client << "): ["
625 << std::setw(4) << its_service << "."
626 << std::setw(4) << its_method << "."
627 << std::setw(4) << its_session << "]";
628 print_status();
629 }
630
631 std::lock_guard<std::recursive_mutex> its_lock(mutex_);
632 is_sending_ = false;
633}

651template<typename Protocol>
652void client_endpoint_impl<Protocol>::shutdown_and_close_socket_unlocked(bool _recreate_socket) {
653
654 if (socket_->is_open()) {
655#if defined(__linux__) || defined(ANDROID) || defined(__QNX__)
656 if (-1 == fcntl(socket_->native_handle(), F_GETFD)) {
657 VSOMEIP_ERROR << "cei::shutdown_and_close_socket_unlocked: socket/handle closed already '"
658 << std::string(std::strerror(errno))
659 << "' (" << errno << ") " << get_remote_information();
660 }
661#endif
662 boost::system::error_code its_error;
663 socket_->shutdown(Protocol::socket::shutdown_both, its_error);
664 socket_->close(its_error);
665 }
666 if (_recreate_socket) {
667 socket_.reset(new socket_type(endpoint_impl<Protocol>::io_));
668 }

671template<typename Protocol>
672bool client_endpoint_impl<Protocol>::get_remote_address(
673 boost::asio::ip::address &_address) const {

719template<typename Protocol>
720typename endpoint_impl<Protocol>::cms_ret_e client_endpoint_impl<Protocol>::check_message_size(
721 const std::uint8_t * const _data, std::uint32_t _size) {
722
723 typename endpoint_impl<Protocol>::cms_ret_e ret(endpoint_impl<Protocol>::cms_ret_e::MSG_OK);
724 if (endpoint_impl<Protocol>::max_message_size_ != MESSAGE_SIZE_UNLIMITED
725 && _size > endpoint_impl<Protocol>::max_message_size_) {
726 if (endpoint_impl<Protocol>::is_supporting_someip_tp_ && _data != nullptr) {
727 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
728 _data[VSOMEIP_SERVICE_POS_MIN],
729 _data[VSOMEIP_SERVICE_POS_MAX]);
730 const method_t its_method = VSOMEIP_BYTES_TO_WORD(
731 _data[VSOMEIP_METHOD_POS_MIN],
732 _data[VSOMEIP_METHOD_POS_MAX]);
733 if (tp_segmentation_enabled(its_service, its_method)) {
734 instance_t its_instance = this->get_instance(its_service);
735 if (its_instance != 0xFFFF) {
736 std::uint16_t its_max_segment_length;
737 std::uint32_t its_separation_time;
738 this->configuration_->get_tp_configuration(
739 its_service, its_instance, its_method, true,
740 its_max_segment_length, its_separation_time);
741 send_segments(tp::tp::tp_split_message(_data, _size,
742 its_max_segment_length), its_separation_time);
743 return endpoint_impl<Protocol>::cms_ret_e::MSG_WAS_SPLIT;
744 }
745 }
746 }
747 VSOMEIP_ERROR << "cei::check_message_size: Dropping to big message ("
748 << std::dec << _size << " Bytes). Maximum allowed message size is: "
749 << endpoint_impl<Protocol>::max_message_size_ << " Bytes.";
750 ret = endpoint_impl<Protocol>::cms_ret_e::MSG_TOO_BIG;
751 }
752 return ret;
753}

755template<typename Protocol>
756bool client_endpoint_impl<Protocol>::check_queue_limit(const uint8_t *_data, std::uint32_t _size) const {
757
758 if (endpoint_impl<Protocol>::queue_limit_ != QUEUE_SIZE_UNLIMITED
759 && queue_size_ + _size > endpoint_impl<Protocol>::queue_limit_) {
760 service_t its_service(0);
761 method_t its_method(0);
762 client_t its_client(0);
763 session_t its_session(0);
764 if (_size >= VSOMEIP_SESSION_POS_MAX) {
765 // this will yield wrong IDs for local communication as the commands
766 // are prepended to the actual payload
767 // it will print:
768 // (lowbyte service ID + highbyte methoid)
769 // [(Command + lowerbyte sender's client ID).
770 // highbyte sender's client ID + lowbyte command size.
771 // lowbyte methodid + highbyte vsomeip length]
772 its_service = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_SERVICE_POS_MIN],
773 _data[VSOMEIP_SERVICE_POS_MAX]);
774 its_method = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_METHOD_POS_MIN],
775 _data[VSOMEIP_METHOD_POS_MAX]);
776 its_client = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_CLIENT_POS_MIN],
777 _data[VSOMEIP_CLIENT_POS_MAX]);
778 its_session = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_SESSION_POS_MIN],
779 _data[VSOMEIP_SESSION_POS_MAX]);
780 }
781 VSOMEIP_ERROR << "cei::check_queue_limit: queue size limit (" << std::dec
782 << endpoint_impl<Protocol>::queue_limit_
783 << ") reached. Dropping message ("
784 << std::hex << std::setfill('0')
785 << std::setw(4) << its_client << "): ["
786 << std::setw(4) << its_service << "."
787 << std::setw(4) << its_method << "."
788 << std::setw(4) << its_session << "] "
789 << "queue_size: " << std::dec << queue_size_
790 << " data size: " << _size;
791 return false;
792 }
793 return true;
794}

820template<typename Protocol>
821void client_endpoint_impl<Protocol>::start_dispatch_timer(
822 const std::chrono::steady_clock::time_point &_now) {
823
824 // Choose the next train
825 std::shared_ptr<train> its_train(train_);
826 if (!dispatched_trains_.empty()) {
827
828 auto its_dispatched = dispatched_trains_.begin();
829 if (its_dispatched->first < its_train->departure_) {
830
831 its_train = its_dispatched->second.front();
832 }
833 }
834
835 std::chrono::nanoseconds its_offset;
836 if (its_train->departure_ > _now) {
837
838 its_offset = std::chrono::duration_cast<std::chrono::nanoseconds>(
839 its_train->departure_ - _now);
840 } else { // already departure time
841
842 its_offset = std::chrono::nanoseconds::zero();
843 }
844
845#if defined(__linux__) || defined(ANDROID) || defined(__QNX__)
846 dispatch_timer_.expires_from_now(its_offset);
847#else
848 dispatch_timer_.expires_from_now(
849 std::chrono::duration_cast<
850 std::chrono::steady_clock::duration>(its_offset));
851#endif
852 dispatch_timer_.async_wait(
853 std::bind(&client_endpoint_impl<Protocol>::flush_cbk,
854 this->shared_from_this(), std::placeholders::_1));
855}
