322 vsomeip_sec_client_t sec_client_;

324 bool operator<(const subscription_data_t &_other) const {
325 return (service_ < _other.service_
326 || (service_ == _other.service_
327 && instance_ < _other.instance_)
328 || (service_ == _other.service_
329 && instance_ == _other.instance_
330 && eventgroup_ < _other.eventgroup_)
331 || (service_ == _other.service_
332 && instance_ == _other.instance_
333 && eventgroup_ == _other.eventgroup_
