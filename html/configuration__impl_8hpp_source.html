41struct suppress_t {
42 service_t service;
43 instance_t instance;
44 event_t event;
45
46 inline bool operator<(const suppress_t& entry_) const {
47 if(service != entry_.service) {
48 return service < entry_.service;
49 }
50 if(instance != entry_.instance) {
51 return instance < entry_.instance;
52 }
53 if(event != entry_.event) {
54 return event < entry_.event;
55 }
56 return false;
57 }
58};
