169 struct vsomeip_sec_client_comparator_t {
170 bool operator()(const vsomeip_sec_client_t &_lhs, const vsomeip_sec_client_t &_rhs) const {
171 if (_lhs.port < _rhs.port) {
172 return true;
173 } else if (_lhs.port == _rhs.port) {
174 if (_lhs.port == VSOMEIP_SEC_PORT_UNUSED) {
175 return ((_lhs.user < _rhs.user)
176 || ((_lhs.user == _rhs.user)
177 && (_lhs.group < _rhs.group)));
178 } else {
179 return ((_lhs.host < _rhs.host)
180 || ((_lhs.host == _rhs.host)
181 && (_lhs.port < _rhs.port)));
182 }
183 }
184 return false;
185 }
186 };
