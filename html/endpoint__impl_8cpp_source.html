56template<typename Protocol>
57uint32_t endpoint_impl<Protocol>::find_magic_cookie(
58 byte_t *_buffer, size_t _size) {
59 bool is_found(false);
60 uint32_t its_offset = 0xFFFFFFFF;
61
62 uint8_t its_cookie_identifier, its_cookie_type;
63
64 if (is_client()) {
65 its_cookie_identifier =
66 static_cast<uint8_t>(MAGIC_COOKIE_SERVICE_MESSAGE);
67 its_cookie_type =
68 static_cast<uint8_t>(MAGIC_COOKIE_SERVICE_MESSAGE_TYPE);
69 } else {
70 its_cookie_identifier =
71 static_cast<uint8_t>(MAGIC_COOKIE_CLIENT_MESSAGE);
72 its_cookie_type =
73 static_cast<uint8_t>(MAGIC_COOKIE_CLIENT_MESSAGE_TYPE);
74 }
75
76 do {
77 its_offset++; // --> first loop has "its_offset = 0"
78 if (_size > its_offset + 16) {
79 is_found = (_buffer[its_offset] == 0xFF
80 && _buffer[its_offset + 1] == 0xFF
81 && _buffer[its_offset + 2] == its_cookie_identifier
82 && _buffer[its_offset + 3] == 0x00
83 && _buffer[its_offset + 4] == 0x00
84 && _buffer[its_offset + 5] == 0x00
85 && _buffer[its_offset + 6] == 0x00
86 && _buffer[its_offset + 7] == 0x08
87 && _buffer[its_offset + 8] == 0xDE
88 && _buffer[its_offset + 9] == 0xAD
89 && _buffer[its_offset + 10] == 0xBE
90 && _buffer[its_offset + 11] == 0xEF
91 && _buffer[its_offset + 12] == 0x01
92 && _buffer[its_offset + 13] == 0x01
93 && _buffer[its_offset + 14] == its_cookie_type
94 && _buffer[its_offset + 15] == 0x00);
95 } else {
96 break;
97 }
98
99 } while (!is_found);
100
101 return (is_found ? its_offset : 0xFFFFFFFF);
102}
