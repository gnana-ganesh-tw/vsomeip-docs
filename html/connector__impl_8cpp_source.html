69void connector_impl::configure(const std::shared_ptr<cfg::trace> &_configuration) {
70 if (_configuration) {
71 is_enabled_ = _configuration->is_enabled_;
72 is_sd_enabled_ = _configuration->is_sd_enabled_;
73 }
74
75 if (is_enabled_) { // No need to create filters if tracing is disabled!
76 for (auto &its_channel : _configuration->channels_) {
77 if (!add_channel(its_channel->id_, its_channel->name_)) {
78 VSOMEIP_ERROR << "Channel " << its_channel->id_
79 << " has multiple definitions.";
80 }
81 }
82
83 for (auto &its_filter : _configuration->filters_) {
84 for (auto &its_channel : its_filter->channels_) {
85 auto its_channel_ptr = get_channel_impl(its_channel);
86 if (its_channel_ptr) {
87 if (its_filter->is_range_) {
88 its_channel_ptr->add_filter(its_filter->matches_[0],
89 its_filter->matches_[1], its_filter->ftype_);
90 } else {
91 its_channel_ptr->add_filter(its_filter->matches_,
92 its_filter->ftype_);
93 }
94 }
95 }
96 }
97 }
98
99 VSOMEIP_INFO << "vsomeip tracing "
100 << (is_enabled_ ? "enabled " : "not enabled. ")
101 << ". vsomeip service discovery tracing "
102 << (is_sd_enabled_ ? "enabled " : "not enabled. ");
103}

133bool connector_impl::is_sd_message(const byte_t *_data, uint16_t _data_size) const {
134 if (VSOMEIP_METHOD_POS_MAX < _data_size) {
135 return (_data[VSOMEIP_SERVICE_POS_MIN] == 0xFF && _data[VSOMEIP_SERVICE_POS_MAX] == 0xFF &&
136 _data[VSOMEIP_METHOD_POS_MIN] == 0x81 && _data[VSOMEIP_METHOD_POS_MAX] == 0x00);
137 }
138 return false;
139}

141std::shared_ptr<channel> connector_impl::add_channel(
142 const trace_channel_t &_id, const std::string &_name) {
143 std::lock_guard<std::mutex> its_channels_lock(channels_mutex_);
144
145 // check whether we already know the requested channel
146 if (channels_.find(_id) != channels_.end())
147 return nullptr;
148
149 // create new channel
150 std::shared_ptr<channel_impl> its_channel
151 = std::make_shared<channel_impl>(_id, _name);
152
153 // add channel
154 channels_[_id] = its_channel;
155
156 // register context
157#ifdef USE_DLT
158#ifndef ANDROID
159 std::lock_guard<std::mutex> its_contexts_lock(contexts_mutex_);
160 std::shared_ptr<DltContext> its_context = std::make_shared<DltContext>();
161 contexts_[_id] = its_context;
162 DLT_REGISTER_CONTEXT_LL_TS(*(its_context.get()), _id.c_str(), _name.c_str(),
163 DLT_LOG_INFO, DLT_TRACE_STATUS_ON);
164#endif
165#endif
166
167 return its_channel;
168}

170bool connector_impl::remove_channel(const trace_channel_t &_id) {
171 if (_id == VSOMEIP_TC_DEFAULT_CHANNEL_ID) {
172 // the default channel can not be removed
173 return false;
174 }
175
176 std::lock_guard<std::mutex> its_channels_lock(channels_mutex_);
177 bool has_removed = (channels_.erase(_id) == 1);
178 if (has_removed) {
179 // unregister context
180#ifdef USE_DLT
181#ifndef ANDROID
182 std::lock_guard<std::mutex> its_contexts_lock(contexts_mutex_);
183 auto its_context = contexts_.find(_id);
184 if (its_context != contexts_.end()) {
185 DLT_UNREGISTER_CONTEXT(*(its_context->second.get()));
186 }
187#endif
188#endif
189 }
190
191 return true;
192}

206void connector_impl::trace(const byte_t *_header, uint16_t _header_size,
207 const byte_t *_data, uint32_t _data_size) {
208
209#if USE_DLT
210 if (!is_enabled_)
211 return;
212
213 if (_data_size == 0)
214 return; // no data
215
216 // Clip
217 const uint16_t its_data_size
218 = uint16_t(_data_size > USHRT_MAX ? USHRT_MAX : _data_size);
219
220 if (is_sd_message(_data, its_data_size) && !is_sd_enabled_)
221 return; // tracing of service discovery messages is disabled!
222
223 service_t its_service = VSOMEIP_BYTES_TO_WORD(
224 _data[VSOMEIP_SERVICE_POS_MIN],
225 _data[VSOMEIP_SERVICE_POS_MAX]);
226
227 // Instance is not part of the SOME/IP header, read it from the trace
228 // header
229 instance_t its_instance = VSOMEIP_BYTES_TO_WORD(
230 _header[VSOMEIP_TC_INSTANCE_POS_MIN],
231 _header[VSOMEIP_TC_INSTANCE_POS_MAX]);
232
233 method_t its_method = VSOMEIP_BYTES_TO_WORD(
234 _data[VSOMEIP_METHOD_POS_MIN],
235 _data[VSOMEIP_METHOD_POS_MAX]);
236
237 // Forward to channel if the filter set of the channel allows
238 std::lock_guard<std::mutex> its_channels_lock(channels_mutex_);
239 #ifndef ANDROID
240 std::lock_guard<std::mutex> its_contexts_lock(contexts_mutex_);
241 #endif
242 for (auto its_channel : channels_) {
243 auto ftype = its_channel.second->matches(its_service, its_instance, its_method);
244 if (ftype.first) {
245 #ifndef ANDROID
246 auto its_context = contexts_.find(its_channel.second->get_id());
247 if (its_context != contexts_.end()) {
248 try {
249 if (ftype.second) {
250 //Positive Filter
251 DLT_TRACE_NETWORK_SEGMENTED(*(its_context->second.get()),
252 DLT_NW_TRACE_IPC,
253 _header_size, static_cast<void *>(const_cast<byte_t *>(_header)),
254 its_data_size, static_cast<void *>(const_cast<byte_t *>(_data)));
255 } else {
256 //Header-Only Filter
257 DLT_TRACE_NETWORK_TRUNCATED(*(its_context->second.get()),
258 DLT_NW_TRACE_IPC,
259 _header_size, static_cast<void *>(const_cast<byte_t *>(_header)),
260 VSOMEIP_FULL_HEADER_SIZE,
261 static_cast<void *>(const_cast<byte_t *>(_data)));
262 }
263 } catch (const std::exception& e) {
264 VSOMEIP_INFO << "connector_impl::trace: "
265 << "Exception caught when trying to log a trace with DLT. "
266 << e.what();
267 }
268 } else {
269 // This should never happen!
270 VSOMEIP_ERROR << "tracing: found channel without DLT context!";
271 }
272 #else
273 std::stringstream ss;
274 ss << "TC:";
275 for(int i = 0; i < _header_size; i++) {
276 ss << ' ' << std::setfill('0') << std::setw(2) << std::hex << int(_header[i]);
277 }
278 if (ftype.second)
279 _data_size = VSOMEIP_FULL_HEADER_SIZE;
280 for(int i = 0; i < its_data_size; i++) {
281 ss << ' ' << std::setfill('0') << std::setw(2) << std::hex << int(_data[i]);
282 }
283 std::string app = runtime::get_property("LogApplication");
284
285 ALOGI(app.c_str(), ss.str().c_str());
286 #endif
287 }
288 }
289#else
290 (void)_header;
291 (void)_header_size;
292 (void)_data;
293 (void)_data_size;
294#endif
295}
