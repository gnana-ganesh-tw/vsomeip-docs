33bool
34security::load() {
35 using symdef_t = std::tuple<const char*, void*, void**>;
36 std::array<symdef_t, 6> symbol_table{
37 VSOMEIP_SEC_POLICY_SYMDEF(initialize),
38 VSOMEIP_SEC_POLICY_SYMDEF(authenticate_router),
39 VSOMEIP_SEC_POLICY_SYMDEF(is_client_allowed_to_offer),
40 VSOMEIP_SEC_POLICY_SYMDEF(is_client_allowed_to_request),
41 VSOMEIP_SEC_POLICY_SYMDEF(is_client_allowed_to_access_member),
42 VSOMEIP_SEC_SYMDEF(sync_client)
43 };
44
45 if (auto manager = plugin_manager::get()) {
46 if (auto lib = manager->load_library(VSOMEIP_SEC_LIBRARY)) {
47 // First we load the symbols into the 2nd tuple element
48 for (auto& symdef : symbol_table) {
49 auto name = std::get<0>(symdef);
50 auto& symbol = std::get<1>(symdef);
51 if (!(symbol = manager->load_symbol(lib, name))) {
52 VSOMEIP_ERROR << __func__
53 << ": security library misses "
54 << std::quoted(name)
55 << " function.";
56 manager->unload_library(lib);
57 return false;
58 }
59 }
60
61 // Now that we have all symbols loaded,
62 // assign the 2nd tuple element to the 3rd
63 for (auto& symdef : symbol_table) {
64 auto symbol = std::get<1>(symdef);
65 auto& stub = std::get<2>(symdef);
66 *stub = symbol;
67 }
68
69 // Symbol loading complete, success!
70 return true;
71 } else {
72#ifdef _WIN32
73 VSOMEIP_ERROR << "vSomeIP Security: Loading " << VSOMEIP_SEC_LIBRARY << " failed.";
74#else
75 VSOMEIP_ERROR << "vSomeIP Security: " << dlerror();
76#endif
77 }
78 }
79
80 return false;
81}
