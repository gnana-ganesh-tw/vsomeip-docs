32void
33routing_info_entry::serialize(std::vector<byte_t> &_buffer,
34 size_t &_index, error_e &_error) const {
35
36 _buffer[_index] = static_cast<byte_t>(type_);
37 _index += sizeof(type_);
38
39 // Size is overall size - size field - command type
40 size_t its_size = get_size() - sizeof(uint32_t) - 1;
41 if (its_size > std::numeric_limits<uint32_t>::max()) {
42
43 _error = error_e::ERROR_MALFORMED;
44 return;
45 }
46
47 uint32_t its_size32(static_cast<uint32_t>(its_size));
48 std::memcpy(&_buffer[_index], &its_size32, sizeof(its_size32));
49 _index += sizeof(its_size32);
50
51 uint32_t its_client_size(sizeof(client_));
52 if (!address_.is_unspecified()) {
53 if (address_.is_v4()) {
54 its_client_size += uint32_t(sizeof(boost::asio::ip::address_v4::bytes_type)
55 + sizeof(port_));
56 } else {
57 its_client_size += uint32_t(sizeof(boost::asio::ip::address_v6::bytes_type)
58 + sizeof(port_));
59 }
60 }
61
62 if (type_ > routing_info_entry_type_e::RIE_DELETE_CLIENT) {
63
64 std::memcpy(&_buffer[_index], &its_client_size, sizeof(its_client_size));
65 _index += sizeof(its_client_size);
66 }
67
68 std::memcpy(&_buffer[_index], &client_, sizeof(client_));
69 _index += sizeof(client_);
70
71 if (!address_.is_unspecified()) {
72
73 if (address_.is_v4()) {
74 std::memcpy(&_buffer[_index], address_.to_v4().to_bytes().data(),
75 sizeof(boost::asio::ip::address_v4::bytes_type));
76 _index += sizeof(boost::asio::ip::address_v4::bytes_type);
77 } else {
78 std::memcpy(&_buffer[_index], address_.to_v6().to_bytes().data(),
79 sizeof(boost::asio::ip::address_v6::bytes_type));
80 _index += sizeof(boost::asio::ip::address_v6::bytes_type);
81 }
82 std::memcpy(&_buffer[_index], &port_, sizeof(port_));
83 _index += sizeof(port_);
84 }
85
86 if (type_ > routing_info_entry_type_e::RIE_DELETE_CLIENT) {
87
88 its_size = (services_.size() *
89 (sizeof(service_t) + sizeof(instance_t) +
90 sizeof(major_version_t) + sizeof(minor_version_t)));
91
92 if (its_size > std::numeric_limits<uint32_t>::max()) {
93
94 _error = error_e::ERROR_MALFORMED;
95 return;
96 }
97
98 its_size32 = static_cast<uint32_t>(its_size);
99 std::memcpy(&_buffer[_index], &its_size32, sizeof(its_size32));
100 _index += sizeof(its_size32);
101
102 for (const auto &s : services_) {
103
104 std::memcpy(&_buffer[_index], &s.service_, sizeof(s.service_));
105 _index += sizeof(s.service_);
106 std::memcpy(&_buffer[_index], &s.instance_, sizeof(s.instance_));
107 _index += sizeof(s.instance_);
108 std::memcpy(&_buffer[_index], &s.major_, sizeof(s.major_));
109 _index += sizeof(s.major_);
110 std::memcpy(&_buffer[_index], &s.minor_, sizeof(s.minor_));
111 _index += sizeof(s.minor_);
112 }
113 }

116void
117routing_info_entry::deserialize(const std::vector<byte_t> &_buffer,
118 size_t &_index, error_e &_error) {
119
120 uint32_t its_size;
121 uint32_t its_client_size;
122
123 if (_buffer.size() < _index + sizeof(type_) + sizeof(its_size)
124 + sizeof(client_)) {
125
126 _error = error_e::ERROR_NOT_ENOUGH_BYTES;
127 return;
128 }
129
130 type_ = static_cast<routing_info_entry_type_e>(_buffer[_index++]);
131 if (type_ == routing_info_entry_type_e::RIE_UNKNOWN) {
132
133 _error = error_e::ERROR_MALFORMED;
134 return;
135 }
136
137 std::memcpy(&its_size, &_buffer[_index], sizeof(its_size));
138 _index += sizeof(its_size);
139
140 if (type_ > routing_info_entry_type_e::RIE_DELETE_CLIENT) {
141 std::memcpy(&its_client_size, &_buffer[_index], sizeof(its_client_size));
142 _index += sizeof(its_client_size);
143 } else {
144 its_client_size = its_size;
145 }
146
147 std::memcpy(&client_, &_buffer[_index], sizeof(client_));
148 _index += sizeof(client_);
149
150 if (its_client_size > sizeof(client_)) {
151
152 uint32_t its_address_size = its_client_size
153 - uint32_t(sizeof(client_t) + sizeof(port_));
154
155 if (its_address_size == sizeof(boost::asio::ip::address_v4::bytes_type)) {
156
157 boost::asio::ip::address_v4::bytes_type its_array;
158 std::memcpy(&its_array, &_buffer[_index], its_array.size());
159 address_ = boost::asio::ip::address_v4(its_array);
160 _index += its_array.size();
161
162 } else if (its_address_size == sizeof(boost::asio::ip::address_v6::bytes_type)) {
163
164 boost::asio::ip::address_v6::bytes_type its_array;
165 std::memcpy(&its_array, &_buffer[_index], its_array.size());
166 address_ = boost::asio::ip::address_v6(its_array);
167 _index += its_array.size();
168
169 } else {
170
171 _error = error_e::ERROR_MALFORMED;
172 return;
173 }
174
175 std::memcpy(&port_, &_buffer[_index], sizeof(port_));
176 _index += sizeof(port_);
177 }
178
179 if (type_ > routing_info_entry_type_e::RIE_DELETE_CLIENT) {
180
181 if (_buffer.size() < _index + sizeof(its_size)) {
182
183 _error = error_e::ERROR_NOT_ENOUGH_BYTES;
184 return;
185 }
186
187 std::memcpy(&its_size, &_buffer[_index], sizeof(its_size));
188 _index += sizeof(its_size);
189
190 if (_buffer.size() < _index + its_size) {
191
192 _error = error_e::ERROR_NOT_ENOUGH_BYTES;
193 return;
194 }
195
196 size_t its_n = (its_size /
197 (sizeof(service_t) + sizeof(instance_t) +
198 sizeof(major_version_t) + sizeof(minor_version_t)));
199
200 for (size_t i = 0; i < its_n; i++) {
201
202 service its_service;
203 std::memcpy(&its_service.service_, &_buffer[_index], sizeof(its_service.service_));
204 _index += sizeof(its_service.service_);
205 std::memcpy(&its_service.instance_, &_buffer[_index], sizeof(its_service.instance_));
206 _index += sizeof(its_service.instance_);
207 its_service.major_ = static_cast<major_version_t>(_buffer[_index]);
208 _index += sizeof(its_service.major_);
209 std::memcpy(&its_service.minor_, &_buffer[_index], sizeof(its_service.minor_));
210 _index += sizeof(its_service.minor_);
211
212 services_.emplace_back(its_service);
213 }
214 }

229size_t
230routing_info_entry::get_size() const {
231
232 size_t its_size(ROUTING_INFO_ENTRY_HEADER_SIZE);
233
234 if (!address_.is_unspecified()) {
235 if (address_.is_v4()) {
236 its_size += (sizeof(boost::asio::ip::address_v4::bytes_type)
237 + sizeof(port_));
238 } else {
239 its_size += (sizeof(boost::asio::ip::address_v6::bytes_type)
240 + sizeof(port_));
241 }
242 }
243
244 if (type_ > routing_info_entry_type_e::RIE_DELETE_CLIENT) {
245 its_size += sizeof(uint32_t); // size of the client info
246 its_size += sizeof(uint32_t); // size of the services array
247 its_size += (services_.size() *
248 (sizeof(service_t) + sizeof(instance_t) +
249 sizeof(major_version_t) + sizeof(minor_version_t)));
250 }
251
252 return its_size;
