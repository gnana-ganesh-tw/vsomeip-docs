152
153 std::lock_guard<std::mutex> its_lock(mutex_);
154 if (is_provided_) {
155 if (prepare_update_payload_unlocked(_payload, _force)) {
156 if (is_updating_on_change_) {

169 VSOMEIP_INFO << __func__ << ":" << __LINE__
170 << " Cannot set payload for event ["
171 << std::hex << std::setw(4) << std::setfill('0')
172 << current_->get_service() << "."
173 << current_->get_instance() << "."
174 << current_->get_method()
175 << "]. It isn't provided";
176 }

447bool
448event::prepare_update_payload_unlocked(
449 const std::shared_ptr<payload> &_payload, bool _force) {
450
451 // Copy payload to avoid manipulation from the outside
452 std::shared_ptr<payload> its_payload
453 = runtime::get()->create_payload(
454 _payload->get_data(), _payload->get_length());
455
456 bool is_change = has_changed(current_->get_payload(), its_payload);
457
458 if (!_force
459 && type_ == event_type_e::ET_FIELD
460 && cycle_ == std::chrono::milliseconds::zero()
461 && !is_change
462 && !is_shadow_) {
463 return false;
464 }
465
466 update_->set_payload(its_payload);
467
468 if (!is_set_) {
469 start_cycle();
470 is_set_ = true;
471 }
472
473 return true;
474}

519bool
520event::add_subscriber(eventgroup_t _eventgroup,
521 const std::shared_ptr<debounce_filter_impl_t> &_filter,
522 client_t _client, bool _force) {
523
524 std::lock_guard<std::mutex> its_lock(eventgroups_mutex_);
525 bool ret = false;
526 if (_force // remote events managed by rm_impl
527 || is_provided_ // events provided by rm_proxies
528 || is_shadow_ // local events managed by rm_impl
529 || is_cache_placeholder_) {
530
531 if (_filter) {
532 VSOMEIP_WARNING << "Using client ["
533 << std::hex << std::setw(4) << std::setfill('0')
534 << _client
535 << "] specific filter configuration for SOME/IP event "
536 << get_service() << "." << get_instance() << "." << get_event() << ".";
537 std::stringstream its_filter_parameters;
538 its_filter_parameters << "(on_change="
539 << std::boolalpha << _filter->on_change_
540 << ", interval=" << std::dec << _filter->interval_
541 << ", on_change_resets_interval="
542 << std::boolalpha << _filter->on_change_resets_interval_
543 << ", ignore=[ ";
544 for (auto i : _filter->ignore_)
545 its_filter_parameters << "(" << std::dec << i.first << ", "
546 << std::hex << std::setw(2) << std::setfill('0')
547 << (int)i.second << ") ";
548 its_filter_parameters << "])";
549 VSOMEIP_INFO << "Filter parameters: "
550 << its_filter_parameters.str();
551
552 filters_[_client] = [_filter](
553 const std::shared_ptr<payload> &_old,
554 const std::shared_ptr<payload> &_new) {
555
556 bool is_changed(false), is_elapsed(false);
557
558 // Check whether we should forward because of changed data
559 if (_filter->on_change_) {
560 length_t its_min_length, its_max_length;
561
562 if (_old->get_length() < _new->get_length()) {
563 its_min_length = _old->get_length();
564 its_max_length = _new->get_length();
565 } else {
566 its_min_length = _new->get_length();
567 its_max_length = _old->get_length();
568 }
569
570 // Check whether all additional bytes (if any) are excluded
571 for (length_t i = its_min_length; i < its_max_length; i++) {
572 auto j = _filter->ignore_.find(i);
573 // A change is detected when an additional byte is not
574 // excluded at all or if its exclusion does not cover all
575 // bits
576 if (j == _filter->ignore_.end() || j->second != 0xFF) {
577 is_changed = true;
578 break;
579 }
580 }
581
582 if (!is_changed) {
583 const byte_t *its_old = _old->get_data();
584 const byte_t *its_new = _new->get_data();
585 for (length_t i = 0; i < its_min_length; i++) {
586 auto j = _filter->ignore_.find(i);
587 if (j == _filter->ignore_.end()) {
588 if (its_old[i] != its_new[i]) {
589 is_changed = true;
590 break;
591 }
592 } else if (j->second != 0xFF) {
593 if ((its_old[i] & ~(j->second)) != (its_new[i] & ~(j->second))) {
594 is_changed = true;
595 break;
596 }
597 }
598 }
599 }
600 }
601
602 if (_filter->interval_ > -1) {
603 // Check whether we should forward because of the elapsed time since
604 // we did last time
605 std::chrono::steady_clock::time_point its_current
606 = std::chrono::steady_clock::now();
607
608 int64_t elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
609 its_current - _filter->last_forwarded_).count();
610 is_elapsed = (_filter->last_forwarded_ == std::chrono::steady_clock::time_point::max()
611 || elapsed >= _filter->interval_);
612 if (is_elapsed || (is_changed && _filter->on_change_resets_interval_))
613 _filter->last_forwarded_ = its_current; }
614
615 return (is_changed || is_elapsed);
616 };
617
618 // Create a new callback for this client if filter interval is used
619 routing_->register_debounce(_filter, _client, shared_from_this());
620 } else {
621 filters_.erase(_client);
622 }
623
624 ret = eventgroups_[_eventgroup].insert(_client).second;
625
626 } else {
627 VSOMEIP_WARNING << __func__ << ": Didnt' insert client "
628 << std::hex << std::setw(4) << std::setfill('0') << _client
629 << " to eventgroup 0x"
630 << std::hex << std::setw(4) << std::setfill('0')
631 << get_service() << "." << get_instance() << "."
632 << _eventgroup;
633 }
634 return ret;
635}

674std::set<client_t>
675event::get_filtered_subscribers(bool _force) {
676
677 std::set<client_t> its_subscribers(get_subscribers());
678 std::set<client_t> its_filtered_subscribers;
679
680 std::shared_ptr<payload> its_payload, its_payload_update;
681 {
682 its_payload = current_->get_payload();
683 its_payload_update = update_->get_payload();
684 }
685
686 if (filters_.empty()) {
687
688 bool must_forward = (type_ != event_type_e::ET_FIELD
689 || _force
690 || epsilon_change_func_(its_payload, its_payload_update));
691
692 if (must_forward)
693 return its_subscribers;
694
695 } else {
696 byte_t is_allowed(0xff);
697
698 std::lock_guard<std::mutex> its_lock(filters_mutex_);
699 for (const auto s : its_subscribers) {
700
701 auto its_specific = filters_.find(s);
702 if (its_specific != filters_.end()) {
703 if (its_specific->second(its_payload, its_payload_update))
704 its_filtered_subscribers.insert(s);
705 } else {
706 if (is_allowed == 0xff) {
707 is_allowed = (type_ != event_type_e::ET_FIELD
708 || _force
709 || epsilon_change_func_(its_payload, its_payload_update)
710 ? 0x01 : 0x00);
711 }
712
713 if (is_allowed == 0x01)
714 its_filtered_subscribers.insert(s);
715 }
716 }
717 }
718
719 return its_filtered_subscribers;
720}
