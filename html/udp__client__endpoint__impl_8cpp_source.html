55void udp_client_endpoint_impl::connect() {
56 std::unique_lock<std::mutex> its_lock(socket_mutex_);
57 boost::system::error_code its_error;
58 socket_->open(remote_.protocol(), its_error);
59 if (!its_error || its_error == boost::asio::error::already_open) {
60 // Enable SO_REUSEADDR to avoid bind problems with services going offline
61 // and coming online again and the user has specified only a small number
62 // of ports in the clients section for one service instance
63 socket_->set_option(boost::asio::socket_base::reuse_address(true), its_error);
64 if (its_error) {
65 VSOMEIP_WARNING << "udp_client_endpoint_impl::connect: couldn't enable "
66 << "SO_REUSEADDR: " << its_error.message() << " remote:"
67 << get_address_port_remote();
68 }
69 socket_->set_option(boost::asio::socket_base::receive_buffer_size(
70 udp_receive_buffer_size_), its_error);
71 if (its_error) {
72 VSOMEIP_WARNING << "udp_client_endpoint_impl::connect: couldn't set "
73 << "SO_RCVBUF: " << its_error.message()
74 << " to: " << std::dec << udp_receive_buffer_size_
75 << " local port:" << std::dec << local_.port()
76 << " remote:" << get_address_port_remote();
77 }
78
79 boost::asio::socket_base::receive_buffer_size its_option;
80 socket_->get_option(its_option, its_error);
81 #ifdef __linux__
82 // If regular setting of the buffer size did not work, try to force
83 // (requires CAP_NET_ADMIN to be successful)
84 if (its_option.value() < 0
85 || its_option.value() < udp_receive_buffer_size_) {
86 its_error.assign(setsockopt(socket_->native_handle(),
87 SOL_SOCKET, SO_RCVBUFFORCE,
88 &udp_receive_buffer_size_, sizeof(udp_receive_buffer_size_)),
89 boost::system::generic_category());
90 if (!its_error) {
91 VSOMEIP_INFO << "udp_client_endpoint_impl::connect: "
92 << "SO_RCVBUFFORCE successful!";
93 }
94 socket_->get_option(its_option, its_error);
95 }
96 #endif
97 if (its_error) {
98 VSOMEIP_WARNING << "udp_client_endpoint_impl::connect: couldn't get "
99 << "SO_RCVBUF: " << its_error.message()
100 << " local port:" << std::dec << local_.port()
101 << " remote:" << get_address_port_remote();
102 } else {
103 VSOMEIP_INFO << "udp_client_endpoint_impl::connect: SO_RCVBUF is: "
104 << std::dec << its_option.value()
105 << " (" << udp_receive_buffer_size_ << ")"
106 << " local port:" << std::dec << local_.port()
107 << " remote:" << get_address_port_remote();
108 }
109
110 if (local_.port() == ILLEGAL_PORT) {
111 // Let the OS assign the port
112 local_.port(0);
113 }
114
115#if defined(__linux__) || defined(ANDROID) || defined(__QNX__)
116 // If specified, bind to device
117 std::string its_device(configuration_->get_device());
118 if (its_device != "") {
119 if (setsockopt(socket_->native_handle(),
120 SOL_SOCKET, SO_BINDTODEVICE, its_device.c_str(), socklen_t(its_device.size())) == -1) {
121 VSOMEIP_WARNING << "UDP Client: Could not bind to device \"" << its_device << "\"";
122 }
123 }
124#endif
125
126 // In case a client endpoint port was configured,
127 // bind to it before connecting
128 if (local_.port() != ILLEGAL_PORT) {
129 boost::system::error_code its_bind_error;
130 socket_->bind(local_, its_bind_error);
131 if(its_bind_error) {
132 VSOMEIP_WARNING << "udp_client_endpoint::connect: "
133 "Error binding socket: " << its_bind_error.message()
134 << " local: " << local_.address().to_string()
135 << ":" << std::dec << local_.port()
136 << " remote:" << get_address_port_remote();
137
138 its_lock.unlock();
139
140 std::shared_ptr<endpoint_host> its_host = endpoint_host_.lock();
141 if (its_host) {
142 // set new client port depending on service / instance / remote port
143 if (!its_host->on_bind_error(shared_from_this(), remote_address_, remote_port_)) {
144 VSOMEIP_WARNING << "udp_client_endpoint::connect: "
145 "Failed to set new local port for uce: "
146 << " local: " << local_.address().to_string()
147 << ":" << std::dec << local_.port()
148 << " remote:" << get_address_port_remote();
149 } else {
150 VSOMEIP_INFO << "udp_client_endpoint::connect: "
151 "Using new new local port for uce: "
152 << " local: " << local_.address().to_string()
153 << ":" << std::dec << local_.port()
154 << " remote:" << get_address_port_remote();
155 }
156 }
157
158
159 try {
160 // don't connect on bind error to avoid using a random port
161 strand_.post(std::bind(&client_endpoint_impl::connect_cbk,
162 shared_from_this(), its_bind_error));
163 } catch (const std::exception &e) {
164 VSOMEIP_ERROR << "udp_client_endpoint_impl::connect: "
165 << e.what() << " remote:" << get_address_port_remote();
166 }
167 return;
168 }
169 }
170
171 state_ = cei_state_e::CONNECTING;
172 socket_->async_connect(
173 remote_,
174 strand_.wrap(
175 std::bind(
176 &udp_client_endpoint_base_impl::connect_cbk,
177 shared_from_this(),
178 std::placeholders::_1
179 )
180 )
181 );
182 } else {
183 VSOMEIP_WARNING << "udp_client_endpoint::connect: Error opening socket: "
184 << its_error.message() << " remote:" << get_address_port_remote();
185 strand_.post(std::bind(&udp_client_endpoint_base_impl::connect_cbk,
186 shared_from_this(), its_error));
187 }
188}

194void udp_client_endpoint_impl::restart(bool _force) {
195 if (!_force && state_ == cei_state_e::CONNECTING) {
196 return;
197 }
198 state_ = cei_state_e::CONNECTING;
199 {
200 std::lock_guard<std::recursive_mutex> its_lock(mutex_);
201 queue_.clear();
202 }
203 std::string local;
204 {
205 std::lock_guard<std::mutex> its_lock(socket_mutex_);
206 local = get_address_port_local();
207 }
208 shutdown_and_close_socket(false);
209 was_not_connected_ = true;
210 reconnect_counter_ = 0;
211 VSOMEIP_WARNING << "uce::restart: local: " << local
212 << " remote: " << get_address_port_remote();
213 start_connect_timer();
214}

216void udp_client_endpoint_impl::send_queued(std::pair<message_buffer_ptr_t, uint32_t> &_entry) {
217
218#if 0
219 std::stringstream msg;
220 msg << "ucei<" << remote_.address() << ":"
221 << std::dec << remote_.port() << ">::sq: ";
222 for (std::size_t i = 0; i < _buffer->size(); i++)
223 msg << std::hex << std::setw(2) << std::setfill('0')
224 << (int)(*_entry.first)[i] << " ";
225 VSOMEIP_INFO << msg.str();
226#endif
227 {
228 std::lock_guard<std::mutex> its_last_sent_lock(last_sent_mutex_);
229 std::lock_guard<std::mutex> its_socket_lock(socket_mutex_);
230
231 // Check whether we need to wait (SOME/IP-TP separation time)
232 if (_entry.second > 0) {
233 if (last_sent_ != std::chrono::steady_clock::time_point()) {
234 const auto its_elapsed
235 = std::chrono::duration_cast<std::chrono::microseconds>(
236 std::chrono::steady_clock::now() - last_sent_).count();
237 if (_entry.second > its_elapsed)
238 std::this_thread::sleep_for(
239 std::chrono::microseconds(_entry.second - its_elapsed));
240 }
241 last_sent_ = std::chrono::steady_clock::now();
242 } else {
243 last_sent_ = std::chrono::steady_clock::time_point();
244 }
245 // Send
246 socket_->async_send(
247 boost::asio::buffer(*_entry.first),
248 std::bind(
249 &udp_client_endpoint_base_impl::send_cbk,
250 shared_from_this(),
251 std::placeholders::_1,
252 std::placeholders::_2,
253 _entry.first
254 )
255 );
256 }

269 std::lock_guard<std::mutex> its_lock(socket_mutex_);
270 if (!socket_->is_open()) {
271 return;
272 }
273 message_buffer_ptr_t its_buffer = std::make_shared<message_buffer_t>(VSOMEIP_MAX_UDP_MESSAGE_SIZE);
274 socket_->async_receive_from(
275 boost::asio::buffer(*its_buffer),
276 const_cast<endpoint_type&>(remote_),
277 strand_.wrap(
278 std::bind(
279 &udp_client_endpoint_impl::receive_cbk,
280 std::dynamic_pointer_cast<

285 its_buffer
286 )

354void udp_client_endpoint_impl::receive_cbk(
355 boost::system::error_code const &_error, std::size_t _bytes,
356 const message_buffer_ptr_t& _recv_buffer) {
357 if (_error == boost::asio::error::operation_aborted) {
358 // endpoint was stopped
359 return;
360 }
361 std::shared_ptr<routing_host> its_host = routing_host_.lock();
362 if (!_error && 0 < _bytes && its_host) {
363#if 0
364 std::stringstream msg;
365 msg << "ucei::rcb(" << _error.message() << "): ";
366 for (std::size_t i = 0; i < _bytes; ++i)
367 msg << std::hex << std::setw(2) << std::setfill('0')
368 << (int) (*_recv_buffer)[i] << " ";
369 VSOMEIP_INFO << msg.str();
370#endif
371 std::size_t remaining_bytes = _bytes;
372 std::size_t i = 0;
373
374 do {
375 uint64_t read_message_size
376 = utility::get_message_size(&(*_recv_buffer)[i],
377 remaining_bytes);
378 if (read_message_size > MESSAGE_SIZE_UNLIMITED) {
379 VSOMEIP_ERROR << "Message size exceeds allowed maximum!";
380 return;
381 }
382 uint32_t current_message_size = static_cast<uint32_t>(read_message_size);
383 if (current_message_size > VSOMEIP_SOMEIP_HEADER_SIZE &&
384 current_message_size <= remaining_bytes) {
385 if (remaining_bytes - current_message_size > remaining_bytes) {
386 VSOMEIP_ERROR << "buffer underflow in udp client endpoint ~> abort!";
387 return;
388 } else if (current_message_size > VSOMEIP_RETURN_CODE_POS &&
389 ((*_recv_buffer)[i + VSOMEIP_PROTOCOL_VERSION_POS] != VSOMEIP_PROTOCOL_VERSION ||
390 !utility::is_valid_message_type(tp::tp::tp_flag_unset((*_recv_buffer)[i + VSOMEIP_MESSAGE_TYPE_POS])) ||
391 !utility::is_valid_return_code(static_cast<return_code_e>((*_recv_buffer)[i + VSOMEIP_RETURN_CODE_POS]))
392 )) {
393 if ((*_recv_buffer)[i + VSOMEIP_PROTOCOL_VERSION_POS] != VSOMEIP_PROTOCOL_VERSION) {
394 VSOMEIP_ERROR << "uce: Wrong protocol version: 0x"
395 << std::hex << std::setw(2) << std::setfill('0')
396 << std::uint32_t((*_recv_buffer)[i + VSOMEIP_PROTOCOL_VERSION_POS])
397 << " local: " << get_address_port_local()
398 << " remote: " << get_address_port_remote();
399 // ensure to send back a message w/ wrong protocol version
400 its_host->on_message(&(*_recv_buffer)[i],
401 VSOMEIP_SOMEIP_HEADER_SIZE + 8, this,
402 false,
403 VSOMEIP_ROUTING_CLIENT,
404 nullptr,
405 remote_address_,
406 remote_port_);
407 } else if (!utility::is_valid_message_type(tp::tp::tp_flag_unset(
408 (*_recv_buffer)[i + VSOMEIP_MESSAGE_TYPE_POS]))) {
409 VSOMEIP_ERROR << "uce: Invalid message type: 0x"
410 << std::hex << std::setw(2) << std::setfill('0')
411 << std::uint32_t((*_recv_buffer)[i + VSOMEIP_MESSAGE_TYPE_POS])
412 << " local: " << get_address_port_local()
413 << " remote: " << get_address_port_remote();
414 } else if (!utility::is_valid_return_code(static_cast<return_code_e>(
415 (*_recv_buffer)[i + VSOMEIP_RETURN_CODE_POS]))) {
416 VSOMEIP_ERROR << "uce: Invalid return code: 0x"
417 << std::hex << std::setw(2) << std::setfill('0')
418 << std::uint32_t((*_recv_buffer)[i + VSOMEIP_RETURN_CODE_POS])
419 << " local: " << get_address_port_local()
420 << " remote: " << get_address_port_remote();
421 }
422 receive();
423 return;
424 } else if (tp::tp::tp_flag_is_set((*_recv_buffer)[i + VSOMEIP_MESSAGE_TYPE_POS])) {
425 const auto res = tp_reassembler_->process_tp_message(
426 &(*_recv_buffer)[i], current_message_size,
427 remote_address_, remote_port_);
428 if (res.first) {
429 its_host->on_message(&res.second[0],
430 static_cast<std::uint32_t>(res.second.size()),
431 this,
432 false,
433 VSOMEIP_ROUTING_CLIENT,
434 nullptr,
435 remote_address_,
436 remote_port_);
437 }
438 } else {
439 its_host->on_message(&(*_recv_buffer)[i], current_message_size,
440 this,
441 false,
442 VSOMEIP_ROUTING_CLIENT,
443 nullptr,
444 remote_address_,
445 remote_port_);
446 }
447 remaining_bytes -= current_message_size;
448 } else {
449 VSOMEIP_ERROR << "Received a unreliable vSomeIP message with bad "
450 "length field. Message size: " << current_message_size
451 << " Bytes. From: " << remote_.address() << ":"
452 << remote_.port() << ". Dropping message.";
453 remaining_bytes = 0;
454 }
455 i += current_message_size;
456 } while (remaining_bytes > 0);
457 }
458 if (!_error) {
459 receive();
460 } else {
461 if (_error == boost::asio::error::connection_refused) {
462 VSOMEIP_WARNING << "uce::receive_cbk: local: " << get_address_port_local()
463 << " remote: " << get_address_port_remote()
464 << " error: " << _error.message();
465 std::shared_ptr<endpoint_host> its_ep_host = endpoint_host_.lock();
466 its_ep_host->on_disconnect(shared_from_this());
467 restart(false);
468 } else {
469 receive();
470 }
471 }
472}

527void udp_client_endpoint_impl::send_cbk(boost::system::error_code const &_error,
528 std::size_t _bytes, const message_buffer_ptr_t &_sent_msg) {
529 (void)_bytes;
530 if (!_error) {
531 std::lock_guard<std::recursive_mutex> its_lock(mutex_);
532 if (queue_.size() > 0) {
533 queue_size_ -= queue_.front().first->size();
534 queue_.pop_front();
535
536 update_last_departure();
537
538 if (queue_.empty())
539 is_sending_ = false;
540 else {
541 auto its_entry = get_front();
542 if (its_entry.first) {
543 send_queued(its_entry);
544 }
545 }
546 }
547 return;
548 } else if (_error == boost::asio::error::broken_pipe) {
549 state_ = cei_state_e::CLOSED;
550 bool stopping(false);
551 {
552 std::lock_guard<std::recursive_mutex> its_lock(mutex_);
553 stopping = sending_blocked_;
554 if (stopping) {
555 queue_.clear();
556 queue_size_ = 0;
557 } else {
558 service_t its_service(0);
559 method_t its_method(0);
560 client_t its_client(0);
561 session_t its_session(0);
562 if (_sent_msg && _sent_msg->size() > VSOMEIP_SESSION_POS_MAX) {
563 its_service = VSOMEIP_BYTES_TO_WORD(
564 (*_sent_msg)[VSOMEIP_SERVICE_POS_MIN],
565 (*_sent_msg)[VSOMEIP_SERVICE_POS_MAX]);
566 its_method = VSOMEIP_BYTES_TO_WORD(
567 (*_sent_msg)[VSOMEIP_METHOD_POS_MIN],
568 (*_sent_msg)[VSOMEIP_METHOD_POS_MAX]);
569 its_client = VSOMEIP_BYTES_TO_WORD(
570 (*_sent_msg)[VSOMEIP_CLIENT_POS_MIN],
571 (*_sent_msg)[VSOMEIP_CLIENT_POS_MAX]);
572 its_session = VSOMEIP_BYTES_TO_WORD(
573 (*_sent_msg)[VSOMEIP_SESSION_POS_MIN],
574 (*_sent_msg)[VSOMEIP_SESSION_POS_MAX]);
575 }
576 VSOMEIP_WARNING << "uce::send_cbk received error: "
577 << _error.message() << " (" << std::dec
578 << _error.value() << ") " << get_remote_information()
579 << " " << std::dec << queue_.size()
580 << " " << std::dec << queue_size_ << " ("
581 << std::hex << std::setfill('0')
582 << std::setw(4) << its_client << "): ["
583 << std::setw(4) << its_service << "."
584 << std::setw(4) << its_method << "."
585 << std::setw(4) << its_session << "]";
586 }
587 }
588 if (!stopping) {
589 print_status();
590 }
591 was_not_connected_ = true;
592 shutdown_and_close_socket(true);
593 strand_.dispatch(std::bind(&client_endpoint_impl::connect,
594 this->shared_from_this()));
595 } else if (_error == boost::asio::error::not_connected
596 || _error == boost::asio::error::bad_descriptor
597 || _error == boost::asio::error::no_permission) {
598 state_ = cei_state_e::CLOSED;
599 if (_error == boost::asio::error::no_permission) {
600 VSOMEIP_WARNING << "uce::send_cbk received error: " << _error.message()
601 << " (" << std::dec << _error.value() << ") "
602 << get_remote_information();
603 std::lock_guard<std::recursive_mutex> its_lock(mutex_);
604 queue_.clear();
605 queue_size_ = 0;
606 }
607 was_not_connected_ = true;
608 shutdown_and_close_socket(true);
609 strand_.dispatch(std::bind(&client_endpoint_impl::connect,
610 this->shared_from_this()));
611 } else if (_error == boost::asio::error::operation_aborted) {
612 VSOMEIP_WARNING << "uce::send_cbk received error: " << _error.message();
613 // endpoint was stopped
614 sending_blocked_ = true;
615 shutdown_and_close_socket(false);

618 << " (" << std::dec << _error.value() << ") "
619 << get_remote_information();

621 } else {
622 if (state_ == cei_state_e::CONNECTING) {

625 } else {

627 shutdown_and_close_socket(false);
628 std::shared_ptr<endpoint_host> its_host = endpoint_host_.lock();
629 if (its_host) {
630 its_host->on_disconnect(shared_from_this());
631 }
632 restart(true);
633 }

635 method_t its_method(0);
636 client_t its_client(0);
637 session_t its_session(0);
638 if (_sent_msg && _sent_msg->size() > VSOMEIP_SESSION_POS_MAX) {
639 its_service = VSOMEIP_BYTES_TO_WORD(
640 (*_sent_msg)[VSOMEIP_SERVICE_POS_MIN],
641 (*_sent_msg)[VSOMEIP_SERVICE_POS_MAX]);
642 its_method = VSOMEIP_BYTES_TO_WORD(
643 (*_sent_msg)[VSOMEIP_METHOD_POS_MIN],
644 (*_sent_msg)[VSOMEIP_METHOD_POS_MAX]);
645 its_client = VSOMEIP_BYTES_TO_WORD(
646 (*_sent_msg)[VSOMEIP_CLIENT_POS_MIN],
647 (*_sent_msg)[VSOMEIP_CLIENT_POS_MAX]);
648 its_session = VSOMEIP_BYTES_TO_WORD(
649 (*_sent_msg)[VSOMEIP_SESSION_POS_MIN],
650 (*_sent_msg)[VSOMEIP_SESSION_POS_MAX]);
651 }
652 VSOMEIP_WARNING << "uce::send_cbk received error: " << _error.message()
653 << " (" << std::dec << _error.value() << ") "
654 << get_remote_information() << " "
655 << " " << std::dec << queue_.size()
656 << " " << std::dec << queue_size_ << " ("
657 << std::hex << std::setfill('0')
658 << std::setw(4) << its_client << "): ["
659 << std::setw(4) << its_service << "."
660 << std::setw(4) << its_method << "."
661 << std::setw(4) << its_session << "]";
662 print_status();
663 }
664
665 std::lock_guard<std::recursive_mutex> its_lock(mutex_);
666 is_sending_ = false;
667}
