70void local_tcp_server_endpoint_impl::start() {
71
72 std::lock_guard<std::mutex> its_lock(acceptor_mutex_);
73 if (!acceptor_.is_open()) {
74 boost::system::error_code ec;
75 acceptor_.open(local_.protocol(), ec);
76 boost::asio::detail::throw_error(ec, "acceptor open");
77 acceptor_.set_option(boost::asio::socket_base::reuse_address(true), ec);
78 boost::asio::detail::throw_error(ec, "acceptor set_option");
79 acceptor_.bind(local_, ec);
80 boost::asio::detail::throw_error(ec, "acceptor bind");
81 acceptor_.listen(boost::asio::socket_base::max_connections, ec);
82 boost::asio::detail::throw_error(ec, "acceptor listen");
83 }
84
85 if (acceptor_.is_open()) {
86 connection::ptr new_connection = connection::create(
87 std::dynamic_pointer_cast<local_tcp_server_endpoint_impl>(
88 shared_from_this()), max_message_size_,
89 buffer_shrink_threshold_,
90 io_);
91
92 {
93 std::unique_lock<std::mutex> its_lock(new_connection->get_socket_lock());
94 acceptor_.async_accept(
95 new_connection->get_socket(),
96 std::bind(
97 &local_tcp_server_endpoint_impl::accept_cbk,
98 std::dynamic_pointer_cast<
99 local_tcp_server_endpoint_impl
100 >(shared_from_this()),
101 new_connection,
102 std::placeholders::_1
103 )
104 );
105 }
106 }

225void local_tcp_server_endpoint_impl::accept_cbk(
226 const connection::ptr& _connection, boost::system::error_code const &_error) {
227 if (_error != boost::asio::error::bad_descriptor
228 && _error != boost::asio::error::operation_aborted
229 && _error != boost::asio::error::no_descriptors) {
230 start();
231 } else if (_error == boost::asio::error::no_descriptors) {
232 VSOMEIP_ERROR << "local_tcp_server_endpoint_impl::accept_cbk: "
233 << _error.message() << " (" << std::dec << _error.value()
234 << ") Will try to accept again in 1000ms";
235 std::shared_ptr<boost::asio::steady_timer> its_timer =
236 std::make_shared<boost::asio::steady_timer>(io_,
237 std::chrono::milliseconds(1000));
238 auto its_ep = std::dynamic_pointer_cast<local_tcp_server_endpoint_impl>(
239 shared_from_this());
240 its_timer->async_wait([its_timer, its_ep]
241 (const boost::system::error_code& _error) {
242 if (!_error) {
243 its_ep->start();
244 }
245 });
246 }
247
248 if (!_error) {
249 _connection->start();
250 }

308void local_tcp_server_endpoint_impl::connection::start() {
309 std::lock_guard<std::mutex> its_lock(socket_mutex_);
310 if (socket_.is_open()) {
311 const std::size_t its_capacity(recv_buffer_.capacity());
312 if (recv_buffer_size_ > its_capacity) {
313 VSOMEIP_ERROR << __func__ << "Received buffer size is greater than the buffer capacity!"
314 << " recv_buffer_size_: " << recv_buffer_size_
315 << " its_capacity: " << its_capacity;
316 return;
317 }
318 size_t left_buffer_size = its_capacity - recv_buffer_size_;
319 try {
320 if (missing_capacity_) {
321 if (missing_capacity_ > MESSAGE_SIZE_UNLIMITED) {
322 VSOMEIP_ERROR << "Missing receive buffer capacity exceeds allowed maximum!";
323 return;
324 }
325 const std::size_t its_required_capacity(recv_buffer_size_ + missing_capacity_);
326 if (its_capacity < its_required_capacity) {
327 // Make the resize to its_required_capacity
328 recv_buffer_.reserve(its_required_capacity);
329 recv_buffer_.resize(its_required_capacity, 0x0);
330 }
331 left_buffer_size = missing_capacity_;
332 missing_capacity_ = 0;
333 } else if (buffer_shrink_threshold_
334 && shrink_count_ > buffer_shrink_threshold_
335 && recv_buffer_size_ == 0) {
336 // In this case, make the resize to recv_buffer_size_initial_
337 recv_buffer_.resize(recv_buffer_size_initial_, 0x0);
338 recv_buffer_.shrink_to_fit();
339 // And set buffer_size to recv_buffer_size_initial_, the same of our resize
340 left_buffer_size = recv_buffer_size_initial_;
341 shrink_count_ = 0;
342 }
343 } catch (const std::exception &e) {
344 handle_recv_buffer_exception(e);
345 // don't start receiving again
346 return;
347 }
348
349 is_stopped_ = false;
350#if VSOMEIP_BOOST_VERSION < 106600
351 socket_.async_receive(
352 boost::asio::buffer(&recv_buffer_[recv_buffer_size_], left_buffer_size),
353 std::bind(
354 &local_tcp_server_endpoint_impl::connection::receive_cbk,
355 shared_from_this(),
356 std::placeholders::_1,
357 std::placeholders::_2
358 )
359 );
360#else
361 socket_.async_receive(
362 boost::asio::buffer(&recv_buffer_[recv_buffer_size_], left_buffer_size),
363 std::bind(
364 &local_tcp_server_endpoint_impl::connection::receive_cbk,
365 shared_from_this(),
366 std::placeholders::_1,
367 std::placeholders::_2
368 )
369 );
370#endif
371 }

471void local_tcp_server_endpoint_impl::connection::receive_cbk(
472 boost::system::error_code const &_error, std::size_t _bytes)
473{
474 std::shared_ptr<local_tcp_server_endpoint_impl> its_server(server_.lock());
475 if (!its_server) {
476 VSOMEIP_TRACE << "local_server_endpoint_impl::connection::receive_cbk "
477 " couldn't lock server_";
478 return;
479 }
480 std::shared_ptr<routing_host> its_host = its_server->routing_host_.lock();
481 if (!its_host)
482 return;
483
484 if (_error == boost::asio::error::operation_aborted) {
485 if (its_server->is_routing_endpoint_ && its_server->configuration_ &&
486 bound_client_ != VSOMEIP_CLIENT_UNSET) {
487 utility::release_client_id(its_server->configuration_->get_network(),
488 bound_client_);
489 set_bound_client(VSOMEIP_CLIENT_UNSET);
490 }
491
492 // connection was stopped
493 return;
494 }
495
496 bool is_error(false);
497 std::size_t its_start = 0;
498 std::size_t its_end = 0;
499 std::size_t its_iteration_gap = 0;
500 std::uint32_t its_command_size = 0;
501
502 if (!_error && 0 < _bytes) {
503#if 0
504 std::stringstream msg;
505 msg << "lse::c<" << this << ">rcb: ";
506 for (std::size_t i = 0; i < _bytes + recv_buffer_size_; i++)
507 msg << std::setw(2) << std::setfill('0') << std::hex
508 << (int) (recv_buffer_[i]) << " ";
509 VSOMEIP_INFO << msg.str();
510#endif
511
512 if (recv_buffer_size_ + _bytes < recv_buffer_size_) {
513 VSOMEIP_ERROR << "receive buffer overflow in local server endpoint ~> abort!";
514 return;
515 }
516 recv_buffer_size_ += _bytes;
517
518 bool message_is_empty(false);
519 bool found_message(false);
520
521 do {
522 found_message = false;
523 message_is_empty = false;
524
525 its_start = 0 + its_iteration_gap;
526 if (its_start + 3 < its_start) {
527 VSOMEIP_ERROR << "buffer overflow in local server endpoint ~> abort!";
528 return;
529 }
530 while (its_start + 3 < recv_buffer_size_ + its_iteration_gap &&
531 (recv_buffer_[its_start] != 0x67 ||
532 recv_buffer_[its_start+1] != 0x37 ||
533 recv_buffer_[its_start+2] != 0x6d ||
534 recv_buffer_[its_start+3] != 0x07)) {
535 its_start++;
536 }
537
538 if (its_start + 3 == recv_buffer_size_ + its_iteration_gap) {
539 message_is_empty = true;
540 } else {
541 its_start += 4;
542 }
543
544 if (!message_is_empty) {
545 if (its_start + protocol::COMMAND_POSITION_SIZE + 3 < recv_buffer_size_ + its_iteration_gap) {
546 its_command_size = VSOMEIP_BYTES_TO_LONG(
547 recv_buffer_[its_start + protocol::COMMAND_POSITION_SIZE+3],
548 recv_buffer_[its_start + protocol::COMMAND_POSITION_SIZE+2],
549 recv_buffer_[its_start + protocol::COMMAND_POSITION_SIZE+1],
550 recv_buffer_[its_start + protocol::COMMAND_POSITION_SIZE]);
551
552 its_end = its_start + protocol::COMMAND_POSITION_SIZE + 3 + its_command_size;
553 } else {
554 its_end = its_start;
555 }
556 if (its_command_size && max_message_size_ != MESSAGE_SIZE_UNLIMITED
557 && its_command_size > max_message_size_) {
558 std::lock_guard<std::mutex> its_lock(socket_mutex_);
559 VSOMEIP_ERROR << "Received a local message which exceeds "
560 << "maximum message size (" << std::dec << its_command_size
561 << ") aborting! local: " << get_path_local() << " remote: "
562 << get_path_remote();
563 recv_buffer_.resize(recv_buffer_size_initial_, 0x0);
564 recv_buffer_.shrink_to_fit();
565 return;
566 }
567 if (its_end + 3 < its_end) {
568 VSOMEIP_ERROR << "buffer overflow in local server endpoint ~> abort!";
569 return;
570 }
571 while (its_end + 3 < recv_buffer_size_ + its_iteration_gap &&
572 (recv_buffer_[its_end] != 0x07 ||
573 recv_buffer_[its_end+1] != 0x6d ||
574 recv_buffer_[its_end+2] != 0x37 ||
575 recv_buffer_[its_end+3] != 0x67)) {
576 its_end ++;
577 }
578 if (its_end + 4 < its_end) {
579 VSOMEIP_ERROR << "buffer overflow in local server endpoint ~> abort!";
580 return;
581 }
582 // check if we received a full message
583 if (recv_buffer_size_ + its_iteration_gap < its_end + 4
584 || recv_buffer_[its_end] != 0x07
585 || recv_buffer_[its_end+1] != 0x6d
586 || recv_buffer_[its_end+2] != 0x37
587 || recv_buffer_[its_end+3] != 0x67) {
588 // command (1 Byte) + version (2 Byte) client id (2 Byte)
589 // + command size (4 Byte) + data itself + stop tag (4 byte)
590 // = 11 Bytes not covered in command size.
591 // If need to change the recv_buffer_, change the value of missing_capacity_
592 // in this if/else, otherwise it is 0
593 if (its_start - its_iteration_gap + its_command_size
594 + protocol::COMMAND_HEADER_SIZE + protocol::TAG_SIZE > recv_buffer_size_) {
595 missing_capacity_ =
596 std::uint32_t(its_start) - std::uint32_t(its_iteration_gap)
597 + its_command_size + std::uint32_t(protocol::COMMAND_HEADER_SIZE + protocol::TAG_SIZE)
598 - std::uint32_t(recv_buffer_size_);
599 } else if (recv_buffer_size_ < protocol::COMMAND_HEADER_SIZE + protocol::TAG_SIZE) {
600 // to little data to read out the command size
601 // minimal amount of data needed to read out command size = 11
602 missing_capacity_ = static_cast<std::uint32_t>(
603 protocol::COMMAND_HEADER_SIZE + protocol::TAG_SIZE - recv_buffer_size_);
604 } else {
605 std::stringstream local_msg;
606 local_msg << std::setfill('0') << std::hex;
607 for (std::size_t i = its_iteration_gap;
608 i < recv_buffer_size_ + its_iteration_gap &&
609 i - its_iteration_gap < 32; i++) {
610 local_msg << std::setw(2) << (int) recv_buffer_[i] << " ";
611 }
612 VSOMEIP_ERROR << "lse::c<" << this
613 << ">rcb: recv_buffer_size is: " << std::dec
614 << recv_buffer_size_ << " but couldn't read "
615 "out command size. recv_buffer_capacity: "
616 << std::dec << recv_buffer_.capacity()
617 << " its_iteration_gap: " << std::dec
618 << its_iteration_gap << " bound client: 0x"
619 << std::hex << bound_client_ << " buffer: "
620 << local_msg.str();
621 recv_buffer_size_ = 0;
622 missing_capacity_ = 0;
623 its_iteration_gap = 0;
624 message_is_empty = true;
625 }
626 }
627 }
628
629 if (!message_is_empty &&
630 its_end + 3 < recv_buffer_size_ + its_iteration_gap) {
631
632 if (its_server->is_routing_endpoint_
633 && recv_buffer_[its_start] == protocol::id_e::ASSIGN_CLIENT_ID) {
634 client_t its_client = its_server->assign_client(
635 &recv_buffer_[its_start], uint32_t(its_end - its_start));
636 {
637 set_bound_client(its_client);
638 its_host->add_known_client(its_client, get_bound_client_host());
639 its_server->add_connection(its_client, shared_from_this());
640 }
641 its_server->send_client_identifier(its_client);
642 assigned_client_ = true;
643 } else if (!its_server->is_routing_endpoint_ || assigned_client_) {
644 boost::system::error_code ec;
645 auto its_endpoint = socket_.remote_endpoint(ec);
646 if (!ec) {
647 auto its_address = its_endpoint.address();
648 auto its_port = its_endpoint.port();
649
650 if (its_address.is_v4()) {
651 sec_client_.host
652 = htonl(uint32_t(its_address.to_v4().to_ulong()));
653 }
654 sec_client_.port = htons(its_port);
655 security::sync_client(&sec_client_);
656
657 its_host->on_message(&recv_buffer_[its_start],
658 uint32_t(its_end - its_start), its_server.get(),
659 false, bound_client_, &sec_client_,
660 its_address, its_port);
661 } else {
662 VSOMEIP_WARNING << std::hex << "Client 0x" << its_host->get_client()
663 << " endpoint encountered an error[" << ec.value() << "]: " << ec.message();
664 }
665 } else {
666 VSOMEIP_WARNING << std::hex << "Client 0x" << its_host->get_client()
667 << " didn't receive VSOMEIP_ASSIGN_CLIENT as first message";
668 }
669 #if 0
670 std::stringstream local_msg;
671 local_msg << "lse::c<" << this << ">rcb::thunk: ";
672 for (std::size_t i = its_start; i < its_end; i++)
673 local_msg << std::setw(2) << std::setfill('0') << std::hex
674 << (int) recv_buffer_[i] << " ";
675 VSOMEIP_INFO << local_msg.str();
676 #endif
677 calculate_shrink_count();
678 recv_buffer_size_ -= (its_end + 4 - its_iteration_gap);
679 missing_capacity_ = 0;
680 its_command_size = 0;
681 found_message = true;
682 its_iteration_gap = its_end + 4;
683 } else {
684 if (its_iteration_gap) {
685 // Message not complete and not in front of the buffer!
686 // Copy last part to front for consume in future receive_cbk call!
687 for (size_t i = 0; i < recv_buffer_size_; ++i) {
688 recv_buffer_[i] = recv_buffer_[i + its_iteration_gap];
689 }
690 // Still more capacity needed after shifting everything to front?
691 if (missing_capacity_ &&
692 missing_capacity_ <= recv_buffer_.capacity() - recv_buffer_size_) {
693 missing_capacity_ = 0;
694 }
695 } else if (message_is_empty) {
696 VSOMEIP_ERROR << "Received garbage data.";
697 is_error = true;
698 }
699 }
700 } while (recv_buffer_size_ > 0 && found_message);
701 }
702
703 if (is_stopped_
704 || _error == boost::asio::error::eof
705 || _error == boost::asio::error::connection_reset
706 || is_error) {
707 shutdown_and_close();
708 its_server->remove_connection(bound_client_);
709 policy_manager_impl::get()->remove_client_to_sec_client_mapping(bound_client_);
710 } else if (_error != boost::asio::error::bad_descriptor) {
711 start();
712 }

873bool local_tcp_server_endpoint_impl::check_packetizer_space(
874 target_data_iterator_type _it, message_buffer_ptr_t* _packetizer,
875 std::uint32_t _size) {
876
877 if ((*_packetizer)->size() + _size < (*_packetizer)->size()) {
878 VSOMEIP_ERROR << "Overflow in packetizer addition ~> abort sending!";
879 return false;
880 }
881 if ((*_packetizer)->size() + _size > max_message_size_
882 && !(*_packetizer)->empty()) {

884 _it->second.queue_size_ += (*_packetizer)->size();

886 }
887 return true;
